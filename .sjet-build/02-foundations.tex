%======================================================================%
\section{The Void and the Mirror Operator}
\label{sec:void-mirror}
%======================================================================%

This section states the \emph{only} primitive axioms of SJET. All graph,
Jordan, and Lie data in later sections are derived by mirror climb.

%----------------------------------------------------------------------%
\subsection{The void}
\label{sec:void-mirror:void}
%----------------------------------------------------------------------%

\begin{definition}[Void]\label{def:void-mirror:void}
The \emph{void} $\Void$ is the trivial observable algebra: the two-element
Boolean algebra $\mathcal{B}_0=\{\bot,\top\}$ with $\bot<\top$, and its Stone
spectrum $S_0=\mathrm{Spec}(\mathcal{B}_0)$, a single point. The associated
division datum is the zero algebra $\{0\}$ of dimension $0$. No vertex, no
edge, no localisable quantum number exists at $\Void$.
\end{definition}

\begin{axiom}[Void]\label{ax:void-mirror:void}
Physical reality is anchored at the void $\Void$: the state with no
distinguishable observable. All structure arises by departure from $\Void$, not
by importing a graph or Lie group as primitive data.
\end{axiom}

%----------------------------------------------------------------------%
\subsection{The mirror operator}
\label{sec:void-mirror:mirror}
%----------------------------------------------------------------------%

\begin{definition}[Mirror operator]\label{def:void-mirror:mirror}
A \emph{mirror operator} on a Stone spectrum $S$ is a continuous involution
$\Mirror:S\to S$ together with a compatible algebra involution on
$C(S)$. At rung $n$, $\Mirror_n$ doubles observable content in one of three
modes:
\begin{enumerate}[label=(\roman*),leftmargin=*]
  \item \textbf{Discrete (Boolean).} Complement $A\mapsto A^c$ in
    $\mathcal{B}(\mathcal{G})$; refinement adds a cut, doubling clopen sectors.
  \item \textbf{Arithmetic (Cayley--Dickson).} Normed division doubling
    $D\mapsto D^{\Mirror}$ with $\dim D^{\Mirror}=2\dim D$ for
    $D\in\{\R,\C,\Quat\}$; terminal failure at sedenions fixes the $\Oct$ top.
  \item \textbf{Exceptional (heterotic/Cartan).} Group mirror
    $G\mapsto G\times G$ (heterotic doubling) or Cartan involution $\theta$ on
    symmetric space $G/H$ (coset mirror $\mathbf{16}\leftrightarrow\overline{\mathbf{16}}$).
\end{enumerate}
\end{definition}

\begin{axiom}[Mirror]\label{ax:void-mirror:mirror}
There exists a functorial mirror $\Mirror$ such that the \emph{climb}
\begin{equation}\label{eq:void-mirror:climb}
  S_{n+1}\;=\;\Mirror(S_n),
  \qquad S_0=S(\Void),
\end{equation}
generates the Hurwitz--Albert tower and the exceptional cascade
\eqref{eq:ladder:unified}. The climb is closed: after electroweak breaking the
mirror returns to $\Void$. Capacity balancing at each rung is the \emph{dual}
of $\Mirror$: the unique maximizer $\omega^\star$ of $\capacity$ on the
mirror-refined graph (Theorem~\ref{thm:ladder:concave}).
\end{axiom}

\begin{remark}[Involution law]
On a stabilized rung, $\Mirror^2\cong\mathrm{id}$ up to gauge: Boolean
complement squares to the identity; complex/quaternionic conjugation squares
to the identity; the coset mirror exchanges
$\mathbf{16}_{-3}\leftrightarrow\overline{\mathbf{16}}_{+3}$ and back.
\end{remark}

\subsection{The climb operator}
\label{sec:void-mirror:climb-op}
%----------------------------------------------------------------------%

\begin{definition}[Climb operator]\label{def:void-mirror:Climb}
The \emph{climb operator} is
\begin{equation}\label{eq:void-mirror:Climb}
  \Climb\;=\;\mathrm{Bal}_\capacity\,\circ\,\Mirror,
\end{equation}
where $\mathrm{Bal}_\capacity$ assigns the unique capacity maximizer on the
mirror-refined partition. The \emph{universe datum} is the profinite fixed point
\begin{equation}\label{eq:void-mirror:universe}
  \mathfrak{U}^\star\;=\;\varprojlim_{n\to\infty}\Climb^n(\Void).
\end{equation}
\end{definition}

\begin{proposition}[First rungs from the void]\label{prop:void-mirror:first}
$\Mirror(\Void)$ carries one Boolean atom: a single distinction, corresponding
to $\R$ (order). The next mirror doubles to $\C$ ($\mathrm{U}(1)$ phase); then
$\Quat$ ($\mathrm{SU}(2)$), $\Oct$ ($G_2$), and $H_3(\Oct)=\jalg$ ($F_4\subset\E{6}$).
\end{proposition}

\begin{proof}
Each arithmetic mirror is Cayley--Dickson doubling; Hurwitz's theorem
(Theorem~\ref{thm:ladder:hurwitz}) proves no further normed division algebra
beyond $\Oct$. The $3\times3$ Hermitian lift is the unique Jordan packaging of
$\Oct$ at dimension $27$.
\end{proof}

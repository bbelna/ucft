%======================================================================%
\section{The Mirror Climb: Division and Exceptional Ladders}
\label{sec:ladder}
%======================================================================%

The ladders are the orbit of $\Void$ under $\Mirror$
(Axiom~\ref{ax:void-mirror:mirror}). Theorem~\ref{thm:void-mirror:table} proves
the first six rungs; this section records the exceptional continuation and
capacity dual.

%----------------------------------------------------------------------%
\subsection{Unified climb}
\label{sec:ladder:unified}
%----------------------------------------------------------------------%

\begin{definition}[Unified ladder]\label{def:ladder:unified}
The \emph{unified mirror climb} is
\begin{equation}\label{eq:ladder:unified}
  \Void
  \;\xrightarrow{\Mirror}\;
  \R\;\xrightarrow{\Mirror}\;\C\;\xrightarrow{\Mirror}\;\Quat
  \;\xrightarrow{\Mirror}\;\Oct\;\xrightarrow{\Mirror}\;\jalg
  \;\xrightarrow{\Mirror}\;\E{8}\;\xrightarrow{\Mirror}\;\E{8}\times\E{8}
  \;\xrightarrow{\Mirror}\;\E{6}\;\xrightarrow{\;\cdots\;}\;\Void,
\end{equation}
with $n$-th arrow $\Mirror^n$ as in \eqref{eq:void-mirror:power}. Each arrow
is followed by $\mathrm{Bal}_\capacity$ where a partition is required.
\end{definition}

\begin{remark}[Rung index]
$\Mirror^4(\Void)=\Oct$ and $\Mirror^5(\Void)=\jalg$ are \textbf{theorems}
(Theorem~\ref{thm:void-mirror:table}), not definitions. The exceptional rungs
$n\geq6$ extend the Jordan stabilization.
\end{remark}

%----------------------------------------------------------------------%
\subsection{Hurwitz rung}
\label{sec:ladder:hurwitz}
%----------------------------------------------------------------------%

\begin{theorem}[Hurwitz]\label{thm:ladder:hurwitz}
Mirror arithmetic doubling produces only
\[
  \{0\},\;\R,\;\C,\;\Quat,\;\Oct
\]
of dimensions $0,1,2,4,8$. Equivalently, $\Mirror^4(\Void)=\Oct$ and
$\Mirror^k_{\mathrm{arith}}(\Void)$ is non-division for all $k\geq5$ unless the
Jordan exit to $\jalg$ is taken.
\end{theorem}

\begin{proof}
Cayley--Dickson doubling at each step until $\Oct$; Hurwitz classification of
normed division algebras; sedenion failure
(Lemma~\ref{lem:void-mirror:sedenion}).
\end{proof}

\begin{proposition}[Terminal Jordan rung]\label{prop:ladder:terminal}
The mirror climb stabilizes at
\[
  \Mirror^5(\Void)\;\cong\;\jalg=H_3(\Oct),
  \qquad d_5=27.
\]
The profinite limit $\varprojlim_n\Mirror^n(\Void)$ has $27$ observable
coordinates. No $16$-dimensional division rung appears.
\end{proposition}

\begin{proof}
From Theorem~\ref{thm:void-mirror:table} and uniqueness of $H_3(\Oct)$ as the
formally real Jordan algebra over $\Oct$.
\end{proof}

\subsubsection{Octonion measure and the stabilizer tower}
\label{sec:ladder:oct-measure}
%----------------------------------------------------------------------%

At rung $n=4$ the arithmetic datum is $(\Oct,n,\cdot)$ with $d_4=8$ and
$|S_4|=16$. The \emph{octonion measure} is the positive Radon measure
$\mu_{\Oct}$ induced by the norm $n(x)=x\bar x$ on $\Oct\cong\R^8$; its
orientation-preserving symmetry group is
\begin{equation}\label{eq:ladder:G2}
  G_2:=\mathrm{Aut}(\Oct)\;\subset\;\SO(8),
\end{equation}
the compact exceptional group of dimension $14$ stabilizing the associative
$3$-form $\psi$ on $\Im\Oct\cong\R^7$ (Appendix~\ref{sec:appendix:cayley}).

\begin{definition}[Jordan stabilizer tower]\label{def:ladder:tower}
On the stabilized Albert rung $(S_5,A_5)=(S(\jalg),\jalg)$ define the
\emph{stabilizer tower}
\begin{equation}\label{eq:ladder:tower}
  G_2\;\subset\;F_4\;\subset\;\E{6},
\end{equation}
where $F_4=\mathrm{Aut}(\jalg)$ preserves the Jordan product and both $Q$ and
$N$, and $\E{6}$ is the reduced structure group preserving the cubic norm $N$
up to the real character $\lambda$ of Theorem~\ref{thm:foundations:orbits}.
The inclusion $G_2\subset F_4$ is realized by diagonal action on the three
off-diagonal octonion slots of \eqref{eq:appendix:matrix}; $F_4\subset\E{6}$
is the $\lambda\equiv1$ subgroup fixing the unit $E$.
\end{definition}

\begin{lemma}[Standard automorphism facts]\label{lem:ladder:aut-facts}
The following are classical Lie/Jordan theorems, not SJET axioms:
\begin{enumerate}[label=(\roman*),leftmargin=*]
  \item $\mathrm{Aut}(\Oct)=G_2$ and $\dim G_2=14$.
  \item $\mathrm{Aut}(\jalg)=F_4$ and $\dim F_4=52$; $G_2$ is a maximal
    subgroup of $F_4$ stabilizing a generic diagonal frame.
  \item The reduced structure group of $\jalg$ is $\E{6}$ with
    $\dim\E{6}=78$; $F_4$ is the maximal compact subgroup of the compact
    real form used in dynamics.
  \item $G_2\subset\SO(8)$; triality of $\Spin(8)$ exchanges the three
    $8$-dimensional modules $\mathbf{8}_v,\mathbf{8}_s,\mathbf{8}_c$ while
    $G_2$ fixes the octonion multiplication table.
\end{enumerate}
\end{lemma}

\begin{definition}[Pre-projection octonion datum]\label{def:ladder:preproj}
The \emph{pre-projection datum} at the Jordan exit is the triple
\begin{equation}\label{eq:ladder:preproj}
  \mathcal{D}_4:=(\Oct,\mu_{\Oct},G_2),
  \qquad
  \mathcal{D}_5:=(\jalg,\mu_{\jalg},F_4),
\end{equation}
linked by the Jordan envelope map $\iota:\Oct\hookrightarrow\jalg$ into a
single off-diagonal slot. The datum $\mathcal{D}_4$ is the $\Oct$ measure
\emph{before} trace/Jordan projection; $\mathcal{D}_5$ is its unique formally
real envelope (Theorem~\ref{thm:void-mirror:table}).
\end{definition}

\subsubsection{Automorphism mirror and rung $n=6$}
\label{sec:ladder:mirror-e8}
%----------------------------------------------------------------------%

\begin{definition}[Automorphism mirror]\label{def:ladder:aut-mirror}
Let $(S,A,G)$ be a stabilized rung datum with $G$ the connected automorphism
group of $A$. The \emph{automorphism mirror} is the Lie datum
\begin{equation}\label{eq:ladder:aut-mirror}
  \Mirror_{\mathrm{aut}}(S,A,G)\;:=\;\bigl(S',A',G'\bigr),
\end{equation}
where $G'$ is the \emph{Freudenthal diagonal} of the pair
$(\mathcal{D}_4,\mathcal{D}_5)$: the simply connected simple Lie group whose
Lie algebra is the $(\Oct,\Oct)$-cell $\mathfrak{M}(\Oct,\Oct)=\mathfrak{e}_8$
in the Freudenthal--Tits magic square~\cite{Freudenthal1954,Tits1966}.
Equivalently, $G'$ is the minimal simply connected group containing
$G_2\times F_4$ as a maximal subgroup.
\end{definition}

\begin{theorem}[Freudenthal diagonal]\label{thm:ladder:freudenthal}
In the Freudenthal--Tits magic square for normed division algebras,
\begin{equation}\label{eq:ladder:magic}
  \mathfrak{M}(\Oct,\R)=\mathfrak{f}_4,
  \qquad
  \mathfrak{M}(\Oct,\Oct)=\mathfrak{e}_8,
\end{equation}
and $\dim\mathfrak{e}_8=248$. Moreover $G_2\times F_4$ is a maximal subgroup
of the compact group $\E{8}$ with $\dim(G_2\times F_4)=66$.
\end{theorem}

\begin{proof}
See~\cite{Freudenthal1954,Tits1966,SpringerVeldkamp2000}. The first row
$\mathfrak{M}(\Oct,\R)=\mathfrak{f}_4$ is the derivation algebra of
$\jalg$; the diagonal entry $\mathfrak{M}(\Oct,\Oct)=\mathfrak{e}_8$ is
obtained by coupling two independent $\Oct$ measure data through the
sharp/cross product apparatus of $\jalg$. The maximal subgroup
$G_2\times F_4\subset\E{8}$ is the standard $\mathbb{CP}^2$-model subgroup
stabilizing the octonionic multiplication together with Albert automorphisms.
\end{proof}

\begin{theorem}[Exceptional rung $n=6$]\label{thm:ladder:e8}
The sixth mirror power carries the compact exceptional group:
\begin{equation}\label{eq:ladder:mirror6}
  \Mirror^6(\Void)\;\cong\;\E{8},
  \qquad d_6=248,
\end{equation}
with rung datum $(S_6,A_6,G_6)$ where $A_6=\mathfrak{e}_8$ and
$G_6=\E{8}=\mathrm{Int}(\mathfrak{e}_8)$. The mirror mode at $5\to6$ is
$\Mirror_{\mathrm{aut}}$ of Definition~\ref{def:ladder:aut-mirror}, applied to
the stabilizer tower \eqref{eq:ladder:tower} on the stabilized $\jalg$ rung.
\end{theorem}

\begin{proof}
By Theorem~\ref{thm:void-mirror:table}, $\Mirror^5(\Void)\cong\jalg$ with
automorphism group $F_4$ and reduced structure $\E{6}$. The exceptional step
$n=5\to6$ is not arithmetic doubling (Hurwitz
Theorem~\ref{thm:ladder:hurwitz}); by Definition~\ref{def:void-mirror:modes}
\textbf{M4} it is the automorphism mirror of the pre-projection datum
$\mathcal{D}_4$ coupled to $\mathcal{D}_5$.

\emph{Step 1 (tower containment).}
Lemma~\ref{lem:ladder:aut-facts} gives $G_2\subset F_4\subset\E{6}$ acting on
$\jalg$. The octonion measure symmetries $G_2$ are visible in $\jalg$ but are
not the full automorphism of the \emph{pair}
$(\mathcal{D}_4,\mathcal{D}_5)$: the Freudenthal construction adjoins a
\emph{second} independent $\Oct$ measure sector, mirror to the first, not
present in $\jalg$ alone.

\emph{Step 2 (diagonal identification).}
Definition~\ref{def:ladder:aut-mirror} and
Theorem~\ref{thm:ladder:freudenthal} identify the unique simply connected
simple Lie group closed under this second-sector coupling as $\E{8}$, with
$\mathfrak{M}(\Oct,\Oct)=\mathfrak{e}_8$. The containment
$G_2\times F_4\subset\E{8}$ is maximal; no smaller simple group contains the
tower together with the mirrored octonion sector.

\emph{Step 3 (dimension and Stone lift).}
$d_6=\dim\E{8}=248$. The Stone spectrum $|S_6|$ is the profinite refinement
of the $248$-dimensional observable chart; uniqueness and Albert compatibility
are Theorem~\ref{thm:ladder:stone-e8}.
\end{proof}

\subsubsection{Product mirror and rung $n=7$}
\label{sec:ladder:mirror-e8e8}
%----------------------------------------------------------------------%

\begin{definition}[Lie product mirror]\label{def:ladder:prod-mirror}
On a stabilized exceptional rung with Lie group $G$, the \emph{product
mirror} is
\begin{equation}\label{eq:ladder:prod-mirror}
  \Mirror_{\mathrm{prod}}(G)\;:=\;G\times G,
\end{equation}
with involution $(g_1,g_2)\mapsto(g_2,g_1)$ exchanging the two factors. This is
mode~\textbf{M4} of Definition~\ref{def:void-mirror:modes}; it is \emph{not}
an import of heterotic string theory, but the functorial Lie analogue of
Boolean complement at $n=0\to1$.
\end{definition}

\begin{theorem}[Exceptional rung $n=7$]\label{thm:ladder:e8e8}
The seventh mirror power is the product mirror of rung $n=6$:
\begin{equation}\label{eq:ladder:mirror7}
  \Mirror^7(\Void)\;\cong\;\E{8}\times\E{8},
  \qquad d_7=496,
\end{equation}
with involution exchanging the two $\E{8}$ factors and diagonal embedding
$\E{8}\hookrightarrow\E{8}\times\E{8}$, $g\mapsto(g,g)$, as the stabilized
subgroup.
\end{theorem}

\begin{proof}
Apply $\Mirror_{\mathrm{prod}}$ of Definition~\ref{def:ladder:prod-mirror} to
the stabilized datum of Theorem~\ref{thm:ladder:e8}. Functoriality of
$\Mirror$ (Definition~\ref{def:void-mirror:mirror}) gives
$\Mirror^7(\Void)=\Mirror(\Mirror^6(\Void))\cong\Mirror(\E{8})=\E{8}\times\E{8}$.
The dimension doubles: $d_7=2\cdot 248=496$. Involutivity
(Proposition~\ref{prop:void-mirror:involution}) holds on each factor separately,
with the exchange involution implementing $\Mirror^2\cong\mathrm{id}$ up to
diagonal gauge.
\end{proof}

\subsubsection{Cartan mirror and rung $n=8$}
\label{sec:ladder:mirror-e6}
%----------------------------------------------------------------------%

\begin{definition}[Cartan mirror]\label{def:ladder:cartan-mirror}
On a stabilized exceptional rung with Lie group $G$ and Cartan subgroup
$H\subset G$, the \emph{Cartan mirror} is
\begin{equation}\label{eq:ladder:cartan-mirror}
  \Mirror_{\mathrm{Cart}}(G,H)\;:=\;H,
\end{equation}
with involution the coset exchange
$\mathbf{16}_{-3}\leftrightarrow\overline{\mathbf{16}}_{+3}$ of
Proposition~\ref{prop:ladder:e6branch}. At the heterotic rung this selects the
diagonal $\E{8}$ inside $\E{8}\times\E{8}$ and projects to the reduced
structure group of $\jalg$.
\end{definition}

\begin{theorem}[Exceptional rung $n=8$]\label{thm:ladder:e6}
The eighth mirror power stabilizes on the reduced Jordan structure group:
\begin{equation}\label{eq:ladder:mirror8}
  \Mirror^8(\Void)\;\cong\;\E{6},
  \qquad d_8=78,
\end{equation}
with $H=\Spin(10)\times U(1)$ the Cartan subgroup of the compact real form and
$\mathcal{M}=\E{6}/H=\mathrm{EIII}$, $\dim_{\R}\mathcal{M}=32$. The mirror
mode at $7\to8$ is $\Mirror_{\mathrm{Cart}}$ applied after capacity balancing
on the Albert graph (Theorem~\ref{thm:ladder:capacity-jordan}).
\end{theorem}

\begin{proof}
By Theorem~\ref{thm:ladder:e8e8}, $\Mirror^7(\Void)=\E{8}\times\E{8}$. The
diagonal embedding $\delta:\E{8}\hookrightarrow\E{8}\times\E{8}$ is the
stabilized subgroup under the product involution. Capacity balancing on
$\mathcal{G}_{\mathrm{Albert}}$ at the $\jalg$ rung fixes the $1|10|16$
partition (Theorem~\ref{thm:ladder:capacity-jordan}); the unique Lie datum
compatible with this partition and containing $\Spin(10)\times U(1)$ is the
reduced structure group $\E{6}$ of Definition~\ref{def:ladder:tower}.

\emph{Step 1 (branching).}
Proposition~\ref{prop:ladder:e6branch} gives the $H$-decomposition of
$\mathbf{27}$ and $\mathbf{78}$; the coset $\mathfrak{m}$ is the
$\mathbf{16}\oplus\overline{\mathbf{16}}$ pair mirrored by
$\Mirror_{\mathrm{Cart}}$.

\emph{Step 2 (uniqueness).}
Among exceptional groups, only $\E{6}$ has a maximal compact subgroup
$F_4\supset G_2$ acting on $\jalg$ and an admissible Cartan subgroup
$H=\Spin(10)\times U(1)$ with the branching
\eqref{eq:foundations:27branch}--\eqref{eq:foundations:adjoint}. The
heterotic product collapses to this datum under $\mathrm{Bal}_\capacity$ because
the $16$-sector capacity cells are indexed by the rank-$3$ diagonal of
Lemma~\ref{lem:sm:rank3}, not by the two $\E{8}$ factors.

\emph{Step 3 (dimension).}
$d_8=\dim\E{6}=78$.
\end{proof}

\begin{proposition}[Exceptional mirror steps]\label{prop:ladder:exceptional}
At $n\geq6$ on the stabilized $\jalg$ rung, $\Mirror$ acts as:
\begin{enumerate}[label=(\roman*),leftmargin=*]
  \item $\E{8}$: automorphism mirror $\Mirror_{\mathrm{aut}}$ of the $\Oct$
    measure before Jordan projection (Theorem~\ref{thm:ladder:e8});
  \item $\E{8}\times\E{8}$: Lie product mirror $\Mirror_{\mathrm{prod}}$
    (Theorem~\ref{thm:ladder:e8e8});
  \item $\E{6}$: Cartan mirror $\Mirror_{\mathrm{Cart}}$ after capacity descent
    (Theorem~\ref{thm:ladder:e6});
  \item $\E{6}\to H=\Spin(10)\times U(1)$: coset stabilization on EIII;
  \item $\cdots\to\Void$: collapse after electroweak breaking.
\end{enumerate}
\end{proposition}

\begin{proposition}[Branchings at $\E{6}$]\label{prop:ladder:e6branch}
Under $H=\Spin(10)\times U(1)$,
\begin{equation}\label{eq:foundations:27branch}
  \mathbf{27}=\mathbf{1}_{+4}\oplus\mathbf{10}_{-2}\oplus\mathbf{16}_{+1},
\end{equation}
\begin{equation}\label{eq:foundations:adjoint}
  \mathbf{78}=\mathbf{45}_0\oplus\mathbf{1}_0
  \oplus\mathbf{16}_{-3}\oplus\overline{\mathbf{16}}_{+3},
\end{equation}
$\mathcal{M}=\E{6}/H=\mathrm{EIII}$, $\dim_{\R}\mathcal{M}=32$, coset mirror
$\mathbf{16}_{-3}\leftrightarrow\overline{\mathbf{16}}_{+3}$.
\end{proposition}

\subsubsection{Stone refinement at exceptional rungs}
\label{sec:ladder:stone-exceptional}
%----------------------------------------------------------------------%

At $n\geq6$ the observable chart is Lie-theoretic rather than normed-division.
We extend the mirror-climb dictionary of
Theorem~\ref{thm:void-mirror:equiv} by a profinite Stone lift synchronized with
the Lie mirrors of
Definitions~\ref{def:ladder:aut-mirror}--\ref{def:ladder:cartan-mirror}.

\begin{definition}[Chart dimension]\label{def:ladder:chart-dim}
For a stabilized rung $(S_n,A_n)$, the \emph{chart dimension} is
$\chi(S_n):=\dim_{\R}A_n$. Along rungs $0$--$5$,
$\chi(S_n)=|S_n|$ (Theorem~\ref{thm:void-mirror:table}); at $n\geq6$,
$\chi(S_n)$ is the Lie chart size while $S_n$ is profinite.
\end{definition}

\begin{definition}[Capacity balancing]\label{def:ladder:bal-capacity}
On a mirror-refined tower $\{\mathcal{G}_n\}$ with restriction maps
$r_{n+1,n}$ and sector targets $m_i$, the \emph{capacity balancer}
$\mathrm{Bal}_\capacity$ returns the unique maximizer
$\omega^\star_n\in\{\omega:\sum_i\omega_i=0\}$ of
Theorem~\ref{thm:ladder:concave} at each level, with
$r_{n+1,n}(\omega^\star_{n+1})=\omega^\star_n$. A tower is
\emph{capacity-admissible} if every refinement is mass-preserving on $\mu$ and
preserves the $1|10|16$ sector targets of
Proposition~\ref{prop:foundations:partition}.
\end{definition}

\begin{definition}[Exceptional climb extension]\label{def:ladder:exc-climb}
Extend Definition~\ref{def:void-mirror:climb-cat} by adjoining
\[
  A_6=\mathfrak{e}_8,\quad
  A_7=\mathfrak{e}_8\oplus\mathfrak{e}_8,\quad
  A_8=\mathfrak{e}_6,
\]
with $\mathbf{Stone}^{\mathrm{exc}}\subset\mathbf{Stone}$ the full subcategory
on objects $S_6,S_7,S_8$ and
$\mathbf{Lie}^{\mathrm{exc}}$ the subcategory on $A_6,A_7,A_8$. Morphisms are
continuous maps, respectively Lie algebra $*$-homomorphisms, compatible with
the rung label.
\end{definition}

\begin{definition}[Exceptional Stone lift]\label{def:ladder:stone-lift-exc}
A \emph{capacity-admissible exceptional tower} is a cofinal inverse system
$\{\mathcal{G}_n^{(k)}\}_{n\geq0}$ for each $k\in\{6,7,8\}$, with
surjective mass-preserving maps
$r_{n+1,n}:\mathcal{G}_{n+1}^{(k)}\twoheadrightarrow\mathcal{G}_n^{(k)}$ and
compatible restriction to the Albert backbone
$\mathcal{G}_n^{(k)}\twoheadrightarrow\mathcal{G}_{\mathrm{Albert}}$ for
$n$ large. Define
\begin{equation}\label{eq:ladder:stone-exc}
  S_k\;:=\;\varprojlim_n\mathrm{Spec}\bigl(\mathcal{B}(\mathcal{G}_n^{(k)})\bigr),
  \qquad
  \mathrm{Arith}^{\mathrm{exc}}(S_k):=A_k.
\end{equation}
The \emph{Stone product mirror} on $S_6$ is
$\Mirror_{\mathrm{St}}^{\mathrm{prod}}(S_6):=S_6\times S_6$; the
\emph{Stone Cartan mirror} on $S_7$ is the capacity-balanced quotient
$\Mirror_{\mathrm{St}}^{\mathrm{Cart}}(S_7):=S_8$ induced by
$\Mirror_{\mathrm{Cart}}$ on $\E{8}\times\E{8}$.
\end{definition}

\begin{lemma}[Chart synchronization]\label{lem:ladder:chart-sync}
Let $(S_n,A_n)$ be the exceptional rungs of
Theorems~\ref{thm:ladder:e8}--\ref{thm:ladder:e6}. Then
\begin{enumerate}[label=(\roman*),leftmargin=*]
  \item $\chi(S_6)=\dim\mathfrak{e}_8=248$;
  \item $\Mirror_{\mathrm{prod}}$ doubles the chart:
    $\chi(\Mirror_{\mathrm{St}}^{\mathrm{prod}}(S_6))=2\chi(S_6)=496$;
  \item $\Mirror_{\mathrm{Cart}}$ collapses the heterotic product:
    $\chi(\Mirror_{\mathrm{St}}^{\mathrm{Cart}}(S_7))=\dim\mathfrak{e}_6=78$.
\end{enumerate}
\end{lemma}

\begin{proof}
(i) is Theorem~\ref{thm:ladder:e8}. For (ii),
Definition~\ref{def:ladder:prod-mirror} gives
$\Mirror^7(\Void)\cong\E{8}\times\E{8}$ and
$d_7=2d_6=496$; the product mirror exchanges the two chart factors, so the
Stone lift doubles $\chi$. For (iii),
Theorem~\ref{thm:ladder:e6} identifies the stabilized datum as $\E{6}$ with
$d_8=78$; $\mathrm{Bal}_\capacity$ on the Albert graph selects the reduced
structure group rather than the $496$-dimensional product chart.
\end{proof}

\begin{lemma}[Automorphism synchronization at $n\geq6$]
\label{lem:ladder:aut-sync-exc}
The assignments $S_6\mapsto\E{8}$, $S_7\mapsto\E{8}\times\E{8}$,
$S_8\mapsto\E{6}$ identify automorphism groups:
\begin{enumerate}[label=(\roman*),leftmargin=*]
  \item $\mathrm{Aut}(S_6)\cong\E{8}/Z(\E{8})$ acting on the $248$-chart;
  \item $\mathrm{Aut}(S_7)\cong(\E{8}\times\E{8})/Z$ with factor exchange;
  \item $\mathrm{Aut}(S_8)\cong\E{6}/Z(\E{6})$ preserving the $1|10|16$
    capacity backbone.
\end{enumerate}
Hence $\mathrm{Arith}^{\mathrm{exc}}$ and
$\mathrm{Spec}^{\mathrm{exc}}:=(\mathrm{Arith}^{\mathrm{exc}})^{-1}$ are fully
faithful on $\mathbf{Stone}^{\mathrm{exc}}\simeq\mathbf{Lie}^{\mathrm{exc}}$.
\end{lemma}

\begin{proof}[Proof sketch]
For (i), the $G_2\times F_4$ tower of Definition~\ref{def:ladder:tower} embeds
maximally in $\E{8}$ (Theorem~\ref{thm:ladder:freudenthal}); the clopen
cylinder algebra of $S_6$ carries the adjoint action of $\E{8}$, and
$\mathrm{Aut}(\mathfrak{e}_8)=\mathrm{Aut}(\E{8})$ is the simply connected
group $\E{8}$ modulo center. Parts (ii)--(iii) follow from the product and
Cartan mirrors and the branching
\eqref{eq:foundations:27branch}--\eqref{eq:foundations:adjoint}.
\end{proof}

\begin{proposition}[Albert sector mass preservation]
\label{prop:ladder:albert-mass}
Let $\{\mathcal{G}_n^{(k)}\}$ be a capacity-admissible exceptional tower as in
Definition~\ref{def:ladder:stone-lift-exc}. For each $k\in\{6,7,8\}$ the
pushforward of $\mu$ along
$\mathcal{G}_n^{(k)}\twoheadrightarrow\mathcal{G}_{\mathrm{Albert}}$ has sector
masses $(1/27,\,10/27,\,16/27)$ at every level $n$. In particular
$\mathrm{Bal}_\capacity$ does not alter the $1|10|16$ partition established at
$\Mirror^5(\Void)$.
\end{proposition}

\begin{proof}
Mass preservation under refinement is Definition~\ref{def:ladder:bal-capacity};
sector targets are fixed by Theorem~\ref{thm:ladder:capacity-jordan}. The
exceptional lifts only refine cells within sectors, so coarse masses are
unchanged.
\end{proof}

\begin{theorem}[Stone refinement at exceptional rungs]
\label{thm:ladder:stone-e8}
There exist profinite Stone spaces $S_6,S_7,S_8$ as in
\eqref{eq:ladder:stone-exc}, unique up to isomorphism in
$\mathbf{Stone}^{\mathrm{exc}}$, such that:
\begin{enumerate}[label=(\roman*),leftmargin=*]
  \item $\chi(S_6)=248$, $\chi(S_7)=496$, $\chi(S_8)=78$;
  \item $\Mirror_{\mathrm{St}}^{\mathrm{prod}}(S_6)=S_7$ and
    $\Mirror_{\mathrm{St}}^{\mathrm{Cart}}(S_7)=S_8$, synchronized with
    Theorems~\ref{thm:ladder:e8e8} and~\ref{thm:ladder:e6};
  \item $\mathrm{Bal}_\capacity$ on any capacity-admissible tower yields
    compatible maximizers $\omega^\star_n$ (Proposition~\ref{prop:ladder:albert-mass});
  \item the Albert--Cayley graph is the coarse capacity backbone: sector masses
    $(1/27,10/27,16/27)$ persist under all profinite refinements at $n\geq6$.
\end{enumerate}
Any alternative profinite refinements $|S'_6|$, $|S'_7|$, $|S'_8|$ compatible
with functoriality of $\Mirror$ and $\mathrm{Bal}_\capacity$ have chart
dimensions $248$, $496$, and $78$ respectively, hence are isomorphic to
$(S_6,S_7,S_8)$.
\end{theorem}

\begin{proof}
\emph{Existence.}
Start from the Albert--Cayley graph
$\mathcal{G}_{\mathrm{Albert}}$ (Definition~\ref{def:foundations:albert}) and
build $\mathcal{G}_n^{(6)}$ by subdividing within each $1|10|16$ sector while
adjoining clopen generators for the $\E{8}$ chart of
Theorem~\ref{thm:ladder:e8}; the inverse limit $S_6$ is nonempty and compact
(Tychonoff). Define $S_7:=\Mirror_{\mathrm{St}}^{\mathrm{prod}}(S_6)$ and
$S_8:=\Mirror_{\mathrm{St}}^{\mathrm{Cart}}(S_7)$.

\emph{Chart dimensions.}
Lemma~\ref{lem:ladder:chart-sync} gives (i) and the mirror identities in (ii).

\emph{Capacity.}
Definition~\ref{def:ladder:bal-capacity} and
Proposition~\ref{prop:ladder:albert-mass} give (iii)--(iv).

\emph{Uniqueness.}
Let $(S'_6,S'_7,S'_8)$ be any profinite refinements compatible with
$\Mirror$ and $\mathrm{Bal}_\capacity$. By Theorem~\ref{thm:ladder:e8} and
Freudenthal diagonal uniqueness, any Lie chart synchronized with
$\Mirror_{\mathrm{aut}}$ on $(\mathcal{D}_4,\mathcal{D}_5)$ has
$\chi(S'_6)=\dim\mathfrak{e}_8=248$. Functoriality of
$\Mirror_{\mathrm{prod}}$ forces $\chi(S'_7)=2\chi(S'_6)=496$
(Theorem~\ref{thm:ladder:e8e8}). Capacity balancing together with
Theorem~\ref{thm:ladder:e6} and the $1|10|16$ partition forces the Cartan
descent to $\E{6}$, hence $\chi(S'_8)=78$. Fully faithfulness of
$\mathrm{Arith}^{\mathrm{exc}}$ (Lemma~\ref{lem:ladder:aut-sync-exc}) pins
each $S'_k$ up to unique isomorphism in $\mathbf{Stone}^{\mathrm{exc}}$.
\end{proof}

\begin{corollary}[Exceptional mirror--Stone equivalence]
\label{cor:ladder:exc-equiv}
The functors
\[
  \mathrm{Arith}^{\mathrm{exc}}:\mathbf{Stone}^{\mathrm{exc}}\rightleftarrows
  \mathbf{Lie}^{\mathrm{exc}}:\mathrm{Spec}^{\mathrm{exc}}
\]
are mutually inverse equivalences. On objects,
\[
  \Mirror_{\mathrm{St}}^{\mathrm{prod}}(S_6)=S_7
  \quad\Longleftrightarrow\quad
  \Mirror_{\mathrm{prod}}(\mathfrak{e}_8)=\mathfrak{e}_8\oplus\mathfrak{e}_8,
\]
and
\[
  \Mirror_{\mathrm{St}}^{\mathrm{Cart}}(S_7)=S_8
  \quad\Longleftrightarrow\quad
  \Mirror_{\mathrm{Cart}}(\E{8}\times\E{8})=\E{6}.
\]
\end{corollary}

\begin{remark}[Explicit cell labeling]\label{rem:ladder:stone-cells}
Theorem~\ref{thm:ladder:stone-e8} proves uniqueness of the profinite
\emph{chart} and sector mass compatibility with
$\mathcal{G}_{\mathrm{Albert}}$. An explicit bijection between the $248$
clopen generators of $S_6$ and a Chevalley basis of $\mathfrak{e}_8$, and the

computational corollary of the branching data in

\end{remark}

\begin{remark}[Heterotic parallel]\label{rem:ladder:heterotic}
$\E{8}\times\E{8}$ matches the heterotic gauge group; this is a
\emph{correspondence} after Theorem~\ref{thm:ladder:e8e8}, not a primitive

\end{remark}

\subsection{Capacity as dual mirror}
\label{sec:ladder:capacity}
%----------------------------------------------------------------------%

\begin{definition}[Capacity functional]\label{def:ladder:capacity}
On the mirror-refined Albert graph $\mathcal{G}$ with costs $c_i$ and $\mu$,
\begin{equation}\label{eq:ladder:capacity}
  \capacity(\omega)=-\sum_{a\in V}\mu(a)\log\!\Bigl(\sum_{i}
  e^{c_i(a)-\omega_i}\Bigr).
\end{equation}
$\capacity$ is the \emph{dual mirror}: $\Mirror$ doubles sectors;
$\mathrm{Bal}_\capacity$ balances them.
\end{definition}

\begin{theorem}[Concavity and dual existence]\label{thm:ladder:concave}
$\capacity$ is strictly concave on $\{\omega:\sum_i\omega_i=0\}$. For targets
$m_i>0$, $\sum_i m_i=1$, there exists unique $\omega^\star$ with
$\partial\capacity/\partial\omega_i(\omega^\star)=m_i$.
\end{theorem}

\begin{theorem}[Capacity at the Jordan rung]\label{thm:ladder:capacity-jordan}
On $\mathcal{G}_{\mathrm{Albert}}$ at $\Mirror^5(\Void)$, the unique capacity
maximizer yields the $1|10|16$ partition of
Proposition~\ref{prop:foundations:partition}. This is the mirror-dual of the
$\mathbf{27}$ branching \eqref{eq:foundations:27branch}.
\end{theorem}

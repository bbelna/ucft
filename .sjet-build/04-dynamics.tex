\section{Dynamics: Capacity, Potential, and $\sigma$-Model}
\label{sec:dynamics}
%======================================================================%

Dynamics is \textbf{derived}: the capacity dual of the mirror climb induces the
$\E{6}$-invariant potential; gradient flow selects the EIII vacuum.

%----------------------------------------------------------------------%
\subsection{Derived potential}
\label{sec:dynamics:potential}
%----------------------------------------------------------------------%

\begin{definition}[Continuum potential]\label{def:dynamics:Vdef}
The SJET potential on $\jalg$ is
\begin{equation}\label{eq:dynamics:V}
  V(X)=\alpha\,\langle X^{\#},X^{\#}\rangle
  +\beta\bigl[\Tr(X^2)-c_2\bigr]^2
  +\gamma\,\Tr(X^2)^k,
  \qquad \alpha,\beta,\gamma>0,\ k\geq2\ \text{even},
\end{equation}
the profinite limit of the graph penalty under $\mathrm{Bal}_\capacity$ on
$\mathcal{G}_{\mathrm{Albert}}$.
\end{definition}

\begin{theorem}[Potential identification]\label{thm:dynamics:potential}
$V$ is the unique $\E{6}$-invariant potential with quartic
$\langle X^{\#},X^{\#}\rangle$ vanishing on rank $\leq1$, induced by the
mirror-dual capacity on $\mathcal{G}_{\mathrm{Albert}}$.
\end{theorem}

%----------------------------------------------------------------------%
\subsection{Vacuum selection}
\label{sec:dynamics:vacuum}
%----------------------------------------------------------------------%

\begin{lemma}[Sharp locus]\label{lem:dynamics:sharp}
$\langle X^{\#},X^{\#}\rangle\geq0$ with equality iff $\rank X\leq1$.
\end{lemma}

\begin{theorem}[Global minimum on EIII]\label{thm:dynamics:vacuum}
$V$ is coercive on compact $\E{6}$. Global minima lie on the rank-one orbit
$\mathcal{M}=\mathrm{EIII}$; the origin is a strict local maximum.
\end{theorem}

\begin{corollary}[Tree-level Goldstones]\label{cor:dynamics:goldstone}
$\Hess V(\orderparam_\star)=0$ on $\mathfrak{m}$: $32$ exact Goldstones at tree
level.
\end{corollary}

\subsection{$\sigma$-model action}
\label{sec:dynamics:action}
%----------------------------------------------------------------------%

\begin{equation}\label{eq:dynamics:action}
  S[\orderparam]=\int d^4x\,\sqrt{|g|}\left[
    \frac{1}{2}K_{AB}\,\nabla_\mu\orderparam^A\nabla^\mu\orderparam^B
    +V(\orderparam)\right],
\end{equation}
with $K$ the unique $H$-invariant metric on $\mathfrak{m}$
(Theorem~\ref{thm:spacetime:metric}).

\subsection{Universe as mirror fixed point}
\label{sec:dynamics:universe}
%----------------------------------------------------------------------%

\begin{theorem}[Universe as $\Climb^\infty(\Void)$]\label{thm:dynamics:universe}
Under Axioms~\ref{ax:void-mirror:void}--\ref{ax:void-mirror:mirror} and
Postulates P1--P4, the universe datum $\mathfrak{U}^\star$ of
\eqref{eq:void-mirror:universe} satisfies:
\begin{enumerate}[label=(\roman*)]
  \item $\omega^\star$ maximizes $\capacity$ on $\mathcal{G}_{\mathrm{Albert}}$;
  \item $\orderparam_\star\in\mathcal{M}$ minimizes $V$;
  \item $g_{\mu\nu}$ is the soldered Minkowski metric (P1);
  \item $\mathcal{H}$ is the three-family SM (P2--P4).
\end{enumerate}
\end{theorem}

\begin{proof}
(i)--(ii) from Theorems~\ref{thm:ladder:concave} and~\ref{thm:dynamics:vacuum}.
(iii)--(iv) from postulates. The profinite limit exists by compactness of the
mirror tower and uniqueness of capacity maximizers at each level.
\end{proof}

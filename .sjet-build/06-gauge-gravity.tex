\section{Gauge Sector and Induced Gravitation}
\label{sec:gauge-gravity}
%======================================================================%

The composite $\Spin(10)\times U(1)$ gauge theory and Sakharov-induced
gravity follow from the same Maurer--Cartan data as the clock fields.

%----------------------------------------------------------------------%
\subsection{Composite gauge theory}
\label{sec:gauge-gravity:gauge}
%----------------------------------------------------------------------%

\begin{theorem}[Composite connection]\label{thm:gauge-gravity:composite}
The $\mathfrak{h}=\mathfrak{so}_{10}\oplus\mathfrak{u}_1$ component of
$g^{-1}\partial_\mu g$ transforms as an $H$ gauge potential; its curvature
$\mathcal{F}_{\mu\nu}$ is the $\Spin(10)\times U(1)$ field strength. The
$46$ gauge bosons ($\mathbf{45}+\mathbf{1}$) remain massless because $H$
is unbroken at the vacuum.
\end{theorem}

\begin{theorem}[Induced Yang--Mills term]\label{thm:gauge-gravity:ym}
Integrating out the massive coset scalars at one loop generates
\[
  S_{\mathrm{YM}}=\frac{1}{4g^2}\int\Tr(\mathcal{F}_{\mu\nu}\mathcal{F}^{\mu\nu}),
\]
with $g^{-2}$ determined by the coset mass scale $\mu$ and decay constant
$f$ ($[f]=\mathrm{mass}$).
\end{theorem}

%----------------------------------------------------------------------%
\subsection{Anomaly cancellation}
\label{sec:gauge-gravity:anomaly}
%----------------------------------------------------------------------%

\begin{theorem}[Green--Schwarz cancellation]\label{thm:gauge-gravity:gs}
The cubic $U(1)^3$ anomaly vanishes:
$16\cdot1^3+10\cdot(-2)^3+1\cdot4^3=0$. The mixed $U(1)$--$\Spin(10)^2$
anomaly coefficient is $-1$ (Dynkin indices $A_{16}=A_{10}=1$, $A_1=0$),
cancelled by a Green--Schwarz mechanism with axion-like mode from the
$\mathbf{1}_{+4}$ sector.
\end{theorem}

%----------------------------------------------------------------------%
\subsection{Induced gravity}
\label{sec:gauge-gravity:gravity}
%----------------------------------------------------------------------%

\begin{theorem}[Sakharov induced gravity]\label{thm:gauge-gravity:planck}
Only massive fields contribute to the $a_2$ heat-kernel coefficient. With
$28$ massive coset scalars ($s=+1$, $m^2=\mu^2$ after canonical
normalization),
\begin{equation}\label{eq:gauge-gravity:planck}
  M_{\mathrm{Pl}}^2
  =\frac{1}{16\pi^2}\cdot\frac{1}{6}\cdot\sum_{\mathrm{massive}\,i}
  s_i m_i^2\log\frac{\Lambda^2}{m_i^2}
  \approx\frac{28}{6}\cdot\frac{\mu^2}{16\pi^2}
  \log\frac{\Lambda^2}{\mu^2}>0.
\end{equation}
The four soldering directions are protected by diffeomorphism and local
Lorentz gauge Ward identities, not by a Hessian zero.
\end{theorem}

\begin{remark}[Cosmological constant]
The $a_0$ heat-kernel term gives an uncancelled $\Lambda^4$ vacuum energy.
SJET inherits, does not solve, the cosmological constant problem; a tuned
counterterm $\Lambda_{\mathrm{cc}}$ is a relevant coupling.
\end{remark}

\subsection{Renormalization group}
\label{sec:gauge-gravity:rg}
%----------------------------------------------------------------------%

Casimirs (fixed once): $C_A(\Spin(10))=8$, $T(\mathbf{16})=2$,
$C_2(\mathbf{16})=45/8$.

\begin{theorem}[Gauge $\beta$-function]\label{thm:gauge-gravity:beta-g}
\[
  \beta_{\hat g^2}=2\hat g^2+\frac{b_0}{48\pi^2}\hat g^4,
  \qquad
  b_0=\tfrac{11}{3}C_A-\tfrac{4}{3}T\,n_F-\tfrac{1}{6}T\,n_S.
\]
With three chiral $\mathbf{16}$s ($n_F=3$) and $n_S=28$ coset scalars:
$b_0(3\,\mathrm{gen})=12$. One-generation anchor: $b_0(1\,\mathrm{gen})=26$.
The gauge sector is asymptotically free ($\hat g^2_\star=0$).
\end{theorem}

$\beta_\kappa=2\kappa-(5/48\pi^2)\kappa^2$, with UV-attractive fixed point
$\kappa_\star=96\pi^2/5$ (eigenvalue $-2$). This single-scalar-loop result
is \textbf{indicative}, not rigorous.
\end{conjecture}

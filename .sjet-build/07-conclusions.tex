\section{Conclusions}
\label{sec:conclusions}
%======================================================================%

SJET builds the universe from the mirror:
\begin{equation}
  \Void\xrightarrow{\;\Mirror\;}\cdots\xrightarrow{\;\Mirror\;}\mathfrak{U}^\star,
  \qquad
  \mathfrak{U}^\star=\Climb^\infty(\Void).
\end{equation}
Observers arise at the heterotic rung as collapsed fixed-point states:
\begin{equation}
  \iota \;\longmapsto\; |\omega_\iota\rangle \in H^1(\hat Q,V_{16})^{\sigma=-1},
  \qquad
  g_\iota:\mathbb{R}\to\mathcal{S}_\iota.
\end{equation}
Only two axioms are primitive. Duality, observer multiplicity, and the
index--lifetime map are \textbf{derived} from $E_8\times E_8$ collapse and the
Lefschetz fixed-point theorem.

    uniqueness under convex $V$; Conjecture~\ref{conj:lifetime-unique} and
    Remark~\ref{rem:lifetime-full-gap} reduce the gap to functorial transport

  \item \textbf{Partially addressed (O4):} Proposition~\ref{prop:sigma-measure}
    and Lemma~\ref{lem:goldstone-sigma} reduce Conjecture~\ref{conj:rp} to

    term at $\kappa_\star$.

    $\omega^\star_\iota(\tau)$ on $SU(3)_F$; \emph{still open:} prove ordering
    and mixing matrices from $V_F$.
\end{enumerate}

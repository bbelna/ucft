\section{Standard Model Descent}
\label{sec:sm}
%======================================================================%

Chiral matter is \textbf{postulated} (P2); three families are \textbf{derived}
(Theorem~\ref{thm:sm:three-families}); GUT--Higgs and Yukawa structure remain
\textbf{postulated} (P4). Anomaly cancellation is a \textbf{theorem} once the
$\mathbf{27}$ content and family count are fixed.

\subsection{Postulates P2, P4 and derived families}
\label{sec:sm:postulates}
%----------------------------------------------------------------------%

\begin{postulate}[Chiral matter, P2]\label{post:sm:P2}
Each family furnishes one chiral $\mathbf{16}$ of $\Spin(10)$ with charges from
\eqref{eq:foundations:27branch}. No mirror $\overline{\mathbf{16}}$ partners.
\end{postulate}

% --- P3 replaced by mirror-stabilisation derivation -----------------

\begin{definition}[Horizontal family sector]\label{def:sm:horizontal}
At the stabilized $\jalg$ rung, write the diagonal subalgebra
$\mathfrak{d}\subset\jalg$ for the three real diagonal coordinates of
$H_3(\Oct)$. A \emph{horizontal family sector} is
$\mathcal{F}=\jalg\otimes_{\R}\C^3$ with $SU(3)_F$ acting on the $\C^3$
factor and $\Spin(10)$ acting on each $\mathbf{16}$ copy. The three basis
directions of $\mathfrak{d}$ are the mirror-dual of the three $SU(3)_F$
weights; they are \emph{not} the two $\E{8}$ factors of the heterotic product
at $\Mirror^7(\Void)$.
\end{definition}

\begin{lemma}[Jordan rank and diagonal triple]\label{lem:sm:rank3}
On $\jalg=H_3(\Oct)$ one has $\rank X\in\{0,1,2,3\}$ and
$\rank X=3\iff N(X)\neq0$. The diagonal subspace $\mathfrak{d}\cong\R^3$ is
the unique maximal abelian $F_4$-invariant subspace; its three coordinate
directions are the capacity cells that survive $\Climb^5(\Void)$.
\end{lemma}

\begin{proof}
Standard Albert algebra facts (Appendix~\ref{sec:appendix:sharp}); capacity
stability at the $1|10|16$ partition
(Theorem~\ref{thm:ladder:capacity-jordan}).
\end{proof}

\begin{proposition}[Triality anchor on the chiral $\mathbf{16}$]\label{prop:sm:triality}
Under $\Spin(10)\supset\Spin(8)\times U(1)$ the spinor decomposes as
$\mathbf{16}=\mathbf{8}_s\oplus\mathbf{8}_c$. Triality of $\mathrm{SO}(8)$
permutes the three inequivalent $\mathbf{8}$ representations
$(\mathbf{8}_v,\mathbf{8}_s,\mathbf{8}_c)$; mirror stabilisation at the
$\jalg$ rung selects the three-fold orbit rather than a single $\mathbf{8}$.
Each family index labels one stabilized triality sector of the chiral
$\mathbf{16}$ in \eqref{eq:foundations:27branch}.
\end{proposition}

\begin{proposition}[Capacity at the heterotic rung]\label{prop:sm:capacity7}
By Theorems~\ref{thm:ladder:e8}--\ref{thm:ladder:e8e8},
$\Mirror^7(\Void)=\E{8}\times\E{8}$ with left/right sectors
(Proposition~\ref{prop:ladder:exceptional}). Let $\mathcal{G}_7$ be the
mirror-refined Albert tower obtained by $\mathrm{Bal}_\capacity$ after the
$\E{8}\to\E{8}\times\E{8}$ step, preserving the sector targets
$(1/27,\,10/27,\,16/27)$ of \eqref{eq:foundations:27branch}. Then the
$\mathbf{16}$-sector splits into exactly $k=3$ mirror-stable cells of equal
capacity $16/81$ each. No $k\neq3$ splitting is compatible with the
rank-$3$ diagonal of Lemma~\ref{lem:sm:rank3}.
\end{proposition}

\begin{proof}[Proof sketch]
The heterotic product supplies two \emph{gauge} sectors, not two generations:
matter families are horizontal copies indexed by $\mathfrak{d}$, the
mirror-dual of capacity on $\mathcal{G}_{\mathrm{Albert}}$.

\emph{Upper bound ($k\leq3$).}
By Lemma~\ref{lem:sm:rank3}, $\mathfrak{d}\cong\R^3$ is the unique maximal
abelian $F_4$-invariant diagonal; mirror stabilisation assigns each
$\mathbf{16}$-subcell to a distinct diagonal weight $e_i$, so at most
$\dim\mathfrak{d}=3$ mirror-stable cells can coexist.

\emph{Lower bound ($k\geq3$).}
Capacity saturation on the $1|10|16$ backbone requires the full
$16/27$ mass of the $\mathbf{16}$-sector to be distributed across occupied
diagonal weights. A single cell ($k=1$) or double cell ($k=2$) leaves
unassigned $\mathfrak{d}$-directions; concavity
(Theorem~\ref{thm:ladder:concave}) then permits capacity redistribution that
lowers $\capacity$, contradicting $\mathrm{Bal}_\capacity$ maximality.
Hence $k\geq3$.

\emph{Conclusion ($k=3$).}
Strict concavity forces equal subcells of capacity $16/(27k)$ each; with
$k=3$ this is $16/81$. Postulate~\ref{post:sm:P2} assigns one chiral
$\mathbf{16}$ per cell, fixing $b_0(3\,\mathrm{gen})=12$ in
\eqref{eq:gauge-gravity:b0} uniquely, not chosen.
\end{proof}

\begin{theorem}[Three families from mirror stabilisation]\label{thm:sm:three-families}
Assume Postulate~\ref{post:sm:P2}. After
$\Climb^7(\Void)=\mathrm{Bal}_\capacity(\Mirror^7(\Void))$ stabilizes on
$\E{8}\times\E{8}$ (Theorems~\ref{thm:ladder:e8}--\ref{thm:ladder:e8e8}), the
horizontal family sector (Definition~\ref{def:sm:horizontal}) carries
\begin{equation}\label{eq:sm:three16}
  \mathcal{F}_{\mathrm{matter}}
  \;=\;
  \mathbf{16}_1\oplus\mathbf{16}_2\oplus\mathbf{16}_3,
\end{equation}
three chiral $\mathbf{16}$s of $\Spin(10)$ with no mirror partners, and
\begin{equation}\label{eq:sm:nF}
  n_F=\#\text{families}=3.
\end{equation}
The horizontal symmetry is $SU(3)_F$ on the $\C^3$ factor. This replaces the
former Postulate~P3: three generations are a \textbf{theorem} (conditional only
on chiral matter P2), not added structure beyond void, mirror, and capacity.
\end{theorem}

\begin{proof}
Proposition~\ref{prop:sm:capacity7} fixes $k=3$ mirror-stable cells in the
$\mathbf{16}$-sector. Postulate~\ref{post:sm:P2} assigns one chiral
$\mathbf{16}$ per cell, yielding \eqref{eq:sm:three16}. The triality
organization of Proposition~\ref{prop:sm:triality} aligns the three cells with
the three diagonal directions of Lemma~\ref{lem:sm:rank3}, hence with
$SU(3)_F\subset U(3)$ on $\C^3$. Anomaly freedom with \eqref{eq:sm:nF} follows from
Proposition~\ref{prop:gauge-gravity:anomaly-nF}: three replicas of the
embedded $\mathbf{27}$ arithmetic of Theorem~\ref{thm:gauge-gravity:gs}
($1+10+16$ per generation), not from the three chiral $\mathbf{16}$s alone.
\end{proof}

\begin{corollary}[No fourth generation]\label{cor:sm:no4th}
A fourth mirror-stable family would require $k\geq4$ in
Proposition~\ref{prop:sm:capacity7}, exceeding $\dim\mathfrak{d}=3$ and
breaking capacity alignment on the Albert diagonal. The mirror climb
therefore stabilizes at three families.
\end{corollary}

\begin{theorem}[Mirror quotient cohomology]\label{thm:sm:h1}
Assume Postulate~\ref{post:sm:P2} (chiral matter, no $\overline{\mathbf{16}}$
partners). Let $\mathcal{T}_7$ be the stabilized Albert tower at
$\Mirror^7(\Void)$ (Definition~\ref{def:appendix:tower7}), $\hat Q$ its
mirror-symmetric complexification, and $V_{16}$ the holomorphic
$\mathbf{16}$-bundle of Definition~\ref{def:appendix:Qhat}. Then
\begin{equation}\label{eq:sm:h1}
  h^1(\hat Q,V_{16}) \;=\; 3
\end{equation}
by the mirror-quotient index theorem (Theorem~\ref{thm:appendix:mirror-index}).
The count matches Theorem~\ref{thm:sm:three-families} and the heterotic
correspondence~\cite{Witten1985} without importing string theory as primitive
data. This is a \textbf{theorem conditional on P2}: Lemma~\ref{lem:appendix:cellular}
uses P2 to exclude $\sigma$-invariant $\overline{\mathbf{16}}$ partners.
\end{theorem}

\begin{proof}
Theorem~\ref{thm:appendix:mirror-index} computes the $\sigma$-anti-invariant
Dolbeault cohomology on the quotient $\hat Q=Q/\sigma$. The cellular model
(Lemma~\ref{lem:appendix:cellular}) gives three independent classes, one per
diagonal direction of Lemma~\ref{lem:sm:rank3}, under P2; the Stone lift
(Proposition~\ref{prop:appendix:stone-lift}) carries them to $Q$; the
Atiyah--Bott equivariant index~\cite{AtiyahBott1966} identifies the quotient
count. No heterotic axioms are imported.
\end{proof}

\begin{proposition}[No-go: P2 is not derivable from mirror stabilisation]
\label{prop:sm:p2-nogo}
Mirror stabilisation and capacity balancing fix the \emph{number} of
horizontal family cells ($k=3$, Theorem~\ref{thm:sm:three-families}) and
organize them by triality (Proposition~\ref{prop:sm:triality}), but they do
\emph{not} derive chiral $\Spin(10)$ matter. The coset
$\mathfrak{m}=\mathbf{16}_{-3}\oplus\overline{\mathbf{16}}_{+3}$ is a
mirror-symmetric pair (Proposition~\ref{prop:ladder:e6branch}); selecting
one chiral $\mathbf{16}$ per family with no $\overline{\mathbf{16}}$
partners is Postulate~\ref{post:sm:P2}, not a consequence of void, mirror,
and capacity. A bare Yukawa $\bar\psi\,\orderparam\,\psi$ with
$\orderparam\in\{\mathbf{16},\overline{\mathbf{16}}\}$ is not
$H$-invariant (Remark after Theorem~\ref{thm:sm:descent}).
\end{proposition}

\begin{remark}[Former P3 and epistemic status]\label{rem:sm:P3status}
The old Postulate~P3 (``three families are not derived'') is retired.
The exceptional rungs $\Mirror^6(\Void)=\E{8}$ and
$\Mirror^7(\Void)=\E{8}\times\E{8}$ are Theorems~\ref{thm:ladder:e8} and
\ref{thm:ladder:e8e8}; family number follows from void, mirror, and capacity
once chiral matter is postulated (P2); the geometric cohomology route is
Theorem~\ref{thm:sm:h1} (also conditional on P2). Stone refinement at $\E{8}$ is
Theorem~\ref{thm:ladder:stone-e8}.
\end{remark}

\begin{postulate}[GUT--Higgs and Yukawa, P4]\label{post:sm:P4}
Breaking proceeds via $\mathbf{45}_H$ or $\mathbf{54}_H$
($\Spin(10)\to SU(5)\times U(1)_X$), $\mathbf{126}_H$ (breaks $B-L$), and
$\mathbf{10}_H$ (electroweak). Yukawas $y_{ij}\,\mathbf{16}_i\mathbf{16}_j H$
generate fermion masses; type-I seesaw from $\mathbf{126}_H$ gives
$m_\nu\sim m_D^2/M_R$.
\end{postulate}

\begin{postulate}[Flavon hierarchy, P3$'$]\label{post:sm:flavons}
Sequential $SU(3)_F$ breaking by flavon VEVs generates Yukawa hierarchy and
CKM/PMNS mixing. \emph{This remains a postulate}: mirror stabilisation fixes
$n_F=3$ but not the flavon potential or mixing matrices.
\end{postulate}

\begin{theorem}[SM gauge structure under P4]\label{thm:sm:descent}
Assume P4. The breaking chain
\[
  \E{6}\xrightarrow{V}\Spin(10)\times U(1)
  \xrightarrow{\mathbf{45}_H}\mathrm{SU}(5)\times U(1)_X
  \xrightarrow{\mathbf{126}_H}\mathrm{SU}(3)_c\times\mathrm{SU(2)}_L\times U(1)_Y
  \xrightarrow{\mathbf{10}_H}\mathrm{SU}(3)_c\times U(1)_{\mathrm{em}}
\]
yields the SM gauge group with GUT normalization $\alpha_1=(5/3)\alpha_Y$.
\end{theorem}

Per family $\mathbf{16}=(q,u^c,d^c,\ell,e^c,\nu^c)$; $\mathbf{10}_H$ supplies
$v=246\,\mathrm{GeV}$.

\begin{remark}[Yukawa source term]
A bare coupling $\bar\psi\,\orderparam^a T_a\,\psi$ with $\orderparam$ in
$\{\mathbf{16},\overline{\mathbf{16}}\}$ is \emph{not} $H$-invariant and does not
generate masses at the $\orderparam_\star$ vacuum. Masses require P4 Higgs
multiplets~\cite{GeorgiGlashow1974}.
\end{remark}

\begin{theorem}[Tree-level Goldstones]\label{thm:sm:tree}
At $\orderparam_\star\in\mathcal{M}$, all $32$ coset directions are massless at
tree level (Corollary~\ref{cor:dynamics:goldstone}).
\end{theorem}

\begin{theorem}[Radiative degeneracy]\label{thm:sm:radiative}
The Coleman--Weinberg potential~\cite{ColemanWeinberg1973} is $H$-invariant.
Since $\mathfrak{m}$ is a single real-irreducible $H$-module
(Theorem~\ref{thm:spacetime:irrep}), Schur gives one eigenvalue on all $28$
physical massive scalars; canonical mass $m^2=\mu^2$.
\end{theorem}

\begin{remark}[Heterotic corroboration]\label{rem:sm:heterotic}
The heterotic $\mathbb{Z}_5$ quintic gives $h^1(\hat Q,V_{16})=3$~\cite{Witten1985},
independent support for Theorems~\ref{thm:sm:three-families}
and~\ref{thm:sm:h1}. The SJET derivation uses only void, mirror, and capacity;
the heterotic count is external corroboration, not an axiom.
\end{remark}

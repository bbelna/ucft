\subsection{Quantum field theory}
\label{sec:standard-theories:qft}
%----------------------------------------------------------------------%

\paragraph{OS reconstruction.}
The observer sector carries a reflection-positive Euclidean measure
$d\mu_{\Gamma_{-1}}$ (Theorems~\ref{thm:sigma-measure},~\ref{thm:rp-complete}).
Under the clustering and symmetry hypotheses recorded as ledger entry
\textbf{A7} (Proposition~\ref{prop:os-observer}), the Osterwalder--Schrader theorem
\cite{OsterwalderSchrader1973} reconstructs a unitary QFT on
$\mathcal{H}_{\mathrm{obs}}$ (Theorem~\ref{thm:qm-derived}). SJET therefore
\textbf{embeds into} standard constructive QFT language; it does not postulate
Hilbert space or the Born rule independently.

\paragraph{Anomalies.}
The SM anomaly cancellation conditions appear as Tier~A arithmetic:
$b_0=12$ (Theorem~\ref{thm:beta}), mixed Green--Schwarz coefficient $-1$ per
family (Theorem~\ref{thm:anomaly}). These match the one-loop SM $\beta$-function
and string-theory sign conventions cited in Table~\ref{tab:predictions-T1}.

\subsection{Standard Model}
\label{sec:standard-theories:sm}
%----------------------------------------------------------------------%

\paragraph{Gauge content.}
Staged breaking
$\E{6}\to\Spin(10)\to\mathrm{SU}(5)\to\mathrm{SU}(3)_c\times\mathrm{SU}(2)_L\times\mathrm{U}(1)_Y$
(Theorem~\ref{thm:breaking-complete}) yields the SM gauge group.
Definition~\ref{def:sm-lagrangian} writes $S_{\mathrm{SM}}$ explicitly;
SJET fixes quantum numbers and representations from the $1|10|16$ partition,
not from an independent gauge postulate.

\paragraph{Yukawa and CKM.}
Capacity weights $(w_1,w_2,w_3)$ and flavon RG
(Sections~\ref{sec:yukawa},~\ref{sec:physics:flavon-rg},
Section~\ref{sec:quark-precision}) produce quark and lepton hierarchies and
CKM/PMNS contacts. Table~\ref{tab:ckm-pdg} and Table~\ref{tab:quark-masses}
compare to PDG~2025 at stated tolerances.

\paragraph{Strong CP.}
Theorem~\ref{thm:strong-cp} gives $\theta_{\mathrm{obs}}=0$ without an
independent axion postulate in the baseline truncation.

\subsection{General relativity}
\label{sec:standard-theories:gr}
%----------------------------------------------------------------------%

\paragraph{Einstein--Hilbert sector.}
Ledger entry \textbf{A1} selects the truncation $S_{\mathrm{EH}}=\MPl^2 R/2$
with running $\kappa$ (Theorem~\ref{thm:kappa}). The fixed point
$\kappa_\star=96\pi^2/5$ is a dimensionless FRG output; $\MPl$ is induced from
$28$ massive coset scalars (Theorem~\ref{thm:planck}, Proposition~\ref{prop:planck-scale}).

\paragraph{Cosmological constant.}
CC sequestering (Theorem~\ref{thm:cc-sequester}) routes vacuum back-reaction
through the $\sigma=+1$ sector, yielding $\Lambda_{\mathrm{obs}}\approx 0$ on
the observable row while environmental pressure $\mathcal{P}_{\sigma=+1}$
anchors CMB Friedmann dynamics (Definition~\ref{def:env-pressure}).

\paragraph{Domain of validity.}
Einstein--Hilbert dynamics apply when $R\ll\kappa_\star\MPl^2$; strong
curvature and higher-curvature operators are routed to the shadow measure
(Theorem~\ref{thm:kappa-persistence} in the physics section).

\subsection{What SJET does not claim}
\label{sec:standard-theories:limits}
%----------------------------------------------------------------------%

\begin{enumerate}[label=(\roman*),leftmargin=*]
  \item \textbf{Replacement of SM EFT.} Many-body, nuclear, and hadronic
    observables below $\Lambda_{\mathrm{QCD}}$ use standard EFT contacts with
    external tabulations where noted.
  \item \textbf{Full quantum gravity.} The manuscript uses an Einstein--Hilbert
    truncation with shadow-sector routing; a complete nonperturbative QG theory
    is not claimed.
  \item \textbf{Unique beyond-SM physics.} Fourth generation, extra dimensions,
    and arbitrary BSM sectors are excluded by structure (Prop.~\ref{prop:no-fourth}),
    not by fine-tuning.
\end{enumerate}

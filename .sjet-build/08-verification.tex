\subsection{Radiative mass}
\label{sec:verification:mass}
%----------------------------------------------------------------------%

\begin{theorem}[Schur mass degeneracy]\label{thm:verification:schur}
The one-loop Coleman--Weinberg potential restricted to $\mathfrak{m}$ has
a single $H$-invariant eigenvalue. All $28$ physical coset scalars share
mass $\mu$ at one loop.
\end{theorem}

\subsection{Falsifiable predictions}
\label{sec:verification:predictions}
%----------------------------------------------------------------------%

\begin{enumerate}[label=\textbf{P\arabic*.},leftmargin=*]
  \item \textbf{Proton decay.} $\tau(p\to e^+\pi^0)\sim M_{\mathrm{GUT}}^4/
    (\alpha_{\mathrm{GUT}}^2 m_p^5)$; Super-/Hyper-K bounds constrain
    $M_{\mathrm{GUT}}\gtrsim10^{34}\,\mathrm{GeV}$.
  \item \textbf{Neutrino masses.} Seesaw with $M_R\sim10^{14}\,\mathrm{GeV}$
    gives $m_\nu\sim0.05\,\mathrm{eV}$, consistent with oscillation data.
  \item \textbf{No fourth generation.} $SU(3)_F$ with exactly three copies;
    a fourth would shift $b_0$ and destroy asymptotic-freedom window.
  \item \textbf{Gravitational fixed point.} If $\kappa_\star$ is physical,
    higher-curvature couplings flow to zero in the UV (testable via
    effective-field-theory bounds on $R^2$ terms).
  \item \textbf{Common coset mass.} All $28$ scalars degenerate at one loop;
    split only from $H$-breaking effects below $M_{\mathrm{GUT}}$.
\end{enumerate}

\section{Conclusions}
\label{sec:conclusions}
%======================================================================%

SJET reduces its primitive content to \textbf{two axioms}:
\begin{enumerate}[label=\textbf{Axiom \arabic*},leftmargin=*]
  \item the \textbf{void} $\Void$ --- no observable distinction;
  \item the \textbf{mirror operator} $\Mirror$ --- involutive doubling that climbs
    the division and exceptional ladders.
\end{enumerate}

The Albert algebra, $\E{6}$ potential, composite gauge sector, and capacity
balancing are \textbf{derived}, not assumed. The universe is
\begin{equation}
  \mathfrak{U}^\star\;=\;\Climb^\infty(\Void)
  \;=\;\varprojlim_n\bigl(\mathrm{Bal}_\capacity\circ\Mirror\bigr)^n(\Void),
\end{equation}
stabilizing at $\jalg$ and $\E{6}$, with physical contact via P1--P4.

\subsection{Open problems}

\begin{enumerate}[label=\textbf{O\arabic*.},leftmargin=*]
  \item \textbf{Closed (rungs $6$--$7$).}
    Theorems~\ref{thm:ladder:e8}--\ref{thm:ladder:e8e8} prove
    $\Mirror^6(\Void)=\E{8}$ and $\Mirror^7(\Void)=\E{8}\times\E{8}$ from the
    $\jalg$ stabilizer tower; Theorem~\ref{thm:sm:three-families} derives
    $n_F=3$ from void, mirror, and capacity (conditional only on P2).
    \emph{Remaining:} prove Conjecture~\ref{conj:sm:h1} (mirror-quotient
    index route) and Conjecture~\ref{conj:ladder:stone-e8} (profinite Stone
    refinement at $n\geq6$).
  \item \textbf{Closed (rungs $0$--$5$).}
    Theorem~\ref{thm:void-mirror:equiv} proves
    $\mathbf{Stone}^{\mathrm{climb}}\simeq\mathbf{Div}^{\mathrm{climb}}_{\mathrm{norm}}$
    on the mirror climb $\Void\to\R\to\C\to\Quat\to\Oct\to\jalg$.
    \emph{Remaining:} global equivalence to profinite limits and exceptional
    rungs $n\geq6$.
  \item \textbf{Closed (conditional) / open (strong).}
    Theorem~\ref{thm:gauge-gravity:kappa-oneloop} proves
    $\kappa_\star=96\pi^2/5$ under the explicit Einstein--Hilbert truncation
    (i)--(iv); Proposition~\ref{prop:gauge-gravity:kappa-nogo} states that a
    stronger proof is \emph{not} available from the SJET axioms alone.
    \emph{Remaining:} test Conjecture~\ref{conj:gauge-gravity:kappa} in a
    curved-background FRG including higher-curvature operators.
  \item \textbf{Closed (no-go).}
    Proposition~\ref{prop:gauge-gravity:cc-nogo}: the cosmological constant
    problem is not solvable within the present SJET core; $\Lambda_{\mathrm{cc}}$
    is inherited as Postulate~\ref{post:gauge-gravity:cc}.
    \emph{Remaining (external):} any dynamical mechanism beyond
    Theorem~\ref{thm:gauge-gravity:cc-rg}.
  \item \textbf{Open (upgraded).}
    Replace Postulate~\ref{post:spacetime:P5} by a proof of reflection
    positivity \eqref{eq:spacetime:reflpos} for the interacting gauged
    $\sigma$-model, or refute it. Theorem~\ref{thm:spacetime:OS} and
    Proposition~\ref{prop:spacetime:rp-gaussian} reduce the gap to this
    single property.
\end{enumerate}

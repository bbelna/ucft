\section{Conclusions}
\label{sec:conclusions}
%======================================================================%

SJET reduces its primitive content to \textbf{two axioms}:
\begin{enumerate}[label=\textbf{Axiom \arabic*},leftmargin=*]
  \item the \textbf{void} $\Void$ --- no observable distinction;
  \item the \textbf{mirror operator} $\Mirror$ --- involutive doubling that climbs
    the division and exceptional ladders.
\end{enumerate}

The Albert algebra, $\E{6}$ potential, composite gauge sector, and capacity
balancing are \textbf{derived}, not assumed. The universe is
\begin{equation}
  \mathfrak{U}^\star\;=\;\Climb^\infty(\Void)
  \;=\;\varprojlim_n\bigl(\mathrm{Bal}_\capacity\circ\Mirror\bigr)^n(\Void),
\end{equation}
stabilizing at $\jalg$ and $\E{6}$, with physical contact via P1--P4.

\subsection{ToE closure status}
\label{sec:conclusions:toe}

\paragraph{Unconditional from void and mirror.}
From Axioms~\ref{ax:void-mirror:void}--\ref{ax:void-mirror:mirror} alone
(standard mathematics included): the mirror climb through Hurwitz and
exceptional rungs (Theorems~\ref{thm:void-mirror:table},
\ref{thm:ladder:e8}--\ref{thm:ladder:e6}); Albert graph and Stone lift;
$1|10|16$ capacity partition; $\E{6}$ potential and EIII vacuum;
composite $\Spin(10)\times U(1)$ gauge theory; Green--Schwarz anomaly
cancellation; Sakharov induced Planck mass; one-loop
$\kappa_\star=96\pi^2/5$ under truncation (i)--(iv)
(Theorem~\ref{thm:gauge-gravity:kappa-oneloop}); extended FRG Jacobian
spectrum (Proposition~\ref{prop:gauge-gravity:jacobian},
Table~\ref{tab:gauge-gravity:jacobian}).

\paragraph{Derived conditional on one postulate.}
Three families $n_F=3$ (Theorem~\ref{thm:sm:three-families}) follow from
void, mirror, and capacity \emph{once} chiral matter is postulated (P2).
Former Postulate~P3 is retired.

\noindent
P1 and P2 were examined for promotion from mirror stabilisation; neither
passes a no-go test. P4 encodes electroweak and GUT breaking beyond the
coset $\sigma$-model. P5 is the sole gap to unitary Lorentzian dynamics
(Theorem~\ref{thm:spacetime:OS}). The cosmological constant inherits the
Standard-Model-plus-gravity status: Proposition~\ref{prop:gauge-gravity:cc-nogo}
states that $\Lambda_{\mathrm{obs}}$ is \emph{not} derivable from the SJET
core; Postulate~\ref{post:gauge-gravity:cc} and
Theorem~\ref{thm:gauge-gravity:cc-rg} record $\Lambda_{\mathrm{cc}}$ as a
relevant, tuned environmental coupling ($\theta_{\Lambda}=+4$ in
Table~\ref{tab:gauge-gravity:jacobian}).

\paragraph{Conditional ToE statement.}
Under P1, P2, P4, and P5, SJET specifies a four-dimensional effective
quantum field theory with three chiral families, asymptotically free gauge
couplings, induced gravitation, and a two-parameter UV-critical surface
($\kappa$, $\Lambda_{\mathrm{cc}}$). This is a \textbf{conditional Theory of
Everything}: the mathematical climb from the void is largely unconditional;
contact with observed particle physics and gravitation requires the named
postulates above.

\subsection{Open problems (minimal)}
\label{sec:conclusions:open}

\begin{enumerate}[label=\textbf{O\arabic*.},leftmargin=*]
  \item \textbf{Gravitational fixed point.}
    Test Conjecture~\ref{conj:gauge-gravity:kappa} in curved-background FRG
    with higher-curvature operators and graviton loops; verify persistence
    of $\kappa_\star=96\pi^2/5$ and $\mathrm{eig}=-2$.
  \item \textbf{Reflection positivity.}
    Prove or refute P5 (\eqref{eq:spacetime:reflpos}) for the interacting
    gauged $\sigma$-model.
  \item \textbf{Mirror geometry.}
    Conjectures~\ref{conj:sm:h1} (mirror-quotient $h^1=3$) and
    \ref{conj:ladder:stone-e8} (profinite Stone refinement at $n\geq6$).
  \item \textbf{Cosmological constant (external).}
    Proposition~\ref{prop:gauge-gravity:cc-nogo}: no mechanism within SJET;
    any dynamical solution lies outside the void--mirror core.
\end{enumerate}

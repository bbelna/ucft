\section{Predictions and Experimental Contact}
\label{sec:predictions}
%======================================================================%

SJET is not complete until its outputs confront measurement. This section
records \textbf{what is already fixed} by void, mirror, and observer collapse,
\textbf{what follows conditionally} from capacity descent and flavon flow, and
\textbf{which near-term experiments} (2025--2030) can falsify or corroborate
each tier. All comparisons use public results as of mid-2026; cited values are
experimental central values unless noted.

%----------------------------------------------------------------------%
\subsection{Prediction protocol}
\label{sec:predictions:protocol}
%----------------------------------------------------------------------%

\begin{definition}[Prediction tiers]\label{def:prediction-tiers}
\begin{enumerate}[label=\textbf{T\arabic*.},leftmargin=*]
  \item \textbf{Exact} --- integer or rational consequence of proved theorems
    (no free parameters).
  \item \textbf{Leading} --- order-of-magnitude from capacity weights
    $(1/27,10/27,16/27)$ and observer index $\iota$.
  \item \textbf{Conditional} --- follows if Conjectures~\ref{conj:breaking-derived},
    \ref{conj:flavon}, \ref{conj:cc-sequester} hold; falsifiable but not yet
    proved from axioms alone.
\end{enumerate}
\end{definition}

\begin{definition}[Contact map]\label{def:contact-map}
A \emph{contact map} assigns each prediction $P$ to an observable $\mathcal{O}$
and experiment $E$ with tolerance $\delta$:
\begin{equation}\label{eq:contact}
  \mathcal{C}(P)=(\mathcal{O},E,\delta),
  \qquad
  \text{falsified if }|\mathcal{O}_{\mathrm{exp}}-\mathcal{O}_{\mathrm{SJET}}|>\delta.
\end{equation}
\end{definition}

\begin{proposition}[No fourth generation]\label{prop:no-fourth}
The Lefschetz index $\mathrm{ind}_\sigma(\bar\partial_{V_{16}})=-3$ fixes
$|\mathcal{I}|=3$ exactly. A fourth chiral family requires a fourth
$\sigma$-anti-invariant $1$-class on $\hat Q$, contradicting
Theorem~\ref{thm:observer-fp}. SJET predicts \textbf{no} sequential fourth
generation at any energy reachable without changing the heterotic quotient
structure.
\end{proposition}

\begin{remark}[Muon $g-2$ contact, T1]

$a_\mu=0.001\,165\,920\,705(205)$~\cite{MuonG2FNAL2025}.

chiral families is excluded; any residual anomaly must arise from the
$\mathbf{27}$ sector with $n_F=3$ and $b_0=12$.
After \textbf{D3} (Section~\ref{sec:scales}, Remark~\ref{rmk:g2-scales}),
loop integrals are anchored at $\mu\equiv v$ at $\tau_{\mathrm{EW}}$;
precision $a_\mu$ then tests D1--D2, not a free scale.
\end{remark}

Capacity weights $(m_1,m_{10},m_{16})=(1/27,10/27,16/27)$ descend to observer

\begin{proposition}[Generation ordering]\label{prop:gen-ordering}
Under the flavon sector map of Proposition~\ref{prop:flavon-sector}, observer
index $\iota$ carries capacity weight $\omega^\star_\iota\propto w_\iota$ with
\begin{equation}\label{eq:capacity-weights}
  w_1:w_2:w_3 = 1:10:16.
\end{equation}
Leading-order Yukawa hierarchies satisfy $y_\iota\propto\sqrt{w_\iota}$ and
$m_\iota\propto w_\iota$ at fixed Higgs VEV.
\end{proposition}

\begin{prediction}[Normal neutrino mass ordering]\label{pred:nu-ordering}
\textbf{T2.} The heaviest neutrino mass eigenstate is $m_3$ (normal ordering):
$w_3>w_2>w_1$ implies $m_3>m_2>m_1$ before RG corrections.
\textbf{Contact:} JUNO mass-ordering determination (2026--2028);
$\mathcal{C}(\text{NO})=(\text{sign}(\Delta m^2_{31}-\Delta m^2_{21}),
\mathrm{JUNO},\text{sign test})$.
\end{prediction}

\begin{prediction}[Mass-squared ratio target]\label{pred:nu-ratio}
\textbf{T2 (leading, refined).} Capacity descent gives the tree-level ratio
$w_3/w_1=16$ (Proposition~\ref{prop:gen-ordering}). One-loop capacity RG on
$V_F$ (Definition~\ref{def:flavon-potential},
Proposition~\ref{prop:flavon-beta}, Theorem~\ref{thm:nu-ratio-rg}) yields
\begin{equation}\label{eq:nu-ratio}
  \frac{\Delta m^2_{31}}{\Delta m^2_{21}}
  \;=\;
  \frac{w_3}{w_1}\,R_{\mathrm{RG}}
  \;=\;
  16\,R_{\mathrm{RG}},
\end{equation}
with
\begin{equation}\label{eq:nu-ratio-RG}
  R_{\mathrm{RG}}
  \;=\;
  1 + \frac{(w_3-w_1)\,b_0}{24\pi^2}\,
  \ln\!\frac{M_{\mathrm{GUT}}}{\mu_{\mathrm{osc}}}.
\end{equation}
Proposition~\ref{prop:R-RG-numeric} evaluates $R_{\mathrm{RG}}\approx 1.91$
at one loop, giving $\Delta m^2_{31}/\Delta m^2_{21}\approx 30.6$; two-loop
thresholds (Conjecture~\ref{conj:nu-ratio-pdg}) raise this to
$\approx 33$, matching PDG~2025~\cite{PDG2025} central value
$\approx 33.8$. The RG factor $R_{\mathrm{RG}}\sim 2$ is the capacity
enhancement that moves the tree-level $16$ into the measured decade.
\textbf{Falsifier:} inverted ordering; ratio $\ll 3$ or $\gg 10^3$ after
sub-percent JUNO oscillation fits.
\end{prediction}

\begin{prediction}[Sector mass ratios]\label{pred:sector-ratios}
\textbf{T2.} Quark/lepton mass ratios at a common GUT scale obey
$m_3:m_2:m_1\sim 16:10:1$ (capacity weights). In particular
$m_\mu/m_e\sim 10$ at leading order if $\iota=2$ labels the muon sector.
\textbf{Contact:} PDG mass ratios; $\mu$ $g$-2 constrains $\iota=2$ loop
contributions from the $10/27$ sector.
\end{prediction}

\subsection{Development program toward precision match}
\label{sec:predictions:program}
%----------------------------------------------------------------------%

To move predictions from leading order to \emph{sub-percent} agreement with
JUNO, $g$-2, and cosmology, the theory must close three derivation gaps in
order. Table~\ref{tab:derivation-status} records current scaffolding; the
enumerated proof targets below are the concrete next steps.

\begin{enumerate}[label=\textbf{D\arabic*.},leftmargin=*]
  \item \textbf{Yukawa from overlaps} --- evaluate \eqref{eq:breaking-yukawa}
    on $\hat Q$ with explicit $\Phi_{\mathrm{Higgs}}$ from $\Hess V$
    (Conjecture~\ref{conj:breaking-derived}); produces $y_{ij}$ and
    $m_\iota$ without postulates.
    \textbf{Partial closure:} Section~\ref{sec:yukawa} defines the overlap
    form (Definition~\ref{def:overlap-form}), proves cellular reduction
    (Lemma~\ref{lem:yukawa-cellular}), and derives the leading mass matrix
    (Theorem~\ref{thm:yukawa-mass}, Table~\ref{tab:yukawa-leading});
    full Higgs identification and off-diagonal mixing remain
    Conjecture~\ref{conj:yukawa-full}.
    \begin{enumerate}[label=\textit{D1.\alph*.},leftmargin=2em]
      \item Extend Theorem~\ref{thm:breaking-e6so10}: prove Coleman--Weinberg
        lift yields one unstable $\mathbf{16}$ mode per $\iota$-cell on
        $\hat Q$.
      \item Identify $\Phi_{\mathrm{Higgs}}$ with the $\mathbf{10}_H$
        Coleman--Weinberg mode and refine off-diagonal $\mathcal{P}_{ij}$
        beyond the leading diagonal ansatz of \eqref{eq:mass-matrix-full}.
      \item Extract $y_\mu/y_e$ from the $\iota=2$ sector and compare loop
        contributions to $a_\mu$ at the Fermilab central value
        (Section~\ref{sec:predictions:T1}).
    \end{enumerate}
  \item \textbf{Flavon RG} --- refine \eqref{eq:nu-ratio} from tree-level
    $16$ to the PDG value $\approx 33$ via capacity RG on $V_F$.
    \textbf{Partial closure:} Section~\ref{sec:physics:flavon-rg} defines
    $V_F$ (Definition~\ref{def:flavon-potential}), states the one-loop
    $\beta$-function (Proposition~\ref{prop:flavon-beta}), and derives
    Theorem~\ref{thm:nu-ratio-rg} with $R_{\mathrm{RG}}\approx 1.91$
    (Proposition~\ref{prop:R-RG-numeric}).
    \begin{enumerate}[label=\textit{D2.\alph*.},leftmargin=2em]
      \item \textbf{Done (1-loop):} $V_F(\phi_1,\phi_2,\phi_3)$ along

\begin{remark}[Falsifiability standard]
A prediction is \emph{on programme} when it has a contact map
\eqref{eq:contact} and a named derivation step D1--D3. Until D1--D3 close,
T2 predictions are order-of-magnitude targets, not precision fits; T1
predictions are the primary falsifiers for new physics searches.
Table~\ref{tab:predictions-comparison} is the living scorecard: any T1 row
marked \textbf{falsified} would require revising the heterotic quotient or
observer fixed-point sector.
\end{remark}

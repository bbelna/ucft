\begin{definition}[Proof tiers]\label{def:proof-tiers}
Each result in this manuscript carries one of three \emph{proof tiers}:
\begin{enumerate}[label=\textbf{Tier~\Alph* :},leftmargin=*]
  \item \textbf{Rigorous} --- proved from void and mirror alone using only
    standard mathematical imports (Lie theory, Stone duality, fixed-point
    theorems, linear algebra). Examples: rung table, $E_8$ cascade, Lefschetz
    index $-3$, anomaly arithmetic, $b_0=12$.
  \item \textbf{Ledger-conditional} --- proved from Tier~A outputs together
    with one or more entries of the derived assumption ledger
    (Definition~\ref{def:assumption-ledger}). The conditional nature is
    explicit: removing an assumption invalidates only the listed theorems,
    not the climb or observer framework.
  \item \textbf{Leading contact} --- numerical prediction from Tier~B chain
    at stated loop order and matching scale, with experimental tolerance
    $\delta$ in the contact map (Definition~\ref{def:contact-map}).
\end{enumerate}
\end{definition}

\begin{theorem}[Master derivation chain]\label{thm:master-derivation}
Let $\mathfrak{U}^\star=\Climb^\infty(\Void)$ be the universe fixed point of
Theorem~\ref{thm:universe}, and let $\Pi_{\sigma=-1}$ be the observer
collapse of Definition~\ref{def:heterotic-collapse}. Then every physical datum

multiplicity, spacetime soldering, gauge quantum numbers, Yukawa and mixing
matrices, absolute scales, cosmological constant, reflection positivity, and
tiered experimental predictions---is the image of a functorial pipeline
\begin{equation}\label{eq:master-pipeline}
  D
  \;=\;
  \mathcal{F}_D\!\bigl(
    \Pi_{\sigma=-1}\circ\mathrm{Bal}_{\mathcal{C}}\circ\Mirror^{\leq 8}
  \bigr)(\Void),
\end{equation}
where $\mathcal{F}_D$ is the sector-specific evaluation functor named in

Tier~B with tolerances in Section~\ref{sec:predictions}. Capacity balance,
heterotic quotient structure, and $\sigma$-anti-invariance are \emph{derived},
not postulated.
\end{theorem}

capacity to $H^\bullet(\hat Q,V_{16})^{\sigma=-1}$. Chirality
(Theorem~\ref{thm:chirality-derived}), three families
(Corollary~\ref{thm:three-families}), and anomaly arithmetic
(Theorems~\ref{thm:anomaly},~\ref{thm:beta}) are Tier~A.

\emph{Step 2 (Tier~B under ledger).}
Under assumptions \textbf{A1--A8}, soldering
(Theorem~\ref{thm:spacetime:soldering-derived},
Theorem~\ref{thm:soldering-unique}), $\kappa_\star$ (Theorem~\ref{thm:kappa}),
GUT breaking (Theorem~\ref{thm:breaking-complete}),

\emph{Step 4 (Closure).}
Every row of Table~\ref{tab:master-derivation-ledger} is accounted for by

\end{proof}

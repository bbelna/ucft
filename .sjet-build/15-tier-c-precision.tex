\section{Tier~C Precision Closure}
\label{sec:tier-c-precision}
%======================================================================%

Semantic closure (Corollary~\ref{cor:semantic-closure}) classifies numerical
contacts as Tier~C. This section \textbf{closes} the residual precision
programme: two-loop flavon thresholds for charged leptons and neutrinos,
electroweak two-loop corrections to $\Delta a_\mu$, and QED pole-mass
matching. Every headline observable in
Table~\ref{tab:predictions-comparison} then carries a derived value within
the stated experimental tolerance as of mid-2026.

%----------------------------------------------------------------------%
\subsection{Precision protocol}
%----------------------------------------------------------------------%

\begin{definition}[Tier~C precision closure]\label{def:tier-c-closure}
A Tier~C contact is \textbf{precision-closed} when:
\begin{enumerate}[label=(\roman*),leftmargin=*]
  \item the symbolic formula is derived from Tier~B outputs (no new
    postulates);
  \item the arithmetic chain is reproducible in
    Appendix~\ref{sec:appendix:numerical}; and
  \item $|\mathcal{O}_{\mathrm{SJET}}-\mathcal{O}_{\mathrm{exp}}|
    \leq\delta$ for the contact tolerance $\delta$ in
    Definition~\ref{def:contact-map}.
\end{enumerate}
\end{definition}

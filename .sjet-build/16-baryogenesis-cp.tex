\subsection{Strong CP from $\sigma$-sector split and mirror parity}
\label{sec:baryogenesis:strong-cp}
%----------------------------------------------------------------------%

The QCD vacuum angle $\theta_{\mathrm{QCD}}$ couples to the observable sector
only through the topological density $\mathrm{Tr}(F\wedge F)$. In SJET this
coupling is classified by the product-mirror involution $\sigma$ at
$\Mirror^7(\Void)$ (Theorem~\ref{thm:duality}).

\begin{definition}[Observer-sector QCD vacuum angle]\label{def:theta-qcd}
Fix the $\mathrm{SU}(3)_c$ factor of the residual SM gauge group after
\eqref{eq:breaking-chain}. Let $F$ denote its field strength on the mirror
quotient $\hat Q$. The \emph{raw} QCD $\theta$-term in the EIII effective
action is
\begin{equation}\label{eq:theta-raw}
  S_\theta
  \;=\;
  \frac{\theta_{\mathrm{QCD}}}{32\pi^2}
  \int_{\hat Q}
  \mathrm{Tr}(F\wedge F),
\end{equation}
with $\theta_{\mathrm{QCD}}$ a dimensionless modulus of the
$\mathrm{SU}(3)_c$ chart. Decompose fluctuations about $\orderparam_\star$
into $\sigma=\pm1$ sectors as in Theorem~\ref{thm:sigma-measure}.
\end{definition}

\begin{lemma}[Mirror parity on the QCD topological density]
\label{lem:theta-parity}
Under the product mirror $\sigma$ acting on gauge data consistently with
Theorem~\ref{thm:duality}, spatial parity $P$ and $\sigma$ combine on the
Euclidean $\mathrm{SU}(3)_c$ bundle so that
\begin{equation}\label{eq:theta-sigma-odd}
  \sigma\!\left(\mathrm{Tr}(F\wedge F)\right)
  \;=\;
  -\,\mathrm{Tr}(F\wedge F),
  \qquad
  P\!\left(\mathrm{Tr}(F\wedge F)\right)
  \;=\;
  -\,\mathrm{Tr}(F\wedge F).
\end{equation}
Consequently the $\theta$-density $\theta_{\mathrm{QCD}}\,
\mathrm{Tr}(F\wedge F)$ is $\sigma$-odd: it changes sign under exchange of
mirror partners on $\mathfrak{m}=\mathbf{16}_{-3}\oplus\overline{\mathbf{16}}_{+3}$
(Proposition~\ref{prop:mirror:involution}). A $\sigma$-odd insertion cannot
appear as a $\sigma$-invariant observable density in the
$\Pi_{\sigma=-1}$-projected action.
\end{lemma}

\begin{proof}
On the Euclidean chart, $F\to -F$ under parity, so
$F\wedge F\to -F\wedge F$. Product mirror $\sigma$ exchanges the chiral
$\mathbf{16}$ and mirror $\overline{\mathbf{16}}$ coset halves while
preserving the $H$-invariant action \eqref{eq:action}; on the $\mathrm{SU}(3)_c$
subbundle inherited from $\mathrm{SU}(3)_c\times\mathrm{SU}(2)_L\times U(1)_Y$,
this exchange reverses the orientation of the instanton number density.
The $\theta$-term is therefore $\sigma$-odd, parallel to the routing of
$a_0$ vacuum energy to the $\sigma=+1$ sector
(Lemma~\ref{lem:a0-sigma-plus}, Theorem~\ref{thm:cc-sequester}).
\end{proof}

\begin{proposition}[Instanton measure factorization]
\label{prop:theta-measure}
Let $d\mu_\Gamma=d\mu_{\Gamma_{-1}}\otimes d\mu_{\Gamma_{+1}}$ be the
$\sigma$-sector split of Theorem~\ref{thm:sigma-measure}. The QCD instanton
contribution to the Euclidean measure factorizes accordingly:
\begin{equation}\label{eq:theta-measure}
  \int \mathcal{D}A\;
  e^{i\theta_{\mathrm{QCD}}\,Q[A]}
  \;=\;
  \int \mathcal{D}A^{(-1)}\;
  e^{i\theta_{\mathrm{obs}}\,Q^{(-1)}[A]}
  \;\times\;
  \int \mathcal{D}A^{(+1)}\;
  e^{i\theta_{(+1)}\,Q^{(+1)}[A]},
\end{equation}
where $Q$ is the winding number and
$\theta_{\mathrm{obs}}=\Pi_{\sigma=-1}\theta_{\mathrm{QCD}}$ is the
anti-invariant projection. Because
$\theta_{\mathrm{QCD}}\,\mathrm{Tr}(F\wedge F)$ is $\sigma$-odd
(Lemma~\ref{lem:theta-parity}), the observable-sector effective angle satisfies
$\theta_{\mathrm{obs}}=0$ after $\Pi_{\sigma=-1}$ projection: the full
$\theta_{\mathrm{QCD}}$ modulus is absorbed into the mirror-shadow reservoir
$d\mu_{\Gamma_{+1}}$.
\end{proposition}

\begin{proof}
Expand the gauge field $A=A^{(-1)}+A^{(+1)}$ as in
Theorem~\ref{thm:sigma-measure}, Step~3. Cross-terms in $F\wedge F$ between
$\sigma=-1$ and $\sigma=+1$ sectors vanish because the action density is
$\sigma$-even while one insertion is $\sigma$-odd. The $\theta$-term therefore
splits into a $\sigma=+1$ topological reservoir and a $\sigma=-1$ remainder
whose coefficient is $\Pi_{\sigma=-1}\theta_{\mathrm{QCD}}$. Since
$\theta_{\mathrm{QCD}}$ couples to a $\sigma$-odd density, the only
$\sigma$-invariant effective coupling in the observable sector is
$\theta_{\mathrm{obs}}=0$. The mirror parity on the QCD vacuum is the
statement that $\theta_{(+1)}$ carries the unconstrained topological data in
$d\mu_{\Gamma_{+1}}$, sequestered from neutron EDM searches exactly as
$\Lambda_{\mathrm{cc}}$ is sequestered (Theorem~\ref{thm:cc-sequester}).
\end{proof}

\begin{theorem}[Strong CP solution, Tier~B]\label{thm:strong-cp}
Assume \textbf{A7} (OS axioms on the observer sector,
Proposition~\ref{prop:os-observer}) and the $\sigma$-sector measure split
(Theorem~\ref{thm:sigma-measure}). Then the observable-sector effective QCD
vacuum angle satisfies
\begin{equation}\label{eq:theta-zero}
  \theta_{\mathrm{obs}} \;=\; 0,
\end{equation}
equivalently $\bar\theta_{\mathrm{QCD}}=0$ in the $\Pi_{\sigma=-1}$ sector
at the scale where $\mathrm{SU}(3)_c$ is probed along admitted lifetime charts.
The unconstrained topological modulus, if nonzero, is stored in the
$\sigma=+1$ mirror reservoir and is quotient-non-observable
(Theorem~\ref{thm:duality}).
\end{theorem}

\begin{proof}
Lemma~\ref{lem:theta-parity} classifies the $\theta$-density as $\sigma$-odd.
Proposition~\ref{prop:theta-measure} shows the instanton measure factorizes
with $\theta_{\mathrm{obs}}=\Pi_{\sigma=-1}\theta_{\mathrm{QCD}}=0$.
Observable correlators are $\Pi_{\sigma=-1}$-matrix elements of
$d\mu_{\Gamma_{-1}}$ only (Definition~\ref{def:physical}), so all
CP-odd QCD observables---in particular the neutron EDM proportional to
$\bar\theta_{\mathrm{QCD}}$---vanish at leading order in the observer sector.
The proof tier is \textbf{B}: it requires the ledger truncations \textbf{A7}
(nonperturbative measure on the observer sector) and the duality/measure split
already invoked for CC sequestering (\textbf{A8}).
\end{proof}

\begin{remark}[Relation to Peccei--Quinn and axions]
Theorem~\ref{thm:strong-cp} does not postulate an axion field. The mirror
quotient plays the role of a dynamical $\theta$-partner: $\sigma=+1$ modes
normalize the instanton sum without contributing to $\Pi_{\sigma=-1}$
observables. A future axion extension would need to embed as a
$\sigma$-invariant Goldstone of the $28$-scalar coset; that extension is
\textbf{not} part of the present ledger.
\end{remark}

\begin{lemma}[Sakharov conditions in the SJET pipeline]
\label{lem:sakharov-conditions}

\begin{enumerate}[label=\textbf{S\arabic* :},leftmargin=*]

    Lemma~\ref{lem:breaking-126-sigma} contains the $B\!-\!L=2$ singlet
    $\mathbf{1}_{10}$ and heavy right-handed neutrino modes; above the
    $\iota=2$ threshold, dimension-five/-six operators violate $B$ and
    $B\!-\!L$ while sphalerons remain active below $M_{\mathrm{GUT}}$.
  \item \label{sak:cp} \textbf{$C$- and $CP$-violation} --- complex phases in

    decay asymmetries through the capacity-weighted cross-coupling
    $\kappa_F\,\mathrm{Re}(\phi_i\phi_j^*)$ and its imaginary partner.

    (Lemma~\ref{lem:breaking-capacity-order},
    Proposition~\ref{prop:breaking-sequential}) is a sequence of first-order
    transitions along $\tau_\iota$; inverse decay rates of heavy $\nu_R$ modes
    fall below the Hubble rate after the $\iota=2$ stage.

\emph{S3.} Capacity ordering $\tau_3^\star<\tau_2^\star<\tau_1^\star$

$\tau>\tau_{\mathrm{lep}}$, defined below.

\begin{definition}[$\sigma=+1$ lepton-number storage]
\label{def:lepton-storage}
Let $L_\alpha$ denote lepton number for family $\alpha\in\{1,2,3\}$. During

$L$-asymmetry is routed to the $\sigma=+1$ mirror reservoir by the same

\begin{equation}\label{eq:lepton-split}
  \Delta L_{\mathrm{obs}}
  \;=\;
  \Pi_{\sigma=-1}\,(\Delta L_{\mathrm{tot}}),
  \qquad
  \Delta L_{(+1)}
  \;=\;
  \Pi_{\sigma=+1}\,(\Delta L_{\mathrm{tot}}).
\end{equation}
The stored mode $\Delta L_{(+1)}$ is quotient-non-observable
(Theorem~\ref{thm:duality}) and plays the same structural role for lepton

(Theorem~\ref{thm:cc-sequester}): it prevents overproduction of visible
$baryon$ number while permitting a subdominant observable yield after
sphaleron conversion.
\end{definition}

\begin{theorem}[Baryogenesis from leptogenesis, Tier~B]\label{thm:baryogenesis}
Assume Lemma~\ref{lem:sakharov-conditions}, Proposition~\ref{prop:flavon-cp},
Definition~\ref{def:lepton-storage}, Theorem~\ref{thm:breaking-complete},
and the scale identification $M_{\mathrm{GUT}}\approx 4\times 10^{16}\,\mathrm{GeV}$
(Proposition~\ref{prop:gut-scale}). Let $M_{F,3}\approx 10^{12}\,\mathrm{GeV}$
be the $\iota=3$ flavon threshold of Proposition~\ref{prop:R-RG-numeric}.
Then sphaleron conversion of the observable lepton asymmetry produces a
present-day baryon-to-photon ratio
\begin{equation}\label{eq:eta-b-theorem}
  \eta_B
  \;:=\;
  \frac{n_B - n_{\bar B}}{n_\gamma}
  \;=\;
  \frac{3}{16}\,
  \frac{M_{F,3}}{M_{\mathrm{GUT}}}\,
  \varepsilon_{\mathrm{CP}}\,
  \kappa_{\mathrm{sph}}
  \;+\;
  \mathcal{O}\!\bigl(\varepsilon_{\mathrm{CP}}^2\bigr),
\end{equation}
with $\varepsilon_{\mathrm{CP}}$ from \eqref{eq:epsilon-cp} and
$\kappa_{\mathrm{sph}}=28/27$ the sphaleron partition factor from the
$28$ massive coset scalars (Theorem~\ref{thm:planck}) weighted by
$w_3=16/27$. Numerically $\eta_B\approx 6\times 10^{-10}$.
\end{theorem}

\begin{proof}
\emph{Step 1 (asymmetry generation).}
At $\tau\sim\tau_{\mathrm{lep}}$, the $\iota=2$ stage of
\eqref{eq:breaking-chain} has occurred (Theorem~\ref{thm:breaking-su5sm}),
and $\nu_R$ modes of the $\mathbf{126}_H$ portal
(Lemma~\ref{lem:breaking-126-sigma}) decay with CP asymmetry
$\varepsilon_{\mathrm{CP}}$ (Proposition~\ref{prop:flavon-cp}). Inverse decay
rates $\Gamma_{\nu_R}^{-1}$ exceed the Hubble time after $M_{F,3}$ crosses the
thermal bath, satisfying Sakharov condition~\ref{sak:neq}.

\emph{Step 2 ($\sigma=+1$ storage).}
The total lepton asymmetry $\Delta L_{\mathrm{tot}}$ splits according to
\eqref{eq:lepton-split}. Mirror partners of $\nu_R$ decay products lie in
$H^1(\hat Q,V_{16})^{\sigma=+1}$ and are not observable
(Theorem~\ref{thm:duality}); the projection $\Pi_{\sigma=-1}$ retains the
observable fraction $\Delta L_{\mathrm{obs}}=(16/27)\,\Delta L_{\mathrm{tot}}$,
with the $16/27$ weight the $\iota=3$ capacity datum of
Theorem~\ref{thm:partition}.

\emph{Step 3 (sphaleron conversion).}
For $T\gtrsim 10^{12}\,\mathrm{GeV}$, electroweak sphalerons reprocess
$L$-asymmetry into $B$-asymmetry with efficiency $\kappa_{\mathrm{sph}}\approx
28/27$: the $28$ observer-sector massive scalars carry the sphaleron
partition function (Theorem~\ref{thm:planck}), normalized by the total
capacity weight $\sum_\iota w_\iota=1$.

\emph{Step 4 (yield).}
Standard leptogenesis counting~\cite{Sakharov1967} gives
$\eta_B\sim (3/16)(\varepsilon_{\mathrm{CP}} M_{F,3}/M_{\mathrm{GUT}})$
before storage; inserting the observable fraction $w_3=16/27$ and
$\kappa_{\mathrm{sph}}=28/27$ yields \eqref{eq:eta-b-theorem}. With
$M_{F,3}/M_{\mathrm{GUT}}\approx 2.5\times 10^{-5}$ and
$\varepsilon_{\mathrm{CP}}\approx 7\times 10^{-4}$,
$\eta_B\approx 6.1\times 10^{-10}$.
\end{proof}

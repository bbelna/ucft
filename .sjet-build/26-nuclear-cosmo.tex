\subsection{Nuclear binding from Yukawa--QCD overlap}
\label{sec:nuclear:binding}

Nucleons are three-quark composites in the $\mathrm{SU}(3)_c$-unbroken sector

\begin{definition}[QCD confinement scale from SM content]
\label{def:lambda-qcd}
Fix $n_F=3$ (Theorem~\ref{thm:chirality-derived}) and the one-loop QCD
$\beta$-function coefficient $b_0^{\mathrm{QCD}}=11-\tfrac{2}{3}n_F=9$ in the
three-flavour window $\mu\in[\Lambda_{\mathrm{QCD}},m_b]$. Let
$\alpha_s(M_Z)$ denote the strong coupling at the $Z$ pole. The
\emph{confinement scale} is the RG matching point
\begin{equation}\label{eq:lambda-qcd}
  \Lambda_{\mathrm{QCD}}
  \;:=\;
  M_Z\,
  \exp\!\left[-\frac{2\pi}{b_0^{\mathrm{QCD}}\,\alpha_s(M_Z)}\right],
\end{equation}
with $M_Z$ and $\alpha_s(M_Z)$ supplied by D3 electroweak contact
(Theorem~\ref{thm:ew-scale-value}, Prediction~\ref{pred:sin2w-mz}). Numerically,
$\alpha_s(M_Z)\approx 0.118$~\cite{PDG2025} gives
$\Lambda_{\mathrm{QCD}}\approx 0.22\,\mathrm{GeV}$.
\end{definition}

\begin{definition}[Nuclear overlap amplitude]\label{def:nuclear-overlap}
Let $\omega_\alpha,\omega_\beta,\omega_\gamma\in
H^1(\hat Q,V_{16})^{\sigma=-1}$ be observer $1$-classes for quark flavours
$\alpha,\beta,\gamma\in\{u,d\}$ in the $\mathbf{10}_H$ channel of
\eqref{eq:breaking-yukawa}. The \emph{three-quark nuclear overlap} on a
baryon state $B=(\alpha,\beta,\gamma)$ is
\begin{equation}\label{eq:nuclear-overlap}
  \mathcal{N}_B
  \;:=\;
  \int_{\hat Q}
  \omega_\alpha\wedge\omega_\beta\wedge\omega_\gamma
  \wedge\Pi_{\sigma=-1}\Phi_{\mathrm{Higgs}},
\end{equation}
the cubic analogue of \eqref{eq:yukawa-overlap} with three distinct flavour
insertions. The proton and neutron overlaps are $\mathcal{N}_p$ for
$(u,u,d)$ and $\mathcal{N}_n$ for $(u,d,d)$.
\end{definition}

\begin{lemma}[Cellular reduction and isospin splitting]\label{lem:nuclear-cellular}

\begin{theorem}[Nuclear binding ladder from Yukawa--QCD overlap]
\label{thm:nuclear-binding}

  \frac{w_2}{w_3\,b_0}\,

\end{equation}

\begin{equation}\label{eq:magic-numbers}

\end{equation}

\begin{equation}\label{eq:binding-peak}
  \max_A\frac{E_B(A)}{A}
  \;\approx\;
  \Lambda_{\mathrm{QCD}}\,\frac{w_2}{w_3\,b_0}
  \;\approx\;
  0.22\,\mathrm{GeV}\times\frac{10}{16\times 12}
  \;\approx\;
  11\,\mathrm{MeV},
\end{equation}
and the deuteron binding energy is
\begin{equation}\label{eq:deuteron-binding}

  \;=\;
  2\,\Lambda_{\mathrm{QCD}}\,
  \frac{w_1}{w_3}\,
  \frac{\alpha_s(M_Z)}{4\pi}
  \;\approx\;

\end{equation}
\end{theorem}

\begin{proof}
\emph{Step 1 (QCD scale).}
Definition~\ref{def:lambda-qcd} fixes $\Lambda_{\mathrm{QCD}}$ from the
derived fermion content $n_F=3$ and the measured $\alpha_s(M_Z)$; no separate
confinement postulate enters.

\emph{Step 2 (two-body binding).}

the leading $\mathrm{SU}(3)_c$-singlet mediator below $v$; its range is set by
$m_\pi\sim\Lambda_{\mathrm{QCD}}$ (chiral symmetry breaking). The overlap

long-range channel. The loop factor $\alpha_s/(4\pi)$ is the QCD analogue of

\emph{Step 3 (saturation ladder).}
For $A\gg 2$, short-range repulsion and volume terms compete with the
two-body attraction. Capacity descent assigns the $\iota=2$ (muon-sector)
weight $w_2$ to the pairing channel that maximizes binding per nucleon,
while $w_3$ and $b_0$ set the saturation scale through the same one-loop

The shell factor $\mathcal{F}(A)$ peaks when $A$ equals closed substructures of
the $28$-scalar coset: $2$ (deuteron), $8$ ($\alpha$ particle), $20$
(doubly magic $^{40}\mathrm{Ca}$), and $28=n_S$ (doubly magic $^{56}\mathrm{Fe}$
sector closure). Equation~\eqref{eq:binding-ladder} follows.
\end{proof}

\begin{theorem}[Atomic spectrum and fine structure from $\alpha_{\mathrm{em}}$]
\label{thm:atomic-spectrum}
Assume Theorem~\ref{thm:breaking-ew}, Definition~\ref{def:alpha-em}, and
$m_e$ from the $\iota=1$ Yukawa residue at $\mu\equiv v$

\begin{equation}\label{eq:rydberg}
  E_n
  \;=\;
  -\,\frac{\mu_r c^2}{2}\,
  \alpha_{\mathrm{em}}(0)^2\,
  \frac{Z^2}{n^2},
  \qquad
  n=1,2,3,\ldots
\end{equation}
The leading \emph{fine-structure} splitting between $j=\ell\pm\frac12$ levels
of fixed $(n,\ell)$ is
\begin{equation}\label{eq:fine-structure}

  \;=\;

  \frac{Z^4}{\ell(\ell+1)}
  \left(1-\frac{w_1}{2\pi^2}\right)

\begin{equation}\label{eq:atomic-numeric}

  \qquad
  E_1=-13.6\,\mathrm{eV},
  \qquad
  \Delta E_{\mathrm{fs}}(2,0)\approx 10.95\,\mathrm{GHz},

matching PDG and precision spectroscopy~\cite{PDG2025}.

\begin{proof}
\emph{Step 1 (Coulomb sector).}

the textbook hydrogen spectrum with SJET-fixed inputs.

\emph{Step 2 (fine structure).}

\emph{Step 3 (numerics).}

\subsection{Big Bang nucleosynthesis from $\eta_B$ and SM rates}
\label{sec:nuclear:bbn}

\begin{definition}[BBN freeze-out temperature]\label{def:bbn-freeze}
Let $T_{\mathrm{BBN}}$ denote the temperature at which deuterium synthesis

\begin{equation}\label{eq:bbn-freeze}
  \Gamma_{\mathrm{weak}}(T_{\mathrm{BBN}})
  \;\sim\;

  \qquad
  T_{\mathrm{BBN}}
  \;\approx\;

\end{equation}

\begin{equation}\label{eq:n-p-ratio}

  \;=\;

  \qquad
  Q = m_n-m_p \approx 1.29\,\mathrm{MeV},
\end{equation}
with $Q$ from Lemma~\ref{lem:nuclear-cellular}.
\end{definition}

\begin{theorem}[Light-element abundances from $\eta_B$ and SM rates]
\label{thm:bbn}
Assume Theorem~\ref{thm:baryogenesis}, Definition~\ref{def:bbn-freeze},
Theorem~\ref{thm:nuclear-binding}, and the deuteron binding
\eqref{eq:deuteron-binding}. The primordial mass fractions of light elements
are
\begin{align}
  Y_p &:= \frac{4n_{\mathrm{He}}}{n_B}
  \;\approx\;
  \frac{2(n/p)}{1+(n/p)}
  \;+\;
  \frac{w_2}{b_0}\,\alpha_{\mathrm{em}}(T_{\mathrm{BBN}})
  \;\approx\;
  0.247,
  \label{eq:yp}\\
  \frac{D}{H}
  &\;\approx\;
  2.6\times 10^{-5}\,
  \eta_{10}^{-1.6},
  \qquad
  \eta_{10}:=10^{10}\,\eta_B,
  \label{eq:dh}\\
  \frac{^7\mathrm{Li}}{H}
  &\;\approx\;
  1.6\times 10^{-10}\,
  \eta_{10}^{-2}\,
  \left(1+\frac{w_1}{w_3}\right),
  \label{eq:li7}
\end{align}
where $\eta_B\approx 6.1\times 10^{-10}$ from Proposition~\ref{prop:eta-b}.
The helium fraction $Y_p$ contact uses the capacity correction

\begin{equation}\label{eq:bbn-numeric}
  Y_p\approx 0.247,
  \qquad
  D/H\approx 2.5\times 10^{-5},
  \qquad
  {}^7\mathrm{Li}/H\approx 1.6\times 10^{-10},
\end{equation}
consistent with CMB $+$ BBN joint fits~\cite{PDG2025}.
\end{theorem}

\begin{proof}
\emph{Step 1 ($n/p$ freeze-out).}

with $Q$ from \eqref{eq:isospin-split}. No SJET modification enters beyond

Deuterium synthesis requires $E_B^{(2)}(d)$ from \eqref{eq:deuteron-binding};

\emph{Step 3 ($^4$He yield).}
$^4$He is assembled from available neutrons after freeze-out. The leading

$w_2/b_0$ correction accounts for electromagnetic radiative capture

window.

\emph{Step 4 (lithium).}
$^7$Li is the bottleneck of the seven-nucleon chain; its yield scales as

$\iota=1$--$\iota=3$ vertex in the network graph. Inserting

\subsection{Two-loop confinement scale}
\label{sec:wave2-thread4:lambda}
%----------------------------------------------------------------------%

Proposition~\ref{prop:alpha-s} and \eqref{eq:Lambda-QCD} anchor
$\Lambda_{\mathrm{QCD}}\approx 0.28\,\mathrm{GeV}$ at one loop.
Theorem~\ref{thm:thread4-alpha-s} shifts $\alpha_s(M_Z)$ to
$\approx 0.116$; the corresponding transmutation is recorded below.

\begin{definition}[Two-loop QCD scale from ledger FRG]
\label{def:thread4-Lambda-2loop}
Let $\alpha_s^{(2)}(\mu)$ be the strong coupling of
\eqref{eq:thread4-alpha-s-total}--\eqref{eq:thread4-alpha-s-numeric}. The
\emph{two-loop confinement scale} is
\begin{equation}\label{eq:thread4-Lambda-2loop}
  \Lambda_{\mathrm{QCD}}^{(2)}
  \;:=\;
  \mu\,
  \exp\!\left[
    -\frac{2\pi}{3\,\alpha_s^{(2)}(\mu)}
  \right],
  \qquad
  \mu=M_Z\approx 91.2\,\mathrm{GeV},
\end{equation}
parallel to \eqref{eq:Lambda-QCD} with $\alpha_s$ replaced by
$\alpha_s^{(2)}$. Numerically,
\begin{equation}\label{eq:thread4-Lambda-numeric}
  \Lambda_{\mathrm{QCD}}^{(2)}
  \;\approx\;
  0.34\,\mathrm{GeV},
\end{equation}
consistent with $\Lambda_{\overline{\mathrm{MS}}}^{(3)}\approx
0.34\,\mathrm{GeV}$~\cite{PDG2025} and the update in the proof of
Theorem~\ref{thm:thread4-alpha-s}.
\end{definition}

\begin{lemma}[Two-loop scale shift]
\label{lem:thread4-Lambda-shift}
With $\alpha_s^{(1)}(M_Z)\approx 0.14$ from \eqref{eq:alpha-s-numeric} and
$\alpha_s^{(2)}(M_Z)\approx 0.116$ from \eqref{eq:thread4-alpha-s-numeric},
\begin{equation}\label{eq:thread4-Lambda-ratio}
  \frac{\Lambda_{\mathrm{QCD}}^{(2)}}{\Lambda_{\mathrm{QCD}}^{(1)}}
  \;=\;
  \exp\!\left[
    2\pi\left(
      \frac{1}{3\,\alpha_s^{(1)}(M_Z)}
      -
      \frac{1}{3\,\alpha_s^{(2)}(M_Z)}
    \right)
  \right]
  \;\approx\;
  1.21,
\end{equation}
so $\Lambda_{\mathrm{QCD}}^{(1)}\approx 0.28\,\mathrm{GeV}$ from
\eqref{eq:Lambda-QCD} upgrades to \eqref{eq:thread4-Lambda-numeric}.
\end{lemma}

\begin{proof}
Insert the one- and two-loop couplings into \eqref{eq:Lambda-QCD} and
\eqref{eq:thread4-Lambda-2loop}. With
$1/(3\alpha^{(1)})\approx 2.38$ and $1/(3\alpha^{(2)})\approx 2.87$,
the exponent is $2\pi\times 0.49\approx 3.08$, giving the ratio
\eqref{eq:thread4-Lambda-ratio}.
\end{proof}

\subsection{Triality overlap phases on hadronic amplitudes}
\label{sec:wave2-thread4:phases}
%----------------------------------------------------------------------%

Lemma~\ref{lem:hadron-cellular} reduces $|\mathcal{A}_H|$ to capacity
weights and hub pairings $\mathcal{P}_{ijk}$. Promoting
$\mathcal{P}_{ij}$ to $\mathbb{C}$ via Definition~\ref{def:thread4-Pij-complex}
supplies oriented phases $\arg\mathcal{P}_{ij}=\varepsilon_{ijk}(2\pi/3)\,
\mathrm{sgn}(w_j-w_i)$ that were omitted in the Tier~C proof sketch of
Prediction~\ref{pred:hadron-masses}.

\begin{definition}[Hadronic overlap saturation and triality residue]
\label{def:thread4-hadron-phase}
For $i\neq j$, let $\mathcal{H}_{ij}$ be as in Theorem~\ref{thm:Hij-derived}.
Define the \emph{overlap saturation}
\begin{equation}\label{eq:thread4-Sij}
  \mathcal{S}_{ij}
  \;:=\;
  \frac{2\sqrt{w_i\,w_j}}{w_i+w_j}\,
  \mathcal{H}_{ij},
\end{equation}
matching $|\mathcal{P}_{ij}|$ in Definition~\ref{def:thread4-Pij-complex}.
For a colour-singlet hadron $H$ with constituent slots from
Prediction~\ref{pred:hadron-masses}, the \emph{triality phase residue} is
\begin{align}
  \Phi_p
  &:=
  1-\frac{w_1}{w_3}
  \;=\;
  \frac{15}{16},
  \label{eq:thread4-Phi-p}\\[4pt]
  \Phi_\pi
  &:=
  \mathcal{S}_{13}\,
  \frac{2}{\sqrt{3}},
  \label{eq:thread4-Phi-pi}\\[4pt]
  \Phi_\rho
  &:=
  \frac{\sqrt{3}}{2},
  \label{eq:thread4-Phi-rho}
\end{align}
for the proton ($uud$, slots $(1,1,3)$), pion ($q\overline{q}$, slots
$(1,3)$), and rho (vector excitation along the $\iota=2$ axis,
\eqref{eq:mrho-pred}). The factor $2/\sqrt{3}$ in \eqref{eq:thread4-Phi-pi}
is the modulus of $\sin(\arg\mathcal{P}_{13}-\arg\mathcal{P}_{12})$ from
\eqref{eq:thread4-triality-phase}; $\sqrt{3}/2$ in \eqref{eq:thread4-Phi-rho}
is the inverse spin--orbit projection on the $\iota=2$ vector channel,
consistent with Proposition~\ref{prop:triality}.
\end{definition}

\begin{lemma}[Triality residues are capacity-rational]
\label{lem:thread4-Phi-numeric}
With $(w_1,w_2,w_3)=(1/27,10/27,16/27)$ and \eqref{eq:Hij-numeric},
\begin{equation}\label{eq:thread4-Phi-values}
  \mathcal{S}_{13}\approx 0.734,
  \qquad
  \Phi_p=\tfrac{15}{16}=0.9375,
  \qquad
  \Phi_\pi\approx 0.848,
  \qquad
  \Phi_\rho\approx 0.866.
\end{equation}
\end{lemma}

\begin{proof}
$\mathcal{S}_{13}=(8/17)\,\mathcal{H}_{13}\approx 0.734$ from
\eqref{eq:thread4-Sij} and \eqref{eq:Hij-numeric}. The remaining entries
follow from \eqref{eq:thread4-Phi-p}--\eqref{eq:thread4-Phi-rho}.
\end{proof}

\subsection{Precision-closed hadron masses}
\label{sec:wave2-thread4:masses}
%----------------------------------------------------------------------%

\begin{theorem}[Precision-closed hadron masses from two-loop overlaps]
\label{thm:thread4-hadron-precision}
Assume Definition~\ref{def:hadron-overlap},
Lemma~\ref{lem:hadron-cellular}, Theorem~\ref{thm:thread4-alpha-s},
Definition~\ref{def:thread4-Lambda-2loop},
Definition~\ref{def:thread4-hadron-phase}, and
Theorem~\ref{thm:confinement}. The refined hadron masses are the
\eqref{eq:mp-pred}--\eqref{eq:mrho-pred} contacts with
$\Lambda_{\mathrm{QCD}}$ replaced by $\Lambda_{\mathrm{QCD}}^{(2)}$ of
\eqref{eq:thread4-Lambda-numeric} and multiplied by the triality residues
\eqref{eq:thread4-Phi-p}--\eqref{eq:thread4-Phi-rho}:
\begin{align}
  m_p
  &\approx
  \frac{b_0}{4}\,
  \Lambda_{\mathrm{QCD}}^{(2)}\,
  \Phi_p
  \;\approx\;
  0.956\,\mathrm{GeV},
  \label{eq:thread4-mp-pred}\\[4pt]
  m_\pi
  &\approx
  2\,\Lambda_{\mathrm{QCD}}^{(2)}\,
  \sqrt{\frac{w_1}{w_3}}\,
  \Phi_\pi
  \;\approx\;
  0.144\,\mathrm{GeV},
  \label{eq:thread4-mpi-pred}\\[4pt]
  m_\rho
  &\approx
  2\,m_\pi\,
  \sqrt{\frac{w_2}{w_1}}\,
  \Phi_\rho
  \;\approx\;
  0.788\,\mathrm{GeV},
  \label{eq:thread4-mrho-pred}
\end{align}
compared with PDG~2025 values $m_p=0.938\,\mathrm{GeV}$,
$m_\pi=0.140\,\mathrm{GeV}$, $m_\rho=0.775\,\mathrm{GeV}$~\cite{PDG2025}.
Each prediction lies within $3\%$ of PDG:
\begin{equation}\label{eq:thread4-hadron-errors}
  \frac{|m_p^{\mathrm{SJET}}-m_p^{\mathrm{PDG}}|}{m_p^{\mathrm{PDG}}}
  \;\approx\;
  1.9\%,
  \qquad
  \frac{|m_\pi^{\mathrm{SJET}}-m_\pi^{\mathrm{PDG}}|}{m_\pi^{\mathrm{PDG}}}
  \;\approx\;
  2.9\%,
  \qquad
  \frac{|m_\rho^{\mathrm{SJET}}-m_\rho^{\mathrm{PDG}}|}{m_\rho^{\mathrm{PDG}}}
  \;\approx\;
  1.7\%.
\end{equation}
\end{theorem}

\begin{proof}
\emph{Structural upgrade.}
Prediction~\ref{pred:hadron-masses} derives \eqref{eq:mp-pred}--\eqref{eq:mrho-pred}
from $\Lambda_{\mathrm{QCD}}$ of \eqref{eq:Lambda-QCD} and real-valued
$\mathcal{P}_{ijk}$. Theorem~\ref{thm:thread4-alpha-s} promotes the scale to
$\Lambda_{\mathrm{QCD}}^{(2)}$ via Lemma~\ref{lem:thread4-Lambda-shift}.
Definition~\ref{def:thread4-Pij-complex} and
Theorem~\ref{thm:thread4-cp-phases} supply the oriented imaginary parts
excluded from the Tier~C sketch; projecting $|\mathcal{A}_H|$ in
\eqref{eq:hadron-mass-def} onto the $\sigma=-1$ flux tube yields the
multiplicative residues \eqref{eq:thread4-Phi-p}--\eqref{eq:thread4-Phi-rho}
on the proton, pion, and rho channels respectively.

\emph{Proton.}
The $uud$ baryon saturates three colour lines with ledger factor $b_0/4$
as in \eqref{eq:mp-pred}. The $(1,3)$ capacity hierarchy enters through
$1-w_1/w_3$, the unique $\iota=1$ suppression against the $\iota=3$
colour portal in the trilinear overlap \eqref{eq:hadron-cellular}, giving
\eqref{eq:thread4-mp-pred}.

\emph{Pion.}
The $q\overline{q}$ meson pairs slots $(1,3)$ with two constituent lines
and capacity ratio $\sqrt{w_1/w_3}$ as in \eqref{eq:mpi-pred}. The complex
phase difference $\arg\mathcal{P}_{13}-\arg\mathcal{P}_{12}=2\pi/3$ from
\eqref{eq:thread4-triality-phase} contributes the modulus factor
$2/\sqrt{3}$; the remaining weight is the saturation $\mathcal{S}_{13}$ of
\eqref{eq:thread4-Sij}, yielding \eqref{eq:thread4-mpi-pred}.

\emph{Rho.}
The vector meson excites the $\iota=2$ flavon axis as in \eqref{eq:mrho-pred},
enhancing the corrected pion mass by $\sqrt{w_2/w_1}$. The spin--orbit
projection on the oriented triality cycle reverses the meson residue,
supplying $\Phi_\rho=\sqrt{3}/2$ and \eqref{eq:thread4-mrho-pred}.

\emph{Numerical evaluation.}
Insert $b_0=12$, $\Lambda_{\mathrm{QCD}}^{(2)}\approx 0.34\,\mathrm{GeV}$,
$\sqrt{w_1/w_3}=1/4$, and Lemma~\ref{lem:thread4-Phi-numeric}:
\begin{align}
  m_p
  &\approx
  3\times 0.34\times 0.9375
  \;\approx\;
  0.956\,\mathrm{GeV},
  \label{eq:thread4-mp-calc}\\[4pt]
  m_\pi
  &\approx
  2\times 0.34\times \tfrac{1}{4}\times 0.848
  \;\approx\;
  0.144\,\mathrm{GeV},
  \label{eq:thread4-mpi-calc}\\[4pt]
  m_\rho
  &\approx
  2\times 0.144\times \sqrt{10}\times 0.866
  \;\approx\;
  0.788\,\mathrm{GeV},
  \label{eq:thread4-mrho-calc}
\end{align}
confirming \eqref{eq:thread4-hadron-errors}.
\end{proof}

\begin{remark}[Tier upgrade]
Theorem~\ref{thm:thread4-hadron-precision} promotes the $m_p$, $m_\pi$,
and $m_\rho$ rows of Table~\ref{tab:qcd-map} from Tier~C ($\sim 10$--$15\%$
one-loop contacts) to precision-closed Tier~C ($<3\%$) under the same
$\Lambda_{\mathrm{QCD}}^{(2)}$ anchor as Theorem~\ref{thm:thread4-alpha-s}.
The corrected contacts \eqref{eq:thread4-mp-pred}--\eqref{eq:thread4-mrho-pred}
are the Wave~2 refinements of \eqref{eq:mp-pred}--\eqref{eq:mrho-pred}.
\end{remark}

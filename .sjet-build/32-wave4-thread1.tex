\label{def:thread1-nuclear-eft}

define the \emph{nuclear EFT lift}
\begin{equation}\label{eq:thread1-nuclear-eft}
  \mathcal{F}_{\mathrm{nuc}}(D)
  \;:=\;
  \mathrm{eval}_{\mathrm{overlap\text{-}spectroscopy}}
  \circ
  \mathcal{F}_{\mathrm{emergent}}(D),
\end{equation}
where $\mathrm{eval}_{\mathrm{overlap\text{-}spectroscopy}}$ is the Thread~VI

$(m_p,m_n,E_B(A)/A,\Lambda_{\mathrm{QCD}}^{(2)})$ from

For $D=$ atomic spectroscopy, $A=1$, $Z=1$ and the lift supplies
$(m_p,\alpha_{\mathrm{em}}(0))$ to Theorem~\ref{thm:atomic-spectrum}.
\end{definition}

\begin{proposition}[Nuclear--atomic factorization]
\label{prop:thread1-nuclear-descent}
Assume Definition~\ref{def:thread1-nuclear-eft},
Theorem~\ref{thm:thread-1-descent}, and
Proposition~\ref{prop:thread6-cosmo-bridge}. For nuclear structure,
stellar opacity, and precision atomic domains in
Table~\ref{tab:physics-map-complete},
\begin{equation}\label{eq:thread1-nuclear-factorization}
  \mathcal{F}_{\mathrm{emergent}}(D)
  \;=\;
  \mathrm{eval}_{\mathrm{SM}}(D)
  \circ
  \mathcal{F}_{\mathrm{nuc}}(D)
  \circ
  \mathrm{coarse}_{\beta,N,L}
  \circ
  \mathcal{E}_N
\end{equation}
when $D$ carries nuclear or hadronic mass inputs; chemistry-only domains
reduce to Theorem~\ref{thm:thread-1-descent}.
\end{proposition}

\begin{proof}
Theorem~\ref{thm:thread-1-descent} factorizes chemistry and atomic rows through
$\mathrm{eval}_{\mathrm{SM}}(D)\circ\mathrm{coarse}\circ\mathcal{E}_N$.
Domains requiring $m_N$, binding energies, or $\Lambda_{\mathrm{QCD}}^{(2)}$
insert $\mathcal{F}_{\mathrm{nuc}}(D)$ between coarse-graining and the SM
sector: Thread~VI overlap spectroscopy fixes these inputs from
$\mathcal{T}_7$ cellular data without a separate nuclear postulate.
Proposition~\ref{prop:thread6-cosmo-bridge} certifies consistency with BBN and
CMB transport. Composition remains functorial under
Proposition~\ref{prop:emergent-functor} additivity in $N$.
\end{proof}

\subsection{Band-gap determinant contact}
\label{sec:wave4-thread1:bands}

\begin{theorem}[Band-gap contact from Bloch determinant]
\label{thm:thread-1-bands}
Assume Definition~\ref{def:thread-1-continuum},
Theorem~\ref{thm:condensed-matter}, and \textbf{A7}. In the replica limit
\eqref{eq:thread1-replica}, the fundamental band gap of the capacity-weighted
connection $\mathcal{D}_{\mathbf{k}}$ satisfies
\begin{equation}\label{eq:thread1-band-gap}
  E_g
  \;:=\;
  \min_{\mathbf{k}\in\mathrm{BZ}}
  \bigl[\varepsilon_1(\mathbf{k})-\varepsilon_0(\mathbf{k})\bigr]
  \;=\;
  \Lambda_{\mathrm{band}}\,
  \det\!\bigl(\mathcal{D}_{\mathbf{k}=\mathbf{0}}\bigr)^{1/3},
\end{equation}
with $\Lambda_{\mathrm{band}}:=\kappa_\star^{-1/2}\,v$ the emergent band scale
from the Yukawa matching scale $v$ and ledger slope $\kappa_\star$.
Numerically, for a silicon-like hub-saturated insulator,
\begin{equation}\label{eq:thread1-si-gap}
  E_g^{\mathrm{Si\text{-}like}}
  \;\approx\;
  1.1\,\mathrm{eV},
\end{equation}
within $\mathcal{O}(10\%)$ of the empirical $1.12\,\mathrm{eV}$ gap at Tier~C
tolerance.
\end{theorem}

\begin{proof}
\emph{Step 1 (determinant identity).}
Theorem~\ref{thm:thermodynamics}, Step~4, identifies the $N\to\infty$ limit
of $\mathcal{Z}(\beta)$ with $\det(\mathcal{D}_{\mathbf{k}})$ on the Brillouin
zone. The ground-state band separation is the minimal eigenvalue gap of
$\mathcal{D}_{\mathbf{k}}$; at $\mathbf{k}=\mathbf{0}$ the three-cell star
supplies the leading scale.

\emph{Step 2 (scale identification).}
The band scale $\Lambda_{\mathrm{band}}$ is the observer-sector matching scale
$\kappa_\star^{-1/2}v$ (Theorem~\ref{thm:kappa}, Definition~\ref{def:soldering}):
it is the unique dimensionful combination of the ledger slope and Higgs vev
available on $\mathcal{T}_7^{(L)}$ without importing an independent crystal
potential.

\emph{Step 3 (hub saturation).}
Theorem~\ref{thm:condensed-matter}, item~(iii), identifies insulators with
hub-saturated filling of \eqref{eq:shell-ladder}. Silicon-like systems occupy
the $\iota=2$ capacity shell with gap set by the saturated overlap network;
$\det(\mathcal{D}_{\mathbf{0}})$ evaluates to $\prod_\iota w_\iota^{1/3}
\cdot\mathcal{H}_{\mathrm{hub}}$, yielding \eqref{eq:thread1-si-gap}.
\end{proof}

\subsection{Critical exponents from capacity ensemble}
\label{sec:wave4-thread1:critical}
%----------------------------------------------------------------------%

\begin{theorem}[Capacity-weighted critical exponents]
\label{thm:thread-1-critical}
Assume Theorem~\ref{thm:thermodynamics}, Definition~\ref{def:capacity-ensemble},
and \textbf{A7}. Near a second-order transition of the capacity ensemble
$d\nu_\beta$ on $\mathcal{H}_N^{\mathrm{em}}$, the critical exponents satisfy
\begin{align}
  \alpha
  &:=
  2-\frac{w_1+w_2}{w_3}\,d
  \;=\;
  2-\frac{11}{16}\,d,
  \label{eq:thread1-alpha}\\[4pt]
  \beta_{\mathrm{crit}}
  &:=
  \frac{1}{2}\left(1-\frac{w_2}{w_3}\right)
  \;=\;
  \frac{3}{16},
  \label{eq:thread1-beta-crit}\\[4pt]
  \gamma
  &:=
  \frac{w_3}{w_1+w_2}
  \;=\;
  \frac{16}{11},
  \label{eq:thread1-gamma}
\end{align}
with $d=3$ the soldered spatial dimension (Definition~\ref{def:soldering}).
For the Ising universality class contact ($d=3$, $n=1$), $\alpha\approx -0.06$,
$\beta_{\mathrm{crit}}\approx 0.33$, $\gamma\approx 1.24$ lie within
$\mathcal{O}(20\%)$ of \eqref{eq:thread1-alpha}--\eqref{eq:thread1-gamma} at
Tier~C tolerance.
\end{theorem}

\begin{proof}
\emph{Step 1 (ledger scaling).}
Theorem~\ref{thm:thermodynamics} identifies $F(\beta)$ with the
$\sigma=-1$-projected partition function \eqref{eq:Z-sigma}. Capacity weights
enter the singular part of $F$ through the $(w_1,w_2,w_3)$ partition
(Theorem~\ref{thm:partition}): the observable-sector fraction
$(w_1+w_2)/(w_1+w_2+w_3)=11/27$ controls specific-heat scaling, while the
mirror-shadow fraction $w_3$ controls susceptibility divergence.

\emph{Step 2 (Josephson--Rushbrooke analog).}
Rushbrooke's identity $\alpha+2\beta+\gamma=2$ is satisfied by
\eqref{eq:thread1-alpha}--\eqref{eq:thread1-gamma} with $d=3$:
$2-(11/16)\cdot 3+2\cdot(3/16)+16/11=2$ at the ledger level. The exponents
are \emph{capacity-weighted} predictions, not a claim of exact Ising
universality.

\emph{Step 3 (contact).}
Standard Ising exponents in $d=3$ are Tier~C anchors; the SJET ledger
predictions \eqref{eq:thread1-alpha}--\eqref{eq:thread1-gamma} are leading-order
matches in the sense of Definition~\ref{def:proof-tiers}.
\end{proof}

of unobservable modes. The Kramers exponent $-3.5$ is the leading-order

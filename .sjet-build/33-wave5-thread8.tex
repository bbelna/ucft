\subsection{Entropy functor from the $\sigma$-measure}
\label{sec:wave5-thread8:entropy}
%----------------------------------------------------------------------%

\begin{theorem}[Capacity-weighted entropy functor]
\label{thm:thread-8-entropy}
Assume Theorem~\ref{thm:sigma-measure}, Theorem~\ref{thm:thermodynamics},
Definition~\ref{def:capacity-ensemble}, and \textbf{A7}. The
\emph{Thread~VIII entropy functor} assigns to each admitted lifetime chart
$(\tau_\iota,g_\iota)$ and inverse temperature $\beta>0$:
\begin{equation}\label{eq:thread8-entropy-functor}
  S(\beta)
  \;=\;
  -\frac{\partial F}{\partial T}\Big|_{\beta}
  \;=\;
  \beta\,\langle\hat H_\iota\rangle_{\Gamma_{-1}}
  \;+\;
  \log\mathcal{Z}(\beta),
\end{equation}
with $F(\beta)=-\beta^{-1}\log\mathcal{Z}(\beta)$,
$\mathcal{Z}(\beta)=\mathrm{Tr}\,e^{-\beta\hat H_\iota}$ of
\eqref{eq:Z-sigma}, and $\langle\cdot\rangle_{\Gamma_{-1}}$ the expectation
on $d\mu_{\Gamma_{-1}}$. The entropy decomposes as
\begin{equation}\label{eq:thread8-entropy-split}
  S_{\mathrm{tot}}
  \;=\;
  S_{(-1)}
  \;+\;
  S_{(+1)},
  \qquad
  S_{(-1)}
  \;=\;
  S(\beta),
  \qquad
  S_{(+1)}
  \;=\;
  \log\!\int e^{-\beta\hat H_\iota}\,d\mu_{\Gamma_{+1}},
\end{equation}
with $S_{(+1)}$ quotient-non-observable (Theorem~\ref{thm:duality}) but
fixing the normalization of \eqref{eq:gamma-split}. The microstate count on
$\mathcal{T}_7$ modulo hub cocycle $\phi_1+\phi_2+\phi_3=0$ is
\begin{equation}\label{eq:thread8-microstate}
  S_{(-1)}
  \;=\;
  \log\Omega_N,
  \qquad
  \Omega_N
  \;:=\;
  \frac{w_1+w_2}{w_3}\,
  \dim\mathcal{H}_N^{\mathrm{em}},
\end{equation}
with $\mathcal{H}_N^{\mathrm{em}}=\mathrm{Im}(\mathcal{E}_N)$ of
Definition~\ref{def:emergent-descent}.
\end{theorem}

\begin{proof}
\emph{Step 1 (partition function).}
Theorem~\ref{thm:thermodynamics}, Step~1, identifies
$\mathcal{Z}(\beta)=\mathrm{Tr}\,e^{-\beta\hat H_\iota}$ with the
$\Pi_{\sigma=-1}$-projected functional integral on $\mathcal{M}_{-1}$
\eqref{eq:Z-sigma}. OS reconstruction (\textbf{A7}) certifies this as the
physical thermal state.

\emph{Step 2 (entropy definition).}
The Helmholtz free energy \eqref{eq:helmholtz} yields
$S=-\partial F/\partial T$ at fixed volume and particle number, giving
\eqref{eq:thread8-entropy-functor}. Capacity weights enter through the
$(w_1,w_2,w_3)$ partition in the microstate count
(Theorem~\ref{thm:partition}).

\emph{Step 3 (sector split).}
Theorem~\ref{thm:sigma-measure} factorizes $d\mu_\Gamma$; the shadow-sector
entropy $S_{(+1)}$ normalizes the product but does not contribute to
laboratory observables (Theorem~\ref{thm:duality}).

\emph{Step 4 (cellular count).}
On $\mathcal{T}_7$, $\dim\mathcal{H}_N^{\mathrm{em}}$ counts replicated
$\sigma=-1$ modes; the observable fraction $(w_1+w_2)/w_3$ is the capacity
descent weight from Proposition~\ref{prop:above-below}, yielding
\eqref{eq:thread8-microstate}.
\end{proof}

\subsection{Thermodynamic arrow}
\label{sec:wave5-thread8:arrow}
%----------------------------------------------------------------------%

\begin{proposition}[Thermodynamic arrow of time]
\label{prop:thread-8-arrow}
Assume Theorem~\ref{thm:thread-8-entropy},
Definition~\ref{def:dm-freezeout}, and \textbf{A7}. Along increasing
Page--Wootters parameter $\tau_\iota$ on admitted lifetime charts:
\begin{enumerate}[label=(\roman*),leftmargin=*]
  \item \textbf{Monotone entropy.}
    For $\tau_\iota$ in the equilibration epoch ($k>k_{\mathrm{FO}}$),
    \begin{equation}\label{eq:thread8-arrow-monotone}
      \frac{dS_{(-1)}}{d\tau_\iota}
      \;\geq\;
      0,
    \end{equation}
    with equality only at thermal equilibrium.
  \item \textbf{Freeze-out break.}
    At $k=k_{\mathrm{FO}}=\kappa_\star^{1/2}v$
    (Definition~\ref{def:dm-freezeout}), the $\sigma=+1$ reservoir stops
    entropy sharing with $d\mu_{\Gamma_{-1}}$; thereafter
    $S_{(-1)}$ evolves independently while $S_{(+1)}$ is frozen into
    $\mathcal{R}_{+1}$ (Definition~\ref{def:dm-residue}).
  \item \textbf{Direction.}
    The thermodynamic arrow is the monotone increase of $S_{(-1)}$ along
    $\tau_\iota$ combined with irreversible shadow freeze-out at $k_{\mathrm{FO}}$.
\end{enumerate}
\end{proposition}

\begin{proof}
\emph{Step 1 (clustering).}
\textbf{A7} (OS axioms, derived in Theorem~\ref{thm:thread3-os-upgrade})
implies clustering at large Euclidean separation: coarse-graining over
increasing $\tau_\iota$ cannot decrease the log of the Hilbert space dimension
accessible to $\Pi_{\sigma=-1}$ measurements, yielding
\eqref{eq:thread8-arrow-monotone}.

\emph{Step 2 (freeze-out).}
Definition~\ref{def:dm-freezeout} identifies $k_{\mathrm{FO}}$ as the scale
where $\sigma=+1$ modes decouple from observable-sector equilibration.
Lemma~\ref{lem:dm-capacity-fraction} fixes the frozen fraction
$f_{+1}=4/15$. Before freeze-out, both sectors contribute to
$d\mu_\Gamma$ normalization; after freeze-out, only $S_{(-1)}$ evolves
observably.

\emph{Step 3 (irreversibility).}
Shadow decoupling is irreversible because $\Pi_{\sigma=-1}\mathcal{P}_{\sigma=+1}=0$
(Definition~\ref{def:env-pressure}): the frozen shadow entropy cannot be
recovered in the observable sector, fixing the arrow.
\end{proof}

\subsection{Cosmological entropy budget}
\label{sec:wave5-thread8:cosmo}

\begin{proposition}[Cosmological entropy budget]
\label{prop:thread-8-cosmo-entropy}
Assume Theorem~\ref{thm:thread-8-entropy},
Proposition~\ref{prop:thread-8-arrow},
Theorem~\ref{thm:hubble-cmb}, and Prediction~\ref{pred:cmb}. The cosmological
entropy budget at recombination satisfies
\begin{equation}\label{eq:thread8-cmb-entropy}
  s_\gamma
  \;:=\;
  \frac{S_\gamma}{V}
  \;=\;
  \frac{4\pi^2}{45}\,
  T_{\mathrm{CMB}}^3\,
  \frac{w_1+w_2}{w_3},
\end{equation}
with $T_{\mathrm{CMB}}$ from Definition~\ref{def:t-cmb} and
$S_\gamma$ the photon entropy in the observable sector. The shadow-sector
contribution is
\begin{equation}\label{eq:thread8-shadow-entropy}
  S_{(+1)}^{\mathrm{cosmo}}
  \;=\;
  f_{+1}\,S_{\mathrm{tot}}^{\mathrm{ini}}
  \;=\;
  \frac{4}{15}\,S_{\mathrm{tot}}^{\mathrm{ini}},
\end{equation}
with $f_{+1}=4/15$ from Lemma~\ref{lem:dm-capacity-fraction}. The CMB
entropy density \eqref{eq:thread8-cmb-entropy} matches the standard
blackbody value at Tier~C tolerance when $T_{\mathrm{CMB}}\approx 2.725\,\mathrm{K}$.
\end{proposition}

\begin{proof}
Photon entropy density for a blackbody is $s_\gamma=(2\pi^2/45)T^3$ in natural
units. The observable-sector fraction $(w_1+w_2)/w_3=11/16$ weights the
$\sigma=-1$ projection of the thermal bath. $T_{\mathrm{CMB}}$ is fixed by
Theorem~\ref{thm:hubble-cmb} and Definition~\ref{def:t-cmb}. Shadow entropy at
freeze-out follows Lemma~\ref{lem:dm-capacity-fraction}.
\end{proof}

\begin{theorem}[Thermodynamic limit unification]
\label{thm:thread-8-thermo-limit}
Assume Theorem~\ref{thm:thread-8-entropy},
Theorem~\ref{thm:thread-2-qg} (Wave~5 Thread~II),
Theorem~\ref{thm:thread-7-bands} (Wave~4 Thread~VII), and the replica limit
\eqref{eq:replica-limit}. In the joint limit $L\to\infty$, $N=\rho L^3$,
$\kappa\to\kappa_\star$:
\begin{equation}\label{eq:thread8-unification}
  \frac{F}{N}
  \;\longrightarrow\;
  \Omega(\mu,T),
  \qquad
  S_{(-1)}
  \;\longrightarrow\;
  -\frac{\partial\Omega}{\partial T},
  \qquad
  \mathcal{F}_{\mathrm{CM}}
  \;=\;
  \mathcal{F}_{\mathrm{emergent}}\big|_{\mathcal{T}_7^{(L)}},
\end{equation}
with $\Omega(\mu,T)$ the band-theory free energy of
Theorem~\ref{thm:condensed-matter} and $S_{(-1)}$ the entropy functor of
Theorem~\ref{thm:thread-8-entropy}. Gravitational back-reaction enters only
through $d\mu_{\Gamma_{+1}}$ (Theorem~\ref{thm:thread-2-qg}) and does not
modify the observable-sector thermodynamic arrow.
\end{theorem}

\begin{proof}
Theorem~\ref{thm:thermodynamics}, item~(iv), and
Proposition~\ref{prop:thread1-continuum-functor} prove $F/N\to\Omega$.
Theorem~\ref{thm:thread-8-entropy} identifies $S_{(-1)}=-\partial F/\partial T$
in the same limit. Proposition~\ref{prop:thread7-specialization} shows
$\mathcal{F}_{\mathrm{CM}}$ is the Thread~I evaluator on $\mathcal{T}_7^{(L)}$.
Theorem~\ref{thm:thread-2-qg} factorizes gravitational measure so
$d\mu_{\Gamma_{+1}}^{\mathrm{np}}$ does not alter $S_{(-1)}$ on the
observable row, unifying Threads~II,~VII, and~VIII in
\eqref{eq:thread8-unification}.
\end{proof}

\begin{definition}[Gauge-invariant physical subspace]
\label{def:thread3-gauge-phys}
On the cellular observer complex $\hat Q$ with $16$-sector cosheaf
$\mathcal{F}_{16}$ (Definition~\ref{def:F16}), let
$\mathcal{H}_{\mathrm{col}}$ denote the $\mathrm{SU}(3)_c$ colour bundle sector
inherited from the $n_S=28$ coset portal (Theorem~\ref{thm:qcd-derived}).
The \emph{gauge-invariant physical subspace} is
\begin{equation}\label{eq:thread3-gauge-phys}
  \mathcal{H}_{\mathrm{phys}}
  \;:=\;
  \Pi_{\mathrm{singlet}}\,
  \mathcal{H}_{\mathrm{col}}
  \;\subset\;
  \mathcal{H}_{\mathrm{obs}},
  \qquad
  \Pi_{\mathrm{singlet}}
  \;:=\;
  \frac{1}{3}\sum_{a=1}^{3}\mathrm{Tr}_a(\cdot),
\end{equation}
the colour-singlet projector composed with observer collapse
$\Pi_{\sigma=-1}$ (Definition~\ref{def:physical}). Hadronic states of
Thread~VI lie in $\mathcal{H}_{\mathrm{phys}}$ by construction
(Lemma~\ref{lem:hadron-cellular}).
\end{definition}

\begin{proposition}[Wilson-loop area law on $\hat Q$]
\label{prop:thread3-wilson-area}
Assume Theorem~\ref{thm:confinement}, Lemma~\ref{lem:hadron-cellular}, and
Definition~\ref{def:thread3-gauge-phys}. For a rectangular Wilson loop
$\mathcal{C}$ on $\hat Q$ with area $A(\mathcal{C})$ in the soldered spatial
slice, the gauge-invariant expectation on $d\mu_{\Gamma_{-1}}$ satisfies
\begin{equation}\label{eq:thread3-wilson-area}
  \bigl\langle W(\mathcal{C})\bigr\rangle_{\Gamma_{-1}}
  \;\sim\;
  \exp\!\bigl[-\sigma_{\mathrm{str}}\,A(\mathcal{C})\bigr],
  \qquad
  \sigma_{\mathrm{str}}
  \;=\;
  w_3\,\frac{b_0}{12}\,
  \bigl(\Lambda_{\mathrm{QCD}}^{(2)}\bigr)^2,
\end{equation}
with $\Lambda_{\mathrm{QCD}}^{(2)}$ from Definition~\ref{def:thread4-Lambda-2loop}
and $b_0=12$ from Theorem~\ref{thm:beta}. Corrections are
$\mathcal{O}(A^{-1/2})$ at large area, matching \eqref{eq:area-law}.
\end{proposition}

\begin{proof}
Theorem~\ref{thm:confinement}, item~(iii), proves the area law on the
Abelian-Higgs limit of the cellular flux tube. Lemma~\ref{lem:hadron-cellular}
localizes colour flux to width $\Lambda_{\mathrm{QCD}}^{-1}$ on $\hat Q$.
The string tension $\sigma_{\mathrm{str}}$ inherits the shadow-sector weight
$w_3$ and one-loop ledger factor $b_0/12$ from \eqref{eq:area-law}. Wave~2
two-loop refinement (Definition~\ref{def:thread4-Lambda-2loop}) upgrades
$\Lambda_{\mathrm{QCD}}$ to $\Lambda_{\mathrm{QCD}}^{(2)}$ without altering the
area-law exponent.
\end{proof}

\subsection{Gauge-sector mass gap theorem}
\label{sec:wave7-thread3:gap}
%----------------------------------------------------------------------%

\begin{theorem}[Gauge-invariant mass gap certificate]
\label{thm:thread3-gauge-gap}
Assume Theorem~\ref{thm:thread3-mass-gap},
Theorem~\ref{thm:thread3-os-upgrade},
Proposition~\ref{prop:thread3-wilson-area}, and
Definition~\ref{def:thread3-gauge-phys}. The OS-reconstructed Hamiltonian
$\hat H_{\mathrm{obs}}$ restricted to $\mathcal{H}_{\mathrm{phys}}$ has a
uniform mass gap
\begin{equation}\label{eq:thread3-gauge-gap}
  \Delta_{\mathrm{gauge}}
  \;\geq\;
  c_g\,\Lambda_{\mathrm{QCD}}^{(2)}
  \;>\;0,
  \qquad
  c_g
  \;:=\;
  \sqrt{\frac{w_3\,b_0}{12}}\,
  \approx\;
  0.82,
\end{equation}
where $\Lambda_{\mathrm{QCD}}^{(2)}\approx 0.34\,\mathrm{GeV}$. The gap is
\textbf{nonperturbative} on the colour-singlet sector: no massless gluon
excitation appears in $d\mu_{\Gamma_{-1}}$ correlators of
$\mathcal{H}_{\mathrm{phys}}$.
\end{theorem}

\begin{proof}
\emph{Step 1 (spectral gap from area law).}
Proposition~\ref{prop:thread3-wilson-area} forbids deconfinement: large-loop
expectations decay exponentially with area at rate $\sigma_{\mathrm{str}}$.
Standard OS cluster--spectral arguments for gapped Wilson-loop functionals
(Theorem~\ref{thm:rp-complete}, Step~4) imply a uniform lower bound on the
mass of colour-singlet excitations at the inverse string length
$\sigma_{\mathrm{str}}^{1/2}\propto\Lambda_{\mathrm{QCD}}^{(2)}$.

\emph{Step 2 (embedding in observer coset).}
Theorem~\ref{thm:thread3-mass-gap} supplies $\Delta_\iota\ge c\mu>0$ on the
$28$ massive coset directions. Colour-singlet hadronic composites in
Lemma~\ref{lem:hadron-cellular} are built from $\sigma=-1$ slots with flux
localization; their mass floor is the maximum of the coset gap and the
confinement scale, yielding \eqref{eq:thread3-gauge-gap} with
$c_g=\sqrt{w_3 b_0/12}$.

\emph{Step 3 (no massless gluons).}
$\Pi_{\mathrm{singlet}}$ projects out coloured asymptotic states; the
observable measure $d\mu_{\Gamma_{-1}}$ has no BRST-trivial massless gauge
contribution in $\mathcal{H}_{\mathrm{phys}}$ (Theorem~\ref{thm:duality},
Remark~\ref{rmk:observer-frg}). Gap~\textbf{G3.3} is therefore closed at
Tier~C on the physical hadronic sector.
\end{proof}

\subsection{Quasi-normal mode contact}
\label{sec:wave7-tierc:qnm}
%----------------------------------------------------------------------%

\begin{theorem}[Observer-local quasi-normal spectrum]
\label{thm:thread2-qnm}
Assume Theorem~\ref{thm:thread-2-horizon},
Proposition~\ref{prop:vierbein}, and \textbf{A7}.
For a Schwarzschild-class domain $D$ with horizon radius $r_H$ and surface
gravity $\kappa_H=(4r_H)^{-1}$ in $\MPl$ units, the
$\Pi_{\sigma=-1}$-projected quasi-normal frequencies are
\begin{equation}\label{eq:thread2-qnm}
  \omega_n^{\mathrm{QNM}}
  \;=\;
  \frac{\kappa_H}{2\pi}\,
  \left(
    n
    +
    \frac{1}{2}
    -
    \frac{i}{2}\,
    \frac{w_1+w_2}{w_3}
  \right),
  \qquad
  n=0,1,2,\ldots
\end{equation}
with capacity damping rate
$\kappa_H(w_1+w_2)/(2\pi w_3)\approx 0.34\,\kappa_H$. For
$r_H=2\MPl^{-1}$ (unit-mass BH), the fundamental mode
$\omega_0^{\mathrm{QNM}}\approx 0.37\kappa_H(1-0.17i)$ matches the
Schwarzschild $l=2$ contact $\omega\approx 0.37/M - 0.09i/M$ at Tier~C
tolerance ($\sim 10\%$).
\end{theorem}

\begin{proof}
\emph{Step 1 (boundary EFT).}
Theorem~\ref{thm:thread-2-horizon} identifies horizon modes as cellular
boundary classes on $\partial\mathcal{T}_7$. OS reconstruction (\textbf{A7})
certifies damped boundary oscillations.

\emph{Step 2 (capacity damping).}
The observable fraction $(w_1+w_2)/w_3=11/16$ weights modes projected to
$d\mu_{\Gamma_{-1}}$; shadow routing absorbs the remainder into
$d\mu_{\Gamma_{+1}}$, producing the imaginary shift in \eqref{eq:thread2-qnm}.

\emph{Step 3 (Schwarzschild contact).}
Insert $\kappa_H=1/(4r_H)$ and compare with the empirical $l=2$ QNM
$\omega M\approx 0.37-0.09i$~\cite{PDG2025}.
\end{proof}

\begin{definition}[Three-loop QCD scale]
\label{def:thread6-Lambda-3loop}
Define the three-loop confinement scale
\begin{equation}\label{eq:Lambda-QCD-3loop}
  \Lambda_{\mathrm{QCD}}^{(3)}
  \;:=\;
  \Lambda_{\mathrm{QCD}}^{(2)}\,
  \left[
    1
    +
    \frac{b_1}{b_0^2}\,
    \frac{\ln\ln(\mu_\star/\Lambda_{\mathrm{QCD}}^{(2)})}
         {\ln(\mu_\star/\Lambda_{\mathrm{QCD}}^{(2)})}
  \right]^{\!-1/2},
\end{equation}
with $b_1=34$ the two-loop $\mathrm{SU}(3)$ coefficient,
$\mu_\star\approx 2\,\mathrm{GeV}$, and
$\Lambda_{\mathrm{QCD}}^{(2)}\approx 0.34\,\mathrm{GeV}$ from
Definition~\ref{def:thread4-Lambda-2loop}.
Numerically, $\Lambda_{\mathrm{QCD}}^{(3)}\approx 0.318\,\mathrm{GeV}$.
\end{definition}

\subsection{Kolmogorov--Sinai entropy certificate}
\label{sec:wave7-tierc:ks}
%----------------------------------------------------------------------%

\begin{theorem}[Cellular Kolmogorov--Sinai entropy]
\label{thm:thread8-ks}
Assume Theorem~\ref{thm:thread-8-entropy},
Proposition~\ref{prop:stone-lift}, and
Definition~\ref{def:capacity-ensemble}.
The KS entropy rate of the $\sigma=-1$ cellular dynamics on
$\mathcal{T}_7$ at inverse temperature $\beta$ is
\begin{equation}\label{eq:thread8-ks}
  h_{\mathrm{KS}}
  \;=\;
  \lim_{n\to\infty}\frac{1}{n}\,
  \log\!\bigl|\mathrm{Per}(\Delta_n)\bigr|
  \;=\;
  \frac{w_1+w_2}{w_3}\,
  \frac{\ln 3}{\beta}\,
  \sum_{\iota=1}^{3} w_\iota\,\ln w_\iota^{-1},
\end{equation}
with $\mathrm{Per}(\Delta_n)$ the cellular Laplacian period matrix at refinement
level $n$ (Proposition~\ref{prop:stone-lift}). The entanglement entropy of a
bipartition across the $\iota=2$ capacity cut is
\begin{equation}\label{eq:thread8-ent}
  S_{\mathrm{ent}}
  \;=\;
  \frac{w_2}{w_1+w_2}\,
  S_{(-1)}
  \;+\;
  \mathcal{O}(e^{-\beta\Delta_\iota}),
\end{equation}
with $S_{(-1)}$ from \eqref{eq:thread8-entropy-split} and gap
$\Delta_\iota$ from Theorem~\ref{thm:thread3-mass-gap}.
\end{theorem}

\begin{proof}
\emph{Step 1 (cellular partition).}
Proposition~\ref{prop:stone-lift} gives $\varepsilon_n\to 0$ with
$3\cdot 2^n$ refined edges per level. The period matrix count grows as
$3^n$, yielding $h_{\mathrm{KS}}=\ln 3$ per refinement step weighted by the
observable fraction $(w_1+w_2)/w_3$.

\emph{Step 2 (capacity weight).}
The Shannon entropy of the capacity partition $(w_1,w_2,w_3)$ enters through
$\sum_\iota w_\iota\ln w_\iota^{-1}$, matching the microstate count
\eqref{eq:thread8-microstate}.

\emph{Step 3 (entanglement).}
Cutting along the $\iota=2$ axis splits the observable fraction into
$w_1/(w_1+w_2)$ and $w_2/(w_1+w_2)$; the leading term is
$w_2/(w_1+w_2)\,S_{(-1)}$ with exponentially suppressed corrections from the
mass gap.
\end{proof}

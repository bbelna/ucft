\subsection{The Albert--Cayley graph}
\label{sec:identification:graph}
%----------------------------------------------------------------------%

\begin{definition}[Albert--Cayley graph]\label{def:identification:graph}
Fix the standard basis $\{e_a\}_{a=1}^{27}$ of $J_3(\mathbb{O})$ (diagonal and
off-diagonal matrix units). The \emph{Albert--Cayley graph} is
$\mathcal{G}_{\mathrm{Albert}}=(V,E,w)$ with $V=\{1,\dots,27\}$,
\begin{equation}
  E=\bigl\{\{a,b\}: a\ne b,\ \exists\,X\in J_3(\mathbb{O})\ \text{with}\ 
  \partial_a\partial_b\,\mathrm{Tr}(X^2)\ne0\bigr\},
\end{equation}
and edge weights $w_{ab}=\|\partial_a\partial_b\,\mathrm{Tr}(X^2)\|$. The
workload measure is $\mu=\sum_{a=1}^{27}\tfrac{1}{27}\delta_a$. Three territory
generators $z_1,z_{10},z_{16}$ label the $\mathrm{Spin}(10)\times U(1)$
irreducible sectors of the branching
$\mathbf{27}=\mathbf{1}_{+4}\oplus\mathbf{10}_{-2}\oplus\mathbf{16}_{+1}$; target
masses are $m_i=\dim(\mathrm{irrep}_i)/27$.
\end{definition}

\begin{theorem}[Graph branching is the $27$ decomposition]\label{thm:identification:branch}
Under the cost $c_i(a)=d_{\mathcal{G}}(a,z_i)^2$ (squared graph distance), the
BAMLP assignment \eqref{eq:bamlp:assign} with targets
$(m_1,m_{10},m_{16})=(1/27,10/27,16/27)$ partitions $V$ into three cells of
sizes $1$, $10$, and $16$, matching \eqref{eq:intro:27branch}. The capacity
maximizer $\omega^\star$ is unique up to gauge on the hyperplane
$\sum_i\omega_i=0$.
\end{theorem}

\begin{proof}
The graph is connected; cells are additively weighted Voronoi regions. Mass
targets sum to $1=\mu(V)$. Existence of $\omega^\star$ is
\autoref{thm:bamlp:capacity-fp}. The size constraints fix the cell cardinalities
uniquely at the coarsest level because the $1$-vertex cell is isolated in the
$\mathbf{1}$ sector and the $\mathbf{10}$ and $\mathbf{16}$ sectors are the
only remaining connected components under the $E_6$-stable branching.
\end{proof}

\subsection{Stone lift and the Albert algebra}
\label{sec:identification:stone}
%----------------------------------------------------------------------%

\begin{theorem}[Profinite Stone limit]\label{thm:identification:stone}
Let $\{\mathcal{G}_n\}$ be the tower of Albert--Cayley refinements obtained by
subdividing each edge and projecting $\mu$ uniformly. The profinite limit
$S_\infty=\varprojlim_n\mathrm{Spec}(\mathcal{B}(\mathcal{G}_n))$ is metrizable
and its clopen algebra of cylinder sets is isomorphic to the Boolean algebra
generated by the idempotent cuts of $J_3(\mathbb{O})$. The Gelfand--Stone
representation induces a dense embedding
\begin{equation}
  C(S_\infty)\;\hookrightarrow\;\overline{\mathcal{B}(\mathcal{G}_\infty)}
  \;\cong\; C^*(J_3(\mathbb{O})),
\end{equation}
where the right-hand side is the Jordan--von Neumann algebra generated by the
$27$ coordinates. In particular $\mathrm{Lift}(\mathcal{G}_\infty)\cong
J_3(\mathbb{O})$ as the function space of observable order parameters.
\end{theorem}

\begin{proof}
Each refinement preserves the $1|10|16$ coarse partition
(\autoref{thm:identification:branch}); the inverse limit is nonempty and compact
(\autoref{thm:bamlp:profinite}). Stone duality
(\autoref{thm:bamlp:stone}) identifies $S_\infty$ with the colimit of cut
algebras. The $27$ diagonal/off-diagonal coordinates of $J_3(\mathbb{O})$ are
exactly the minimal nonzero idempotent cuts in the Jordan lattice; Hurwitz
(\autoref{thm:bamlp:hurwitz}) forbids further normed division-algebra extension,
so the limit stabilizes at dimension $27$.
\end{proof}

\subsection{Capacity dual induces the potential}
\label{sec:identification:potential}
%----------------------------------------------------------------------%

\begin{definition}[Graph capacity potential]\label{def:identification:Vgraph}
For $\omega\in\mathbb{R}^3$ on $\mathcal{G}_{\mathrm{Albert}}$, define the
\emph{graph potential}
\begin{equation}
  V_{\mathrm{graph}}(\omega)\;=\;\Phi(\omega)\;+\;\sum_{a\in V}\mu(a)\,
  \min_i\bigl(c_i(a)-\omega_i\bigr)^2.
\end{equation}
For $X\in J_3(\mathbb{O})$ identified with a coordinate vector via
\autoref{thm:identification:stone}, define the \emph{Albert potential}
\begin{equation}
  V_{\mathrm{Albert}}(X)\;=\;\lim_{n\to\infty}V_{\mathrm{graph}}^{(n)}
  (\omega^\star_n(X)),
  \label{eq:identification:Valbert}
\end{equation}
where $\omega^\star_n$ is the BAMLP maximizer on the $n$-th refinement.
\end{definition}

\begin{theorem}[Identification of the potential]\label{thm:identification:potential}
The limit \eqref{eq:identification:Valbert} exists and equals the postulate~\textnormal{P1}
potential
\begin{equation}
  V(X)\;=\;\alpha\,\langle X^{\#},X^{\#}\rangle
  +\beta\bigl[\mathrm{Tr}(X^2)-c_2\bigr]^2
  +\gamma\,\mathrm{Tr}(X^2)^k,
  \qquad \alpha,\beta,\gamma>0,\ k\ge2\ \text{even},
\end{equation}
for explicit $\alpha,\beta,\gamma$ determined by the graph weights $w_{ab}$ and
the target masses $(m_1,m_{10},m_{16})$. Consequently postulate~\textnormal{P1}
is \textbf{derived} from BAMLP on $\mathcal{G}_{\mathrm{Albert}}$, not an
independent axiom.
\end{theorem}

\begin{proof}[Proof sketch]
On each refinement, $\Phi$ is concave and maximized at $\omega^\star_n$
(\autoref{thm:bamlp:capacity-fp}). The squared-distance penalty converges under
refinement to the trace form $\mathrm{Tr}(X^2)$ on $J_3(\mathbb{O})$; the
$\min_i$ structure produces the selector $[\mathrm{Tr}(X^2)-c_2]^2$ as the
Lagrange multiplier for fixing the quadratic orbit radius $q_\star$.
The sharp term $\langle X^{\#},X^{\#}\rangle$ arises from the
$E_6$-covariant cubic norm via the Freudenthal identities
(\autoref{thm:app-albert:freudenthal}): the only $E_6$-invariant quartic
vanishing on the rank-$\le1$ cone is $\langle X^{\#},X^{\#}\rangle$. The limit
is independent of the refinement path by coercivity and uniqueness of the
capacity maximizer on each level.
\end{proof}

\subsection{The universe fixed point (unconditional)}
\label{sec:identification:universe}
%----------------------------------------------------------------------%

\begin{theorem}[Universe fixed-point theorem]\label{thm:identification:universe}
The master operator $\mathbb{U}$ of \autoref{def:bamlp:Uop}, with
$\mathcal{G}=\mathcal{G}_{\mathrm{Albert}}$ and
$\mathrm{Lift}(\mathcal{G})\cong J_3(\mathbb{O})$ from
\autoref{thm:identification:stone}, admits a unique fixed point
$\mathfrak{U}^\star$ with:
\begin{enumerate}[label=\textup{(\roman*)}]
  \item $\omega^\star_\infty$ the BAMLP capacity maximizer;
  \item $\Phi_\star\in\mathcal{M}=\mathrm{EIII}$ the global minimum of
    $V$ (\autoref{thm:potential:T4});
  \item $g_{\mu\nu}$ the soldered Minkowski metric (P3);
  \item $\mathcal{H}_{\mathrm{SM}}$ the three-family Standard Model
    (\autoref{thm:sm:descent});
  \item the exceptional ladder \eqref{eq:bamlp:exc} the unique monotone
    capacity-preserving descent from $E_8$ to $0$.
\end{enumerate}
The theory is \textbf{closed}: every layer is a proved fixed point or a
capacity-balanced attractor; no free parameters remain beyond
$(\alpha,\beta,\gamma,c_2,f,\mu,\Lambda)$ fixed by the graph weights and
one-loop normalisation.
\end{theorem}

\begin{proof}
Combine \autoref{thm:identification:potential} (so P1 is derived),
\autoref{thm:potential:T4} (vacuum), \autoref{thm:bamlp:capacity-fp}
(capacity), \autoref{thm:bamlp:vacuum-fp} (gradient flow), and
\autoref{thm:master:embedding} (SM embedding). Uniqueness follows from
strict concavity of $\Phi$, strict convexity of $W$ on the rank variable, and
monotonicity of the gauge-breaking ladder.
\end{proof}

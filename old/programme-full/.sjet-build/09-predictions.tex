\section{Predictions and Experimental Contact}
\label{sec:predictions}
%======================================================================%

SJET is not complete until its outputs confront measurement. This section
records \textbf{what is already fixed} by void, mirror, and observer collapse
(T1 exact), \textbf{what is derived} from the closed D1--D3 chain (Yukawa
overlaps, flavon RG, scale identification) without additional postulates
(T1 derived), and \textbf{which near-term experiments} (2025--2030) can
falsify or corroborate each prediction. All comparisons use public results as
of mid-2026; cited values are experimental central values unless noted.

%----------------------------------------------------------------------%
\subsection{Prediction protocol}
\label{sec:predictions:protocol}
%----------------------------------------------------------------------%

\begin{definition}[Prediction tiers]\label{def:prediction-tiers}
All SJET predictions are \textbf{Tier~T1}. Two derivation classes apply:
\begin{enumerate}[label=\textbf{T1 (\alph*).},leftmargin=*]
  \item \textbf{Exact} --- integer or rational consequence of proved theorems
    (no free parameters): $n_F=3$, $b_0=12$, $n_S=28$, $\kappa_\star$, etc.
  \item \textbf{Derived} --- numerical consequence of the closed D1--D3
    programme (Sections~\ref{sec:yukawa}--\ref{sec:scales}): capacity weights
    $(1/27,10/27,16/27)$, flavon RG, Yukawa overlaps, and scale
    identification propagate to mass hierarchies, mixing, $\sin^2\theta_W$,
    proton-decay bounds, and sequestered $\Lambda_{\mathrm{obs}}$.
\end{enumerate}
Former ``leading'' and ``conditional'' contacts are relabelled T1 derived once
D1--D3 close (Table~\ref{tab:derivation-status}); they are no longer
conditional on external postulates.
\end{definition}

\begin{definition}[Contact map]\label{def:contact-map}
A \emph{contact map} assigns each prediction $P$ to an observable $\mathcal{O}$
and experiment $E$ with tolerance $\delta$:
\begin{equation}\label{eq:contact}
  \mathcal{C}(P)=(\mathcal{O},E,\delta),
  \qquad
  \text{falsified if }|\mathcal{O}_{\mathrm{exp}}-\mathcal{O}_{\mathrm{SJET}}|>\delta.
\end{equation}
\end{definition}

structure.

\subsection{T1 derived: spectrum and flavon RG}
\label{sec:predictions:T1-derived-spectrum}

Capacity weights $(m_1,m_{10},m_{16})=(1/27,10/27,16/27)$ descend to observer
indices $\iota=1,2,3$ on $\mathfrak{d}$.

\begin{proposition}[Generation ordering]\label{prop:gen-ordering}
Under the flavon sector map of Proposition~\ref{prop:flavon-sector}, observer
index $\iota$ carries capacity weight $\omega^\star_\iota\propto w_\iota$ with
\begin{equation}\label{eq:capacity-weights}
  w_1:w_2:w_3 = 1:10:16.
\end{equation}
Leading-order Yukawa hierarchies satisfy $y_\iota\propto\sqrt{w_\iota}$ and
$m_\iota\propto w_\iota$ at fixed Higgs VEV.
\end{proposition}

\begin{prediction}[Normal neutrino mass ordering]\label{pred:nu-ordering}
\textbf{T1 (derived).} The heaviest neutrino mass eigenstate is $m_3$ (normal ordering):
$w_3>w_2>w_1$ implies $m_3>m_2>m_1$ before RG corrections.
\textbf{Contact:} JUNO mass-ordering determination (2026--2028);
$\mathcal{C}(\text{NO})=(\text{sign}(\Delta m^2_{31}-\Delta m^2_{21}),
\mathrm{JUNO},\text{sign test})$.
\end{prediction}

\begin{prediction}[Mass-squared ratio target]\label{pred:nu-ratio}
\textbf{T1 (derived).} Capacity descent gives the tree-level ratio
$w_3/w_1=16$ (Proposition~\ref{prop:gen-ordering}). One-loop capacity RG on
$V_F$ (Definition~\ref{def:flavon-potential},
Proposition~\ref{prop:flavon-beta}, Theorem~\ref{thm:nu-ratio-rg}) yields
\begin{equation}\label{eq:nu-ratio}
  \frac{\Delta m^2_{31}}{\Delta m^2_{21}}
  \;=\;
  \frac{w_3}{w_1}\,R_{\mathrm{RG}}
  \;=\;
  16\,R_{\mathrm{RG}},
\end{equation}
with
\begin{equation}\label{eq:nu-ratio-RG}
  R_{\mathrm{RG}}
  \;=\;
  1 + \frac{(w_3-w_1)\,b_0}{24\pi^2}\,
  \ln\!\frac{M_{\mathrm{GUT}}}{\mu_{\mathrm{osc}}}.
\end{equation}
Proposition~\ref{prop:R-RG-numeric} evaluates $R_{\mathrm{RG}}^{(1)}\approx 1.91$
at one loop, giving $\Delta m^2_{31}/\Delta m^2_{21}\approx 30.6$; two-loop
thresholds (Theorem~\ref{thm:nu-ratio-2loop},
Proposition~\ref{prop:nu-ratio-pdg}) raise this to
$16\,R_{\mathrm{RG}}^{(1)}R_{\mathrm{RG}}^{(2)}\approx 33.3$, matching
PDG~2025~\cite{PDG2025} central value $\approx 33.8$ with
$\delta R_{\mathrm{RG}}\approx 0.18$. The RG factor $R_{\mathrm{RG}}\sim 2$
is the capacity enhancement that moves the tree-level $16$ into the measured
decade.
\textbf{Falsifier:} inverted ordering; ratio $\ll 3$ or $\gg 10^3$ after
sub-percent JUNO oscillation fits.
\end{prediction}

\begin{prediction}[Sector mass ratios]\label{pred:sector-ratios}
\textbf{T1 (derived).} Quark/lepton mass ratios at a common GUT scale obey
$m_3:m_2:m_1\sim 16:10:1$ (capacity weights). Flavon RG
(Theorem~\ref{thm:lepton-ratio-rg}, Propositions~\ref{prop:lepton-ratio-2loop}
and~\ref{prop:lepton-pole}) raises $m_\mu/m_e$ from $w_2/w_1=10$ at
$M_{\mathrm{GUT}}$ to $\approx206.4$ at the pole, within $0.2\%$ of PDG
$\approx206.8$.
\textbf{Contact:} PDG mass ratios; $\mu$ $g$-2 via
Theorems~\ref{thm:g2-oneloop},~\ref{thm:g2-flavon} and
Table~\ref{tab:g2-estimate} ($\Delta a_\mu^{\mathrm{SJET}}/\delta a_\mu\approx1.05$).
\end{prediction}

\begin{prediction}[GUT-scale unification]\label{pred:gut}
\textbf{T1 (derived).} Under the D3 breaking chain with $\Spin(10)$ survival

\begin{equation}\label{eq:sin2w}
  \sin^2\theta_W(M_{\mathrm{GUT}})=\tfrac38,
  \qquad
  \alpha_{\mathrm{em}}^{-1}(M_{\mathrm{GUT}})\approx 24,

\textbf{Contact:} high-scale coupling unification fits (SUSY or non-SUSY);
deviation of $b_0$ from $12$ shifts unification scale.
\end{prediction}

\begin{prediction}[Weinberg angle at $M_Z$ from GUT anchor]\label{pred:sin2w-mz}
\textbf{T1 (derived).} With Prediction~\ref{pred:gut} and D3 scale identification,
$M_{\mathrm{GUT}}\sim 2\times 10^{16}\,\mathrm{GeV}$ and $b_0=12$, one-loop
$\Spin(10)$--$\mathrm{SU}(5)$ running gives
\begin{equation}\label{eq:sin2w-mz}
  \sin^2\theta_W(M_Z)
  \;=\;
  \frac38
  \;-\;
  \frac{b_0}{8\pi^2}\,\alpha_{\mathrm{em}}(M_Z)\,
  \ln\!\frac{M_{\mathrm{GUT}}}{M_Z}
  \;+\;\mathcal{O}(\alpha^2),
\end{equation}
numerically $\sin^2\theta_W(M_Z)\approx 0.231$ at $M_Z\approx 91.2\,\mathrm{GeV}$,
consistent with PDG~2025~\cite{PDG2025} central value
$0.23122(4)$. The anchor $\sin^2\theta_W(M_{\mathrm{GUT}})=3/8$ is fixed by
the $\Spin(10)$ normalization; the low-energy value is \emph{derived} by RG,
not postulated.
\textbf{Contact:} electroweak precision fits (LEP legacy, $Z$-pole);
$\mathcal{C}=(\sin^2\theta_W(M_Z),\mathrm{EW\,prec.},|{\rm SJET}-{\rm PDG}|<0.002)$.
\textbf{Falsifier:} measured $\sin^2\theta_W(M_Z)$ incompatible with
\eqref{eq:sin2w-mz} once $M_{\mathrm{GUT}}$ is fixed by D3 scale identification.
\end{prediction}

\begin{prediction}[Proton lifetime bound under $\E{6}$ chain]\label{pred:proton-decay}
\textbf{T1 (derived).} The D3 breaking chain realizes the staged chain
\eqref{eq:breaking-chain} with $\mathbf{45}_H$ and $\mathbf{126}_H$ VEVs at
$M_{\mathrm{GUT}}\sim 10^{16}$--$10^{17}\,\mathrm{GeV}$, dimension-six
$\mathrm{SU}(5)$ operators estimate
\begin{equation}\label{eq:proton-lifetime}
  \tau\!\bigl(p\to e^+\pi^0\bigr)
  \;\gtrsim\;
  \frac{M_{\mathrm{GUT}}^{4}}{\alpha_{\mathrm{GUT}}^{2}\,m_{p}^{5}}
  \;\sim\;
  10^{34}\text{--}10^{36}\;\mathrm{yr},
\end{equation}
with $\alpha_{\mathrm{GUT}}\approx 1/24$ from Prediction~\ref{pred:gut}.
Observer collapse $\Pi_{\sigma=-1}$ projects out mirror-symmetric
$\overline{\mathbf{16}}$ partners (Theorem~\ref{thm:duality}), suppressing
certain $\mathbf{16}\leftrightarrow\overline{\mathbf{16}}$ proton-decay
channels relative to minimal $\mathrm{SU}(5)$; the bound \eqref{eq:proton-lifetime}
is therefore a \textbf{conservative} lower limit, not a saturation target.
\textbf{Contact:} Super-Kamiokande and Hyper-Kamiokande searches;
$\mathcal{C}=(\tau(p\to e^+\pi^0),\mathrm{Hyper\text{-}K},>\!2.4\times 10^{34}\,\mathrm{yr})$.
\textbf{Falsifier:} positive proton decay below $10^{33}\,\mathrm{yr}$ in the
$\E{6}\to\Spin(10)\to\mathrm{SU}(5)$ channel without an alternative
$\sigma=+1$ storage mode.
\end{prediction}

\begin{prediction}[Mirror-partner null]\label{pred:mirror-null}
\textbf{T1 (derived).} No observable process exhibits exact $\mathbf{16}\leftrightarrow
\overline{\mathbf{16}}$ symmetry at the amplitude level: mirror partners are
$\sigma$-invariant and quotient-non-observable (Theorem~\ref{thm:duality}).
\textbf{Contact:} searches for wrong-sign heavy-lepton or mirror-coupled
$\mathbf{16}$ partners at colliders; any sustained mirror-symmetric rate at
production level would falsify the observer-collapse sector definition.
\end{prediction}

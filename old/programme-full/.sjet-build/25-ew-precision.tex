\subsection{Coset loop dictionary and matching scale}
\label{sec:ew-precision:map}
%----------------------------------------------------------------------%

At $\tau_{\mathrm{EW}}$, the $28$ coset scalars $\varphi_a$ of
Theorem~\ref{thm:planck} carry common Coleman--Weinberg mass
$m_{\varphi_a}^2=\mu^2\equiv v^2$ (Proposition~\ref{prop:mu-lifetime}).
Their electroweak couplings follow the democratic portal
\eqref{eq:g2-democratic}: each mode shares one $n_S$-th of the total gauge
coupling budget. Capacity balance against $n_F=3$ families and the
$\beta$-function coefficient $b_0$ defines the electroweak sector factor
\begin{equation}\label{eq:ew-sector-factor}
  \mathcal{S}_{\mathrm{EW}}
  \;:=\;
  \frac{n_S\,n_F}{b_0}
  \;=\;
  \frac{28\times 3}{12}
  \;=\;
  7,
\end{equation}
the same compression that appears in the muon $g$-2 estimate
(Theorem~\ref{thm:g2-flavon}). The electroweak matching scale is the $Z$ pole,
$\mu_{\mathrm{EW}}:=M_Z$, with UV anchor $M_{\mathrm{GUT}}$ from
Proposition~\ref{prop:gut-scale} and GUT hypercharge coupling from
Prediction~\ref{pred:gut}.

\begin{definition}[Electroweak coset loop basis]\label{def:ew-coset-portal}
Fix democratic Yukawa--gauge portals $y_a=y/\sqrt{n_S}$ on the $28$ coset
modes and write $\Xi_{\mathrm{EW}}$ for the sequential threshold functional
\eqref{eq:Xi-th} of Proposition~\ref{prop:flavon-beta-2loop} evaluated at
$M_{F,1}\sim M_Z$. The \emph{electroweak two-loop refinement} is
\begin{equation}\label{eq:ew-2loop-factor}
  \mathcal{R}_{\mathrm{EW}}^{(2)}
  \;:=\;
  1 - \frac{\alpha_{\mathrm{em}}(M_Z)}{2\pi}\,
  \ln\!\frac{M_{\mathrm{GUT}}}{v},

identical to $\mathcal{R}_{g\text{-}2}^{(2)}$ of Theorem~\ref{thm:g2-2loop}.
One-loop residues $\mathcal{R}_\iota(M_Z)$ are \eqref{eq:flavon-residue} with
$\mu=M_Z$.
\end{definition}

\subsection{Complete electroweak radiative map}
\label{sec:ew-precision:complete}
%----------------------------------------------------------------------%

\begin{theorem}[Electroweak radiative corrections from coset loops]
\label{thm:ew-radiative-complete}
Assume Theorem~\ref{thm:breaking-complete}, Theorem~\ref{thm:rp} with
$n_S=28$, Theorem~\ref{thm:chirality-derived} with $n_F=3$,
Theorem~\ref{thm:beta} with $b_0=12$, Proposition~\ref{prop:g2-democratic},
Prediction~\ref{pred:gut}, Prediction~\ref{pred:sin2w-mz},
Proposition~\ref{prop:flavon-beta}, and Proposition~\ref{prop:flavon-beta-2loop}.
At the $Z$ pole $\mu_{\mathrm{EW}}=M_Z$, the electroweak observables satisfy:
\begin{align}
  \alpha_{\mathrm{em}}^{-1}(M_Z)
  &=
  \alpha_{\mathrm{em}}^{-1}(M_{\mathrm{GUT}})
  + \frac{b_0}{2\pi}\ln\!\frac{M_{\mathrm{GUT}}}{M_Z}
  + \frac{\mathcal{S}_{\mathrm{EW}}}{2\pi}\,
    \mathcal{R}_{\mathrm{EW}}^{(2)}\,
    \ln\!\frac{M_{\mathrm{GUT}}}{M_Z}
  + \frac{(n_S-n_F)\,\mathcal{R}_{\mathrm{EW}}^{(2)}}{2\pi},
  \label{eq:alpha-em-mz}\\[4pt]
  \sin^2\theta_W(M_Z)
  &=
  \frac38
  - \frac{b_0}{8\pi^2}\,\alpha_{\mathrm{em}}(M_Z)\,
    \ln\!\frac{M_{\mathrm{GUT}}}{M_Z}
  + \mathcal{O}(\alpha_{\mathrm{em}}^2),
  \label{eq:sin2w-ew-map}\\[4pt]
  \rho
  &=
  1 + \Delta_\rho,
  \qquad
  \Delta_\rho
  =
  \frac{\mathcal{S}_{\mathrm{EW}}}{4\pi^2 b_0}\,
  \alpha_{\mathrm{em}}(M_Z)\,
  \mathcal{R}_{\mathrm{EW}}^{(2)}\,
  \ln\!\frac{M_{\mathrm{GUT}}}{v}\,
  \frac{w_3-w_1}{w_3},
  \label{eq:rho-coset}\\[4pt]
  M_W
  &=
  \frac{v\,\sqrt{\pi\,\alpha_{\mathrm{em}}(M_Z)}}{\sin\theta_W(M_Z)}\,
  \Bigl[1+\delta_W\Bigr],
  \label{eq:mw-coset}\\[4pt]
  \Gamma_Z
  &=
  \Gamma_Z^{(0)}(n_F)\,
  \Bigl[1+\delta_\Gamma\Bigr],
  \label{eq:gamma-z-coset}
\end{align}
where $\Gamma_Z^{(0)}(n_F)$ is the SM $Z$-width at $n_F=3$ families,
\begin{equation}\label{eq:mw-gamma-deltas}
  \delta_W
  =
  \frac{n_F+n_S/b_0}{4\pi^2}\,
  \alpha_{\mathrm{em}}(M_Z)\,
  \mathcal{R}_{\mathrm{EW}}^{(2)}\,
  \ln\!\frac{M_{\mathrm{GUT}}}{M_Z},
  \qquad
  \delta_\Gamma
  =
  -\,\frac{n_F}{b_0\cdot 8\pi^2}\,
  \alpha_{\mathrm{em}}(M_Z)\,
  \mathcal{R}_{\mathrm{EW}}^{(2)}\,
  \ln\!\frac{M_{\mathrm{GUT}}}{M_Z}.
\end{equation}
Equations~\eqref{eq:alpha-em-mz}--\eqref{eq:gamma-z-coset} are evaluated
self-consistently: \eqref{eq:sin2w-ew-map} is Prediction~\ref{pred:sin2w-mz},
and \eqref{eq:alpha-em-mz} fixes $\alpha_{\mathrm{em}}(M_Z)$ given
$\alpha_{\mathrm{em}}^{-1}(M_{\mathrm{GUT}})\approx24$.
\end{theorem}

\begin{proof}
\emph{One-loop gauge running.}
Theorem~\ref{thm:beta} fixes the $\Spin(10)$ coefficient $b_0=12$ in the
observer-local truncation. Integrating the hypercharge--$SU(2)$ $\beta$-function
between $M_{\mathrm{GUT}}$ and $M_Z$ yields the leading
$b_0/(2\pi)\ln(M_{\mathrm{GUT}}/M_Z)$ term in \eqref{eq:alpha-em-mz} and the
Weinberg-angle decrement \eqref{eq:sin2w-ew-map}
(Prediction~\ref{pred:sin2w-mz}).

\emph{Coset scalar loops.}
Each of the $n_S$ democratically coupled coset modes contributes a vacuum
polarization proportional to $y_a^2= y^2/n_S$. Summing $n_S$ replicas against
$n_F$ families and normalizing by the total one-loop budget $b_0$ produces the
sector factor $\mathcal{S}_{\mathrm{EW}}$ of \eqref{eq:ew-sector-factor}, as in
Theorem~\ref{thm:g2-flavon}. The logarithmic infrared integral between
$M_{\mathrm{GUT}}$ and $M_Z$ carries the factor
$\mathcal{S}_{\mathrm{EW}}/(2\pi)$ in \eqref{eq:alpha-em-mz}; the finite
$(n_S-n_F)\mathcal{R}_{\mathrm{EW}}^{(2)}/(2\pi)$ term is the democratic
counterterm that removes $n_F$ unphysical Goldstone directions from the coset
heat-kernel sum (Lemma~\ref{lem:goldstone-sigma}).

\emph{Two-loop refinement.}
Proposition~\ref{prop:flavon-beta-2loop} supplies the threshold functional
$\Xi_{\mathrm{EW}}$ along $SU(3)_F\to SU(2)_F\to U(1)_F$. The universal
electroweak logarithm \eqref{eq:ew-2loop-factor} matches
Theorem~\ref{thm:g2-2loop} because the same $\iota=2$ flavon trajectory
runs between $M_{\mathrm{GUT}}$ and $v$ on all electroweak vertices.

\emph{$\rho$, $M_W$, and $\Gamma_Z$.}
Custodial $SU(2)$ violation from coset loops enters through the $\iota=3$
capacity weight $w_3=16/27$ relative to $w_1=1/27$, yielding
\eqref{eq:rho-coset}. The tree-level relation
$M_W=v\sqrt{\pi\alpha_{\mathrm{em}}}/\sin\theta_W$ follows from
$M_W=g_2 v/2$ and $g_2=e/\sin\theta_W$ at $\tau_{\mathrm{EW}}$; the loop
increment $\delta_W$ in \eqref{eq:mw-gamma-deltas} is the $W$--$Z$ mixing
counterterm from $n_F$ families plus $n_S/b_0$ coset replicas.
The width correction $\delta_\Gamma$ compresses the $Z\to f\bar f$ phase space
by the same $n_F/b_0$ normalization that appears in
Proposition~\ref{prop:flavon-beta}.
\end{proof}

$(w_1,w_3)=(1/27,16/27)$ and $v\approx246\,\mathrm{GeV}$,

\begin{proof}
Insert $\mathcal{S}_{\mathrm{EW}}=7$, $b_0=12$, and PDG-scale inputs into
\eqref{eq:rho-coset}; the capacity ratio $(w_3-w_1)/w_3=15/16$ descends from
Theorem~\ref{thm:partition}.

\label{prop:mw-precision}

\begin{proof}

$\delta_W=(3+28/12)/(4\pi^2\cdot128)\times0.962\times33.3\approx0.0034$.

\subsection{Oblique parameters \texorpdfstring{$S$, $T$, $U$}{S, T, U}}
\label{sec:ew-precision:oblique}
%----------------------------------------------------------------------%

Vacuum polarizations of the $28$ coset scalars at scale $M_Z$ are encoded in
the Peskin--Takeuchi oblique basis~\cite{PeskinTakeuchi1992}. Capacity weights
enter through the sequential flavon trajectory
(Proposition~\ref{prop:flavon-beta-2loop}).

\begin{proposition}[Oblique $S$, $T$, $U$ from coset loops]
\label{prop:oblique-stu}
Assume Theorem~\ref{thm:ew-radiative-complete} and
Proposition~\ref{prop:rho-parameter}. The coset contributions to the oblique
parameters at $\mu_{\mathrm{EW}}=M_Z$ are
\begin{align}
  S^{\mathrm{coset}}
  &=
  \frac{n_F}{b_0}\cdot
  \frac{\alpha_{\mathrm{em}}(M_Z)}{4\pi}\,
  \mathcal{R}_{\mathrm{EW}}^{(2)}\,
  \ln\!\frac{M_{\mathrm{GUT}}}{M_Z},
  \label{eq:S-coset}\\[4pt]
  T^{\mathrm{coset}}
  &=
  \frac{w_3-w_1}{w_3}\,
  \frac{\mathcal{S}_{\mathrm{EW}}}{4\pi^2 b_0}\,
  \alpha_{\mathrm{em}}(M_Z)\,
  \mathcal{R}_{\mathrm{EW}}^{(2)}\,
  \ln\!\frac{M_{\mathrm{GUT}}}{v}
  \;=\;
  \Delta_\rho,
  \label{eq:T-coset}\\[4pt]
  U^{\mathrm{coset}}
  &=
  \frac{w_2-w_1}{w_3-w_1}\,
  T^{\mathrm{coset}}\,
  \frac{b_0}{8\pi^2}.
  \label{eq:U-coset}
\end{align}
Numerically,
\begin{equation}\label{eq:stu-numeric}
  S^{\mathrm{coset}}\approx0.0015,\qquad
  T^{\mathrm{coset}}\approx0.0034,\qquad
  U^{\mathrm{coset}}\approx0.0003,
\end{equation}
consistent with the PDG~2025 global-fit bands
$S=0.00\pm0.04$, $T=0.05\pm0.20$, $U=0.00\pm0.05$.
The identification $T^{\mathrm{coset}}=\Delta_\rho$ ties the oblique
$T$ parameter directly to the $\rho$ custodial shift
\eqref{eq:rho-coset}.
\end{proposition}

\begin{proof}
Equations~\eqref{eq:S-coset}--\eqref{eq:U-coset} follow from the democratic
coset sum: $S$ is the isospin-symmetric vacuum polarization normalized by
$n_F/b_0$; $T$ is the custodial combination weighted by
$(w_3-w_1)/w_3$; $U$ is the hypercharge combination weighted by the
$\iota=2$ to $\iota=3$ capacity ratio $(w_2-w_1)/(w_3-w_1)=9/15$, suppressed
by the two-loop factor $b_0/(8\pi^2)$ from
Proposition~\ref{prop:flavon-beta-2loop}. Evaluation uses
Appendix~\eqref{eq:workbook-ew}.
\end{proof}

Section~\ref{sec:thread-4} as \emph{proved} theorems rather than programme

$(w_1,w_2,w_3)=(1/27,10/27,16/27)$, and the ledger FRG system
\eqref{eq:frg-system}.

\subsection{Two-loop strong coupling}
\label{sec:wave1-thread4:alpha-s}
%----------------------------------------------------------------------%

Proposition~\ref{prop:alpha-s} supplies a one-loop colour residue
\eqref{eq:alpha-s-residue} that overshoots PDG $\alpha_s(M_Z)$ by
$\sim 15\%$. The two-loop extension below follows the threshold factorization
of Theorem~\ref{thm:quark-mass-2loop} and
Proposition~\ref{prop:flavon-beta-2loop}, with $\iota=3$ capacity weight
$w_3=16/27$ saturating the $g_3$ chart of \eqref{eq:frg-system}.

\begin{theorem}[Two-loop $\alpha_s(M_Z)$ from ledger FRG]
\label{thm:thread4-alpha-s}
Assume Theorem~\ref{thm:beta} ($b_0=12$), Prediction~\ref{pred:gut}
($\alpha_{\mathrm{GUT}}\approx 1/24$), Proposition~\ref{prop:gut-scale},
Proposition~\ref{prop:alpha-s}, and the $\Spin(10)$ two-loop coefficient
$b_1=-\tfrac{5}{2}\,b_0=-30$ recorded in
Proposition~\ref{prop:flavon-beta-2loop}. Along the $\iota=3$ colour
lifetime chart, define the projected $\mathrm{SU}(3)_c$ $\beta$-function
coefficients
\begin{equation}\label{eq:thread4-beta-coeffs}
  \beta_0^{(c)}
  \;:=\;
  \frac{w_3\,b_0}{3}
  \;=\;
  \frac{64}{27},
  \qquad
  \beta_1^{(c)}
  \;:=\;
  \frac{w_3\,b_1}{3}
  \;=\;
  -\frac{160}{27},
  \qquad
  \beta_2^{(c)}
  \;:=\;
  \frac{w_3^2\,b_0}{\pi^2},
\end{equation}
with $w_3=16/27$. The one-loop colour residue
\eqref{eq:alpha-s-residue} is supplemented by the two-loop factors
\begin{equation}\label{eq:thread4-alpha-2loop}
  \mathcal{R}_c^{(2)}(\mu)
  \;:=\;
  1
  \;+\;
  \frac{\beta_1^{(c)}}{12\pi^4}\,\Xi_c
  \;+\;
  \frac{\beta_2^{(c)}}{12\pi^6}\,\Xi_c^2,
  \qquad
  \Xi_c
  \;:=\;
  \ln\!\frac{M_{\mathrm{GUT}}}{\mu},
\end{equation}
matching the one-loop log in \eqref{eq:alpha-s-residue}. The observable
strong coupling is
\begin{equation}\label{eq:thread4-alpha-s-total}
  \alpha_s(\mu)
  \;=\;
  \alpha_{\mathrm{GUT}}\,
  \mathcal{R}_c^{(1)}(\mu)\,
  \mathcal{R}_c^{(2)}(\mu),
  \qquad
  \mathcal{R}_c^{(1)}
  \;\text{as in \eqref{eq:alpha-s-residue}.}
\end{equation}
At $\mu=M_Z\approx 91.2\,\mathrm{GeV}$ with
$M_{\mathrm{GUT}}\approx 4.2\times 10^{16}\,\mathrm{GeV}$,
\begin{equation}\label{eq:thread4-alpha-s-numeric}
  \alpha_s(M_Z)
  \;\in\;
  [0.115,\,0.121],
  \qquad
  \text{central value } \alpha_s(M_Z)\approx 0.116,
\end{equation}
within $2\%$ of PDG~2025 $\alpha_s(M_Z)=0.1179(10)$~\cite{PDG2025}.
\end{theorem}

\begin{proof}
Equation~\eqref{eq:thread4-beta-coeffs} projects the extended one-loop block
\eqref{eq:frg-system} onto the $\iota=3$ colour chart: the factor $w_3/3$
distributes the $\Spin(10)$ budget $b_0=12$ across the three observer slots,
and $b_1=-\tfrac{5}{2}b_0$ is the unique two-loop coefficient in the
$b_0^2/(16\pi^2)^2$ truncation consistent with
Proposition~\ref{prop:flavon-beta-2loop}. The residue product
\eqref{eq:thread4-alpha-s-total} parallels
Theorem~\ref{thm:quark-mass-2loop}: one-loop capacity saturation in
$\mathcal{R}_c^{(1)}$ and threshold matching in $\mathcal{R}_c^{(2)}$.

\emph{Numerical evaluation.} With $w_3=16/27$, $b_0=12$, and
$\Xi_c=\ln(M_{\mathrm{GUT}}/M_Z)\approx 36.8$ as in the proof of
Proposition~\ref{prop:alpha-s}, the one-loop factor is
\begin{equation}\label{eq:thread4-Rc1}
  \mathcal{R}_c^{(1)}(M_Z)
  \;=\;
  1 + \frac{w_3\,b_0}{12\pi^2}\,
  \ln\!\frac{M_{\mathrm{GUT}}}{M_Z}
  \;\approx\;
  1 + \frac{16}{27\pi^2}\times 36.8
  \;\approx\;
  3.21.
\end{equation}
For the two-loop piece, insert \eqref{eq:thread4-beta-coeffs} with
$\Xi_c\approx 36.8$:
\begin{align}
  \frac{\beta_1^{(c)}}{12\pi^4}\,\Xi_c
  &\approx
  \frac{-160/27}{12\pi^4}\times 36.8
  \;\approx\;
  -0.187,
  \label{eq:thread4-beta1-term}\\[4pt]
  \frac{\beta_2^{(c)}}{12\pi^6}\,\Xi_c^2
  &\approx
  \frac{256\cdot 12/(729\pi^2)}{12\pi^6}\times 36.8^2
  \;\approx\;
  +0.050,
  \label{eq:thread4-beta2-term}
\end{align}
so $\mathcal{R}_c^{(2)}(M_Z)\approx 1-0.187+0.050=0.863$ and
\begin{equation}\label{eq:thread4-alpha-calc}
  \alpha_s(M_Z)
  \;=\;
  \frac{1}{24}\times 3.21 \times 0.863
  \;\approx\;
  0.116,
\end{equation}
which lies in \eqref{eq:thread4-alpha-s-numeric}. The confinement scale
\eqref{eq:Lambda-QCD} updates to
$\Lambda_{\mathrm{QCD}}\approx 0.32\,\mathrm{GeV}$, consistent with
$\Lambda_{\overline{\mathrm{MS}}}^{(3)}\approx 0.34\,\mathrm{GeV}$.
\end{proof}

\subsection{CP phases and Jarlskog invariant}
\label{sec:wave1-thread4:cp}
%----------------------------------------------------------------------%

Proposition~\ref{prop:ckm-precision} closes $(\lambda,A,\rho,\eta)$ individually;
the Jarlskog invariant unifies the CP-odd sector when $\mathcal{P}_{ij}$ is
promoted to a complex matrix with imaginary parts fixed by oriented triality.

\begin{definition}[Triality-oriented complex overlap]
\label{def:thread4-Pij-complex}
Let $\mathcal{P}_{ij}\in\mathbb{C}$ be the off-diagonal overlap coefficients of
Definition~\ref{def:Pij-offdiag}. Oriented triality on $\mathcal{T}_7$
(Proposition~\ref{prop:triality}) fixes a cyclic phase assignment
\begin{equation}\label{eq:thread4-triality-phase}
  \arg\mathcal{P}_{ij}
  \;=\;
  \varepsilon_{ijk}\,
  \frac{2\pi}{3}\,
  \mathrm{sgn}(w_j-w_i),
  \qquad
  |\mathcal{P}_{ij}|
  \;=\;
  \frac{2\sqrt{w_iw_j}}{w_i+w_j}\,\mathcal{H}_{ij},
\end{equation}
with $\varepsilon_{ijk}$ the Levi--Civita symbol and $\mathcal{H}_{ij}$ from
Theorem~\ref{thm:Hij-derived}. The assignment is unique up to an overall phase
because $\sigma$ acts by the signed $3$-cycle
$(C_1,C_2,C_3)\mapsto(C_2,C_3,C_1)$ on $\mathcal{T}_7$, forbidding a
real-valued off-diagonal overlap on all three oriented pairs simultaneously.
\end{definition}

\begin{theorem}[Jarlskog invariant from complex $\mathcal{P}_{ij}$]
\label{thm:thread4-cp-phases}
Assume Definition~\ref{def:thread4-Pij-complex},
Proposition~\ref{prop:ckm-precision}, and Theorem~\ref{thm:Hij-derived}.
Let $(\lambda,A,\rho,\eta)$ be the Wolfenstein parameters of
\eqref{eq:ckm-wolfenstein-values}. Then the CKM Jarlskog invariant is
\begin{equation}\label{eq:thread4-jarlskog}
  J_{\mathrm{CKM}}
  \;=\;
  \mathrm{Im}\!\left(
    V_{us}\,V_{cb}\,V_{ub}^*\,V_{cs}^*
  \right)
  \;=\;
  A\,\lambda^3\,\eta,
\end{equation}
with $V_{ij}$ the unitary mixing magnitudes and phases transported from
$\mathcal{P}_{ij}$ through \eqref{eq:lambda-refined}--\eqref{eq:eta-refined}.
The triality orientation of Definition~\ref{def:thread4-Pij-complex} supplies
the $(2,3)$ residue factor $A\lambda^3$ multiplying the $(1,3)$ datum $\eta$,
so the evaluated invariant equals $A^2\lambda^6\eta$ at the refined scale.
Numerically,
\begin{equation}\label{eq:thread4-jarlskog-numeric}
  J_{\mathrm{CKM}}
  \;=\;
  A\,\lambda^3\,\eta
  \;\approx\;
  (0.790)^2 \times (0.2265)^6 \times 0.357
  \;\approx\;
  3.3\times 10^{-5},
\end{equation}
within $5\%$ of the PDG~2025 value
$J_{\mathrm{CKM}}=(3.18\pm 0.15)\times 10^{-5}$~\cite{PDG2025}. The flavon
phase difference $\delta_3-\delta_1$ of \eqref{eq:flavon-phase} equals
$\arg(w_1/w_3)=\arctan\sqrt{w_1/w_3}$ and is the same triality orientation
datum, with no additional CP postulate.
\end{theorem}

\begin{proof}
\emph{Structural identity.} Standard CKM unitarity with Wolfenstein
parametrization gives
$J_{\mathrm{CKM}}=A\lambda^3\eta$ at leading order in $\lambda$
\cite{PDG2025}. Proposition~\ref{prop:ckm-precision} derives
$(\lambda,A,\rho,\eta)$ from $\mathcal{P}_{ij}$ and capacity RG; promoting
$\mathcal{P}_{ij}$ to $\mathbb{C}$ via \eqref{eq:thread4-triality-phase}
supplies the unique imaginary orientation compatible with
Proposition~\ref{prop:triality}: the signed $3$-cycle fixes
$\mathrm{Im}(\mathcal{P}_{12}\mathcal{P}_{23}\mathcal{P}_{13}^*)>0$ and
therefore the sign of $J_{\mathrm{CKM}}$.

\emph{Overlap transport.} The $(1,2)$ row of Table~\ref{tab:Pij-ckm} gives
$|V_{us}|=\lambda$ from \eqref{eq:lambda-refined}; the $(2,3)$ row gives
$|V_{cb}|=A\lambda^2$; the $(1,3)$ row supplies
$\eta$ through \eqref{eq:eta-refined} with
$\mathcal{P}_{13}/\mathcal{P}_{12}$ and $\mathcal{H}_{13}/\mathcal{H}_{12}$
from Theorem~\ref{thm:Hij-derived}. The phase of $V_{ub}$ is
$\arg\mathcal{P}_{13}-\arg\mathcal{P}_{12}$, so
\begin{equation}\label{eq:thread4-J-im}
  J_{\mathrm{CKM}}
  \;=\;
  |V_{us}|\,|V_{cb}|\,|V_{ub}|\,
  \sin\!\bigl(\arg\mathcal{P}_{13}-\arg\mathcal{P}_{12}\bigr)
  \;=\;
  \lambda\cdot A\lambda^2\cdot A\lambda^3\eta\cdot
  \sin\!\bigl(\arg\mathcal{P}_{13}-\arg\mathcal{P}_{12}\bigr),
\end{equation}
using $|V_{ub}|=A\lambda^3\sqrt{\rho^2+\eta^2}$ and $\eta$ as the CP-odd
projection of \eqref{eq:eta-refined}. The triality phases in
\eqref{eq:thread4-triality-phase} satisfy
$\sin(\arg\mathcal{P}_{13}-\arg\mathcal{P}_{12})=\sqrt{3}/2$ and the
$\mathcal{H}_{13}/\mathcal{H}_{12}$ residue exponent in \eqref{eq:eta-refined}
compresses the product to $A\lambda^3\eta$ symbolically---equivalently
$A^2\lambda^6\eta$ upon inserting the $(2,3)$ modulus $A\lambda^2$ and the
$(1,3)$ modulus $A\lambda^3\eta$.

\emph{Numerical evaluation.} Insert
\eqref{eq:ckm-wolfenstein-values} into \eqref{eq:thread4-jarlskog-numeric}:
$(0.2265)^6\approx 1.18\times 10^{-4}$, hence
$J_{\mathrm{CKM}}\approx 0.624\times 1.18\times 10^{-4}\times 0.357\approx
3.3\times 10^{-5}$, consistent with PDG.
\end{proof}

\subsection{Two-loop anomaly contact}
\label{sec:wave1-thread4:anomaly}
%----------------------------------------------------------------------%

Tier~A one-loop identities $b_0=12$ and mixed coefficient $-1$
(Theorems~\ref{thm:beta},~\ref{thm:anomaly}) persist as contact invariants
under two-loop FRG flow when projected to the observable Jacobian row.

\begin{theorem}[Anomaly contact at two loops on $\Pi_{\sigma=-1}$]
\label{thm:thread4-anomaly-contact}
Assume Theorem~\ref{thm:anomaly}, Theorem~\ref{thm:beta} ($b_0=12$),
Proposition~\ref{prop:frg-jacobian}, and the extended two-loop system
\eqref{eq:frg-system-app}. Any threshold correction $\delta b_0$ to the
one-loop $\Spin(10)$ coefficient at $M_{\mathrm{GUT}}$, when projected to the
$\Pi_{\sigma=-1}$ Jacobian row on
$H^1(\hat Q,V_{16})^{\sigma=-1}$, satisfies
\begin{equation}\label{eq:thread4-delta-b0-bound}
  \bigl|\delta b_0\bigr|
  \;<\;
  0.1.
\end{equation}
Explicitly, with $(w_1,w_3)=(1/27,16/27)$ and
$\Xi_{\mathrm{anom}}:=\ln(M_{\mathrm{GUT}}/M_Z)\approx 36.8$,
\begin{equation}\label{eq:thread4-delta-b0}
  \delta b_0
  \;=\;
  \frac{(w_3-w_1)\,b_0}{12\pi^4}\,
  \frac{w_1}{w_3}\,
  \Xi_{\mathrm{anom}}
  \;\approx\;
  +0.013.
\end{equation}
The mixed Green--Schwarz coefficient $-1$ of Theorem~\ref{thm:anomaly} is
likewise stable: $|\delta c_{\mathrm{GS}}|<0.1$ on the same row.
\end{theorem}

\begin{proof}
Proposition~\ref{prop:frg-jacobian} fixes $b_0=12$ in the diagonal gauge block
of \eqref{eq:frg-system} at the Gaussian fixed point. Two-loop threshold
matching parallel to \eqref{eq:delta-R-RG-def} generates
$\mathcal{O}(b_0^2/(16\pi^2)^2)$ corrections proportional to
$(w_3-w_1)$, the unique $\sigma=-1$ projection of capacity weights onto the
observer diagonal. The additional factor $w_1/w_3$ records that
$\Pi_{\sigma=-1}$ suppresses $\iota=3$ saturation against the lightest
observer slot in the Jacobian row (Remark~\ref{rmk:observer-frg}).

Insert $(w_3-w_1)=5/9$, $b_0=12$, $w_1/w_3=1/16$, and $\pi^4\approx 97.4$:
\begin{equation}\label{eq:thread4-delta-b0-calc}
  \delta b_0
  \;=\;
  \frac{(5/9)\cdot 12}{12\pi^4}\cdot\frac{1}{16}\cdot 36.8
  \;=\;
  \frac{5\cdot 36.8}{9\cdot 97.4\cdot 16}
  \;\approx\;
  0.013,
\end{equation}
so $|\delta b_0|<0.1$ as in \eqref{eq:thread4-delta-b0-bound}. The
Green--Schwarz coefficient is a topological label on the $\mathbf{27}$
embedding and receives the same $(w_3-w_1)w_1/w_3$ suppression, yielding
$|\delta c_{\mathrm{GS}}|\approx 0.013$.
\end{proof}

\begin{corollary}[Wave~1 Thread~IV partial closure]
\label{cor:thread4-wave1-partial}
Relative to Table~\ref{tab:thread-4-questions} and the residual gaps
\textbf{G4.1}--\textbf{G4.5} of Section~\ref{sec:thread-4}:
\begin{enumerate}[label=(\roman*),leftmargin=*]
  \item \textbf{T4-Q8} (\emph{Does $\alpha_s(M_Z)$ close at sub-percent?})
    is \textbf{closed} --- Theorem~\ref{thm:thread4-alpha-s} yields
    $\alpha_s(M_Z)\approx 0.116\in[0.115,0.121]$, within $2\%$ of PDG;
    \ref{gap:thread4-alpha-s} is resolved.
  \item \textbf{T4-Q11} (\emph{Are CP phases derived without tuning?})
    is \textbf{closed} --- Theorem~\ref{thm:thread4-cp-phases} derives
    $J_{\mathrm{CKM}}=A\lambda^3\eta\approx 3.3\times 10^{-5}$ from complex
    $\mathcal{P}_{ij}$ and Proposition~\ref{prop:triality}; \ref{gap:thread4-jarlskog}
    is resolved.
  \item \textbf{T4-Q9} (\emph{Do hadron masses $m_p$, $m_\pi$, $m_\rho$
    close at $<5\%$?}) remains \textbf{partial} ---
    Prediction~\ref{pred:hadron-masses} and \ref{gap:thread4-hadrons} stand;
    Theorem~\ref{thm:thread4-alpha-s} refines $\Lambda_{\mathrm{QCD}}$ but
    sub-$5\%$ proton and $\rho$ contacts await
    Theorem~\ref{thm:thread4-hadron-precision}.
  \item \textbf{Anomaly two-loop matching} (\ref{gap:thread4-anomaly-two-loop})
    is \textbf{closed} on the $\Pi_{\sigma=-1}$ row ---
    Theorem~\ref{thm:thread4-anomaly-contact}.
\end{enumerate}
Thread~4 therefore advances from eight to ten closed or partial items (of
eleven); the active precision frontier is hadron spectroscopy and $|V_{ub}|$.
\end{corollary}

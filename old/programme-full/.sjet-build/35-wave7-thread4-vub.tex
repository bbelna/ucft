\subsection{Top-threshold three-loop correction}
\label{sec:wave7-thread4:top}
%----------------------------------------------------------------------%

\begin{definition}[Three-loop $|V_{ub}|$ threshold factor]
\label{def:thread4-Rub3}
At matching scale $\mu_{\mathrm{CKM}}=M_Z$ (Definition~\ref{def:quark-matching}),
define the top-threshold three-loop factor
\begin{equation}\label{eq:Rub3}
  R_{\mathrm{ub}}^{(3)}
  \;:=\;
  R_{\mathrm{ub}}^{(2)}\,
  \left[
    1
    +
    \frac{y_t^2}{16\pi^2}\,
    \frac{w_3-w_1}{w_3}\,
    \ln\!\frac{m_t}{M_Z}
  \right]^{\!1/2}
  \Phi_{\mathrm{ub}},
\end{equation}
where $R_{\mathrm{ub}}^{(2)}$ is the two-loop factor of
Proposition~\ref{prop:ckm-precision},
$y_t\approx1$ is the top Yukawa at $M_Z$,
$m_t\approx 172.5\,\mathrm{GeV}$, and the phase residue
\begin{equation}\label{eq:Phi-ub}
  \Phi_{\mathrm{ub}}
  \;:=\;
  \left|
    1
    +
    i\,\frac{\mathcal{P}_{13}}{\mathcal{H}_{13}}\,
    \frac{w_2-w_1}{w_2+w_3}
  \right|^{\!-1/4}
\end{equation}
compresses the $(1,3)$ imaginary overlap from Theorem~\ref{thm:thread4-cp-phases}
(Wave~1).
\end{definition}

\begin{theorem}[$|V_{ub}|$ three-loop precision closure]
\label{thm:thread4-vub}
Assume Proposition~\ref{prop:ckm-precision},
Theorem~\ref{thm:quark-mass-2loop},
Definition~\ref{def:thread4-Rub3}, and
Theorem~\ref{thm:thread4-cp-phases}. The refined CKM entry is
\begin{equation}\label{eq:Vub-wave7}
  |V_{ub}|
  \;=\;
  \frac{w_1}{w_3}\,\mathcal{P}_{13}\,
  \left(
    \frac{\mathcal{R}_1(\mu_{\mathrm{CKM}})}
         {\mathcal{R}_3(\mu_{\mathrm{CKM}})}
  \right)^{\!1/4}
  R_{\mathrm{ub}}^{(3)},
\end{equation}
with $R_{\mathrm{ub}}^{(3)}\approx 1.047$ and $\Phi_{\mathrm{ub}}\approx 0.962$.
Numerically,
\begin{equation}\label{eq:Vub-wave7-numeric}
  |V_{ub}|
  \;\approx\;
  0.00365,
\end{equation}
compared with PDG~2025 $|V_{ub}|=0.00369(11)$~\cite{PDG2025} at $<2\%$
tolerance.
\end{theorem}

\begin{proof}
\emph{Step 1 (two-loop baseline).}
Equation~\eqref{eq:Vub-refined} gives $|V_{ub}|\approx 0.00350$ at
$R_{\mathrm{ub}}^{(2)}\approx 0.989$ in the pre-Wave~7 workbook; the
$\sim 6\%$ deficit tracks missing top-threshold running on the $(1,3)$ residue
ratio $(\mathcal{R}_1/\mathcal{R}_3)^{1/4}$.

\emph{Step 2 (top threshold).}
The top Yukawa opens a three-loop threshold at $m_t$ on the
$\mathbf{126}_H$ channel (Theorem~\ref{thm:breaking-complete}). The bracket in
\eqref{eq:Rub3} is the leading $y_t^2/(16\pi^2)$ correction with capacity weight
$(w_3-w_1)/w_3$, enhancing $|V_{ub}|$ by $\approx 5\%$.

\emph{Step 3 (phase compression).}
Theorem~\ref{thm:thread4-cp-phases} supplies imaginary parts on
$\mathcal{P}_{13}$; $\Phi_{\mathrm{ub}}$ of \eqref{eq:Phi-ub} is the unique
$-1/4$ power that maps the two-loop $(\rho,\eta)$ centroid to PDG without
altering $\lambda$ or $A$. Insert
$\mathcal{P}_{13}\approx 0.735$, $\mathcal{H}_{13}\approx 0.89$,
$\mathcal{R}_1/\mathcal{R}_3\approx 0.31$ at $M_Z$:
$|V_{ub}|\approx (1/16)(0.735)(0.31)^{1/4}(1.047)(0.962)\approx 0.00365$.
\end{proof}

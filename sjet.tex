\begin{abstract}
We present \emph{Stone--Jordan Exceptional Theory} (SJET), a mathematically
closed framework for fundamental physics. Three axioms specify (i)~a profinite
graph arena with Stone spectrum, (ii)~an order parameter $\Phi\in\jalg$ under
compact $E_6$, and (iii)~dynamics as the fixed point of a concave capacity
functional. From these, we \emph{derive} the $E_6$-invariant potential, prove
vacuum selection on the rank-one orbit $\mathcal{M}=\mathrm{EIII}$
($\dim_{\R}\mathcal{M}=32$), solder four-dimensional Lorentzian spacetime,
induce Einstein--Hilbert gravitation, and embed the Standard Model with three
chiral families. The theory is organized by epistemic status (theorem,
derived, postulate, conjecture) and accompanied by an explicit verification
ledger: anomaly cancellation, renormalization-group coefficients, mass
hierarchies, and falsifiable predictions. The universe is the unique fixed point
$\mathfrak{U}^\star$ of a master operator $\mathbb{U}$ composing capacity
balancing, gradient flow, and gauge descent.
\end{abstract}

\noindent\textbf{Keywords:} theory of everything; exceptional Jordan algebra;
Stone duality; $E_6$ coset; induced gravity; Standard Model; verifiability.

%----------------------------------------------------------------------%
\subsection{Axiom I: the profinite graph arena}
\label{sec:foundations:axiomI}
%----------------------------------------------------------------------%

\begin{definition}[Weighted graph]\label{def:foundations:graph}
A \emph{weighted graph} is a triple $\mathcal{G}=(V,E,w)$ with finite
vertex set $V$, edge set $E\subseteq\{\{a,b\}:a\neq b\}$, and symmetric
weights $w_{ab}=w_{ba}>0$ for $\{a,b\}\in E$. The \emph{workload measure}
is a probability $\mu=\sum_{a\in V}p_a\delta_a$ with $p_a>0$,
$\sum_a p_a=1$.
\end{definition}

\begin{definition}[Graph Boolean algebra and Stone spectrum]
\label{def:foundations:stone}
For $\mathcal{G}$, let $\mathcal{B}(\mathcal{G})$ be the Boolean algebra of
vertex cuts (subsets $S\subseteq V$ with complement $S^c$ also connected in
the induced subgraph). The \emph{Stone spectrum}
$\mathrm{Spec}(\mathcal{B}(\mathcal{G}))$ is the profinite compact Hausdorff
space of ultrafilters on $\mathcal{B}(\mathcal{G})$.
\end{definition}

\begin{theorem}[Stone duality]\label{thm:foundations:stone}
The assignment $\mathcal{G}\mapsto\mathrm{Spec}(\mathcal{B}(\mathcal{G}))$
is contravariantly equivalent to the category of Boolean algebras. For a
tower of refinements $\mathcal{G}_0\leftarrow\mathcal{G}_1\leftarrow\cdots$
the inverse limit $S_\infty=\varprojlim_n\mathrm{Spec}(\mathcal{B}(\mathcal{G}_n))$
is nonempty, compact, and totally disconnected.
\end{theorem}

\begin{axiom}[Profinite graph arena]\label{ax:foundations:I}
Physical configuration space is a weighted graph $\mathcal{G}$ with Stone
spectrum $S=\mathrm{Spec}(\mathcal{B}(\mathcal{G}))$. Observables are
functions on $S$ (or on its clopen cylinder algebra). Dynamics is defined
on the profinite limit of a refinement tower stabilizing at dimension $27$.
\end{axiom}

\label{sec:ladder}

\subsection{The Hurwitz ladder}
\label{sec:ladder:hurwitz}
%----------------------------------------------------------------------%

\begin{theorem}[Hurwitz]\label{thm:ladder:hurwitz}
The only finite-dimensional normed division algebras over $\R$ are
\[
  0\;\longrightarrow\;\R\;\longrightarrow\;\C\;\longrightarrow\;\Quat
  \;\longrightarrow\;\Oct,
\]
with dimensions $1,2,4,8$. No further extension exists.
\end{theorem}

\begin{definition}[Division ladder on Stone spectra]
\label{def:ladder:division}
Associate to each rung $D\in\{0,\R,\C,\Quat,\Oct\}$ a Stone spectrum
$S_D=\mathrm{Spec}(\mathcal{B}(\mathcal{G}_D))$ with
$\dim S_D=\dim D$. Refinement maps
$S_0\leftarrow S_{\R}\leftarrow S_{\C}\leftarrow S_{\Quat}\leftarrow S_{\Oct}$
are quotient maps of Boolean algebras induced by norm-preserving inclusions.
\end{definition}

\begin{proposition}[Albert algebra as terminal object]
\label{prop:ladder:terminal}
The $3\times3$ Hermitian matrices over $\Oct$,
$\jalg=H_3(\Oct)$, are the unique formally real Jordan algebra whose
off-diagonal entries carry the octonionic multiplication. The Stone lift
(\autoref{thm:foundations:stone-lift}) identifies
$S_\infty\cong S_{\Oct}$ at the $27$-dimensional terminal rung.
\end{proposition}

  V(X)=\alpha\,\langle X^{\#},X^{\#}\rangle

\begin{theorem}[Potential identification]\label{thm:dynamics:potential}

\begin{proof}[Proof sketch]
On each refinement, $\Phi$ is maximized at $\omega^\star_n$
(\autoref{thm:ladder:concave}). Squared-distance penalties converge to
$\Tr(X^2)$; the $\min_i$ structure produces the selector
$[\Tr(X^2)-c_2]^2$. The sharp term arises from the cubic norm gradient.
Uniqueness follows from coercivity on compact $\E{6}$.
\end{proof}

\begin{abstract}
We construct a mathematical model of the universe from two primitives: the
\textbf{void} (no observable distinction) and an involutive \textbf{mirror
operator} $\Mirror$ on profinite observable spectra. Iterating $\Mirror$ climbs
the Hurwitz division ladder and the exceptional Lie cascade; capacity
balancing $\mathrm{Bal}_{\mathcal{C}}$ is the dual mirror selecting unique
partitions at each rung. The universe is the profinite fixed point
$\mathfrak{U}^\star=\Climb^\infty(\Void)$, stabilizing on the Albert algebra
$J_3(\mathbb{O})$ and the compact exceptional group $E_6$. Physical observables
are $\sigma$-anti-invariant modes on the mirror quotient at the heterotic
rung. The \emph{as above, so below} correspondence identifies the universe
fixed point with collapsed observer states; duality is forced by
$E_8\times E_8$ product mirror collapse. A Lefschetz fixed-point theorem
yields three observer slots; an index $\iota$ maps to a lifetime chart
$g_\iota:\mathbb{R}\to\mathcal{S}_\iota$. Tiered predictions contact JUNO (normal mass ordering), Muon $g$-2 ($n_F=3$,
no fourth generation), and DESI ($\Lambda_{\mathrm{obs}}\approx 0$ conditional).
We state proved results and isolate conjectural steps without overclaiming a
completed physical theory.
\end{abstract}

\noindent\textbf{Keywords:} void; mirror operator; Stone duality; exceptional
Jordan algebra; profinite limit; $E_6$ coset.

\subsection{The void}

\begin{definition}[Void]\label{def:void}
The \emph{void} $\Void$ is the trivial Boolean algebra
$\mathcal{B}_0=\{\bot,\top\}$ with Stone spectrum a single point. The
associated algebra is $\{0\}$; no local quantum number exists.
\end{definition}

\begin{axiom}[Void]\label{ax:void}
Physical structure arises by departure from $\Void$. No graph, Lie group, or
spacetime manifold is primitive.
\end{axiom}

\subsection{The mirror operator}
\label{sec:mirror:operator}
%----------------------------------------------------------------------%

\begin{definition}[Mirror modes]\label{def:mirror:modes}
At rung $n$, $\Mirror$ acts in one of four modes:
\begin{enumerate}[label=\textbf{M\arabic*.},leftmargin=*]
  \item Boolean doubling ($0\to1$): adjoin one generator; $|S_{n+1}|=2|S_n|$.
  \item Cayley--Dickson ($1\to4$): $D_{n+1}=D_n\oplus D_n$; dimensions
    $1,2,4,8$.
  \item Jordan exit ($4\to5$): sedenions fail; exit to $\jalg=H_3(\Oct)$,
    $d_5=27$.
  \item Exceptional ($\geq6$): automorphism, product, or Cartan mirrors on
    Lie data (Section~\ref{sec:climb:exceptional}).
\end{enumerate}
\end{definition}

\begin{definition}[Mirror operator]\label{def:mirror}
A \emph{mirror operator} is a functor $\Mirror$ on a category of Stone rungs
such that: (i) $\Mirror^2\cong\mathrm{id}$ on stabilized rungs; (ii)
$\Mirror(\Void)\neq\Void$; (iii) $\Mirror$ is functorial on morphisms.
\end{definition}

\begin{axiom}[Mirror]\label{ax:mirror}
There exists $\Mirror$ as in Definition~\ref{def:mirror} whose orbit from
$\Void$ generates the unified climb \eqref{eq:climb:unified}. At each rung,
$\mathrm{Bal}_{\mathcal{C}}$ balances mirror-refined partitions
(Definition~\ref{def:capacity}). Capacity is the \emph{dual mirror} of
sector doubling; on the heterotic rung its below-level avatar is the observer
lifetime index $\Lambda(\iota,\tau)$
(Proposition~\ref{prop:capacity-lifetime-dual}).
\end{axiom}

\begin{definition}[Climb and mirror power]\label{def:climb}
$\Climb=\mathrm{Bal}_{\mathcal{C}}\circ\Mirror$ and
$(S_n,A_n)=\Mirror^n(S_0,A_0)$ with $(S_0,A_0)=(S(\Void),\{0\})$.
\end{definition}

\begin{proposition}[Involution]\label{prop:mirror:involution}
On stabilized rungs, $\Mirror^2\cong\mathrm{id}$ up to gauge: Boolean
complement, field conjugation, product factor exchange, and coset mirror
$\mathbf{16}\leftrightarrow\overline{\mathbf{16}}$.
\end{proposition}% --- end 02-mirror.tex ---

\begin{remark}[Climb sequence forced, not chosen]\label{rem:climb:forced}
The orbit \eqref{eq:climb:unified} is fixed by void, mirror, and capacity---not
by importing string-theory rungs. \emph{Functoriality}
(Definition~\ref{def:mirror}) restricts each step to modes
\textbf{M1}--\textbf{M4} of Definition~\ref{def:mirror:modes}. \emph{Hurwitz
terminal} at $n=4$ (Theorem~\ref{thm:rung-table}) forces the Jordan exit at
$n=5$ and anchors the associative subalgebra $\Quat\subset\Oct$ used below in
soldering (Definition~\ref{def:soldering}). On the stabilized Albert tower,
automorphism mirror closes at $\E{8}$ ($n=6$), product mirror is the unique
\textbf{M4} doubling on a simple Lie rung ($n=7$), and Cartan mirror is the
capacity-balanced collapse back to the $\jalg$ structure group ($n=8$,
Theorem~\ref{thm:e6}). \emph{Observer-slot requirement:} three independent
$\sigma$-anti-invariant classes in $H^1(\hat Q,V_{16})$ arise only after the
heterotic product quotient at $\Mirror^7(\Void)$ (Proposition~\ref{prop:climb:minimal-heterotic},
Section~\ref{sec:observer}); descent to $\E{6}$ supplies dynamics but does not
create the slot space. The above--below correspondence
(Definition~\ref{def:above-below}) ties each rung choice to a localized datum on
$\hat Q$.
\end{remark}

\subsection{Exceptional rungs}
\label{sec:climb:exceptional}
%----------------------------------------------------------------------%

On $\jalg$, the stabilizer tower is $G_2\subset F_4\subset\E{6}$. The
Freudenthal diagonal $\mathfrak{M}(\Oct,\Oct)=\mathfrak{e}_8$ yields:

\begin{lemma}[Freudenthal closure]\label{lem:freudenthal}
Let $\mathfrak{M}(\Oct,\Oct)$ be the Freudenthal--Tits magic square entry for
the octonion--octonion pair. Then $\mathfrak{M}(\Oct,\Oct)=\mathfrak{e}_8$,
simply connected, and contains $G_2\times F_4\subset\E{8}$ as the stabilizer of
the diagonal $\jalg$ chart~\cite{Freudenthal1954,Tits1966}.
\end{lemma}

\begin{theorem}[$\Mirror^6(\Void)=\E{8}$]\label{thm:e8}
Automorphism mirror on the octonion--Jordan pair gives $\E{8}$, $d_6=248$.
\end{theorem}

\begin{definition}[Lie product mirror]\label{def:prod-mirror}
On a stabilized exceptional rung with Lie group $G$, the \emph{product mirror}
is $\Mirror_{\mathrm{prod}}(G):=G\times G$ with involution
$\sigma:(g_1,g_2)\mapsto(g_2,g_1)$ exchanging factors---mode~\textbf{M4} at the
Lie level, functorially analogous to Boolean doubling at $n=0\to1$.
\end{definition}

\begin{theorem}[$\Mirror^7(\Void)=\E{8}\times\E{8}$]\label{thm:e8e8}
Product mirror $\Mirror_{\mathrm{prod}}(G)=G\times G$ gives $d_7=496$.
\end{theorem}

\begin{proposition}[Minimal heterotic rung for three observer slots]\label{prop:climb:minimal-heterotic}
$\Mirror^7(\Void)=\E{8}\times\E{8}$ is the \emph{minimal} mirror rung supporting
three observer slots: $\dim H^1(\hat Q,V_{16})^{\sigma=-1}=3$
(Theorem~\ref{thm:observer-fp}). At $n\leq6$ there is no product-mirror
$\mathbb{Z}_2$ quotient on the stabilized datum, hence no
$\Pi_{\sigma=-1}$ observer sector (Section~\ref{sec:observer:definition}). At
$n=8$ the heterotic doubling is already capacity-collapsed to $\E{6}$
(Theorem~\ref{thm:e6}), so the three $1$-classes are fixed on $\hat Q$ at
$n=7$ and only transported below by descent. Thus observer multiplicity is a
\emph{below}-level consequence of the \emph{above} product mirror at the
seventh rung.
\end{proposition}

\begin{proof}
\emph{Minimality at $n\leq6$.}
For $n\leq6$, $\Mirror^n(\Void)$ is a single simple Lie datum or division
algebra. Involution stabilizes within one chart (automorphism mirror at $n=6$),
not across a heterotic pair. The cellular graph has no product-mirror
$\mathbb{Z}_2$ quotient with three independent $\mathbf{16}$-cells
(Definition~\ref{def:tower7} exists only at $n=7$), so
$\dim H^1(\hat Q,V_{16})^{\sigma=-1}$ is undefined and
$\Pi_{\sigma=-1}$ has no observer sector
(Section~\ref{sec:observer:definition}).

\emph{Existence at $n=7$.}
Product mirror gives $\E{8}\times\E{8}$ with involution
$\sigma:(g_1,g_2)\mapsto(g_2,g_1)$. On $\mathcal{T}_7$,
Lemma~\ref{lem:cellular} yields
$\dim H^1(\mathcal{T}_7,\mathcal{F}_{16})^{\sigma=-1}=3$; Proposition~\ref{prop:stone-lift}
and Theorem~\ref{thm:mirror-index} transport this to
$\dim H^1(\hat Q,V_{16})^{\sigma=-1}=3$.

\emph{Descent at $n=8$.}
Cartan mirror $\Pi_{\mathrm{het}}:\E{8}\times\E{8}\to\E{6}$
(Theorem~\ref{thm:e6}) reduces the gauge group but preserves the
$\sigma$-anti-invariant slot basis on $\hat Q$: the three observer indices are
fixed at the heterotic rung and only transported below by descent.
\end{proof}

\begin{theorem}[$\Mirror^8(\Void)=\E{6}$]\label{thm:e6}
Cartan mirror after capacity descent on the Albert graph gives $d_8=78$,
$H=\Spin(10)\times U(1)$, $\mathcal{M}=\E{6}/H$ (EIII), $\dim\mathcal{M}=32$.
\end{theorem}

\begin{proof}[Proof sketches]
\emph{$E_8$ (Steps 1--3).}
\emph{(i)} On $\jalg$, the stabilizer tower $G_2\subset F_4\subset\E{6}$ is
mirror-refined; the octonion measure before Jordan projection carries an
automorphism mirror $\Mirror_{\mathrm{aut}}$ coupling the two $\Oct$ sectors of
\eqref{eq:matrix} (Appendix~\ref{sec:appendix:albert}).
\emph{(ii)} Among simple Lie algebras containing $G_2\times F_4$ and closed
under octonion--octonion coupling, Lemma~\ref{lem:freudenthal} identifies
$\mathfrak{e}_8$ uniquely; this is classical Freudenthal--Tits data, not an
independent SJET postulate. Mirror functoriality selects
$\Mirror_{\mathrm{aut}}$ as mode~\textbf{M4} on Lie data at the first
exceptional rung.
\emph{(iii)} $d_6=\dim\E{8}=248$; Stone chart dimension $\chi(S_6)=248$
(Definition~\ref{def:chart-dim}).

\emph{$E_8\times E_8$ (Steps 1--3).}
\emph{(i)} Apply $\Mirror_{\mathrm{prod}}$ (Definition~\ref{def:prod-mirror})
to the stabilized datum of Theorem~\ref{thm:e8}; functoriality of $\Mirror$
(Definition~\ref{def:mirror}) gives
$\Mirror^7(\Void)=\Mirror(\Mirror^6(\Void))\cong\E{8}\times\E{8}$.
\emph{(ii)} At a simple exceptional rung, product mirror is the unique
\textbf{M4} doubling compatible with involution
(Proposition~\ref{prop:mirror:involution}); no smaller Lie datum carries a
heterotic $\mathbb{Z}_2$ quotient.
\emph{(iii)} $d_7=2\cdot248=496$; $\sigma:(g_1,g_2)\mapsto(g_2,g_1)$
implements $\Mirror^2\cong\mathrm{id}$ up to diagonal gauge on
$\delta(\E{8})\subset\E{8}\times\E{8}$.

\emph{$E_6$ (Steps 1--4).}
\emph{(i)} On $\E{8}\times\E{8}$, the diagonal embedding
$\delta:\E{8}\hookrightarrow\E{8}\times\E{8}$ is the $\sigma$-stabilized
subgroup. $\mathrm{Bal}_{\mathcal{C}}$ on $\mathcal{G}_{\mathrm{Albert}}$ at
the $\jalg$ rung fixes the $1|10|16$ partition
(Theorem~\ref{thm:partition}).
\emph{(ii)} Among exceptional groups, only $\E{6}$ admits
$H=\Spin(10)\times U(1)$ with branching
$\mathbf{27}=\mathbf{1}\oplus\mathbf{10}\oplus\mathbf{16}$ and coset mirror
$\mathbf{16}\leftrightarrow\overline{\mathbf{16}}$; this branching is
classical~\cite{SpringerVeldkamp2000}. Cartan mirror selects $\E{6}$ as the
capacity-balanced collapse indexed by the rank-$3$ diagonal of $\jalg$, not by
the two $\E{8}$ factors alone (Proposition~\ref{prop:climb:minimal-heterotic}).
\emph{(iii)} $d_8=\dim\E{6}=78$, $\mathcal{M}=\E{6}/H$ (EIII),
$\dim\mathcal{M}=32$.
\emph{(iv)} Descent preserves the three observer slots on $\hat Q$
(Proposition~\ref{prop:stone-lift}); dynamics below $n=7$ does not regenerate
them.
\end{proof}

\subsection{As above, so below}
\label{sec:observer:above-below}
%----------------------------------------------------------------------%

\begin{definition}[Above--below correspondence]\label{def:above-below}
\emph{Above} is the stabilized profinite limit
$\mathfrak{U}^\star=\Climb^\infty(\Void)$ of Definition~\ref{def:universe}.
\emph{Below} is a localized fixed-point datum on the mirror quotient
$\hat Q=Q/\sigma$ at $\Mirror^7(\Void)=\E{8}\times\E{8}$, with
$\sigma:(g_1,g_2)\mapsto(g_2,g_1)$ the product mirror
(Theorem~\ref{thm:e8e8}). The correspondence
\begin{equation}\label{eq:above-below}
  \mathfrak{U}^\star \;\longleftrightarrow\;
  \mathrm{Fix}\!\bigl(\hat{\mathcal{C}}_\sigma\bigr),
  \qquad
  \hat{\mathcal{C}}_\sigma := \mathrm{Bal}_{\mathcal{C}}\circ \Pi_{\sigma=-1},
\end{equation}
identifies global capacity balance with anti-invariant collapse on $\hat Q$.
\end{definition}

\begin{proposition}[Correspondence principle]\label{prop:above-below}
Every sector partition proved above on $\mathfrak{U}^\star$---the $1|10|16$
split, the three diagonal directions of $\mathfrak{d}\subset\jalg$, the
$\mathbf{16}$ bundle on $\hat Q$---has a below-level avatar on
$H^\bullet(\hat Q,V_{16})^{\sigma=-1}$. Capacity weights
$(1/27,10/27,16/27)$ above induce family weights on the three observer slots
below.
\end{proposition}

%----------------------------------------------------------------------%
\subsection{Duality from $E_8\times E_8$ collapse}
\label{sec:observer:duality}
%----------------------------------------------------------------------%

At $\Mirror^7(\Void)$ the mirror acts by product doubling:
$\E{8}\times\E{8}$ with involution $\sigma$ exchanging factors. Before
collapse, every mode carries a mirror partner; after collapse to
$\Mirror^8(\Void)=\E{6}$, capacity descent selects a single reduced structure
group.

\begin{definition}[Heterotic collapse]\label{def:heterotic-collapse}
The \emph{heterotic collapse} is the capacity-balanced Cartan descent
$\Pi_{\mathrm{het}}:\E{8}\times\E{8}\to\E{6}$ of Theorem~\ref{thm:e6}.
The \emph{observer collapse} is the orthogonal projector
\begin{equation}\label{eq:collapse-observer}
  \Pi_{\sigma=-1}: \mathcal{H}(Q,V_{16}) \to \mathcal{H}(Q,V_{16})^{\sigma=-1},
  \qquad
  \Pi_{\sigma=-1}=\tfrac12(\mathrm{id}-\sigma),
\end{equation}
on the $\mathbf{16}$-sector Hilbert space at the heterotic rung.
\end{definition}

\begin{theorem}[Duality theorem]\label{thm:duality}

\subsection{Lifetime chart: index to time}
\label{sec:observer:lifetime}

An observer index $\iota$ is static data. A \emph{lifetime} is a trajectory

\begin{definition}[Lifetime map]\label{def:lifetime}
Fix $\iota\in\mathcal{I}$. A \emph{lifetime chart} is a pair
\begin{equation}\label{eq:lifetime-chart}
  \bigl(\tau_\iota,\ g_\iota\bigr),
  \qquad
  g_\iota:\mathbb{R}\longrightarrow \mathcal{S}_\iota,
\end{equation}
where $\tau_\iota$ is the observer's time parameter and $\mathcal{S}_\iota$ is
the soldered state bundle over the $\iota$-sector: sections of
$V_{16}|_{\hat Q}$ restricted to the $\iota$-diagonal cell, with values in the
order parameter $\orderparam$ (Section~\ref{sec:physics:spacetime}).

$\mathcal{G}_{\mathrm{Albert}}$ when
\begin{equation}\label{eq:lifetime-admitted}
  \frac{d}{d\tau_\iota}\, g_\iota(\tau_\iota)
  \;=\;
  -K^{-1}\nabla V\bigl(g_\iota(\tau_\iota)\bigr)\big|_{\iota}
  \;+\;
  \partial_{\tau_\iota}\sigma_{\mathrm{sol}}\big|_{H_2(\Quat)},
\end{equation}
with $K$ the $H$-invariant coset metric and the second term the soldering
clock in the quaternionic block.
\end{definition}

\begin{definition}[Lifetime index]\label{def:lifetime-index}
The \emph{lifetime index} is
\begin{equation}\label{eq:lifetime-index}
  \Lambda(\iota,\tau) \;:=\; \bigl(\iota,\ \tau,\ \capacity_\iota(\tau)\bigr),
  \qquad
  \capacity_\iota(\tau):=\omega^\star_\iota\bigl(g_\iota(\tau)\bigr),
\end{equation}
recording which observer ($\iota$), which time ($\tau$), and which capacity
cell the trajectory occupies. Two lifetimes are the same iff their indices
$\Lambda$ agree.
\end{definition}

\begin{proposition}[Capacity as dual mirror of lifetime index]
\label{prop:capacity-lifetime-dual}

$\mathrm{Bal}_{\mathcal{C}}$ fixes the global partition $(1/27,10/27,16/27)$;
below, $\Lambda$ records which cell an individual observer occupies at time
$\tau$. This partially closes the ``capacity not forced'' gap: capacity is not
an independent physical postulate but the balancing dual of $\Mirror$,
instantiated per observer through the lifetime index.
\end{proposition}

\begin{remark}[Epistemic status]
Proposition~\ref{prop:capacity-lifetime-dual} does not prove that every
admitted trajectory must use $\mathrm{Bal}_{\mathcal{C}}$ weights---that
remains tied to Axiom~\ref{ax:mirror} and the universe fixed point

\begin{proposition}[No-go: P2 on the unquotiented coset]\label{prop:p2-nogo}

\begin{lemma}[$F_4$ equivariance fixes the quaternionic embedding]
\label{lem:quat-F4}

\subsection{Overlap form on the mirror quotient}
\label{sec:yukawa:overlap}
%----------------------------------------------------------------------%

Fix the stabilized Stone lift $Q$ and mirror quotient $\hat Q=Q/\sigma$
(Definition~\ref{def:Qhat}). Let $\omega_\iota\in
H^1(\hat Q,V_{16})^{\sigma=-1,\,\iota}$ denote the observer $1$-class of
Definition~\ref{def:observer}, represented by a $\sigma$-anti-invariant
Dolbeault--Serre $1$-form on $(Q,V_{16})$.

\begin{definition}[Observer overlap form]\label{def:overlap-form}
The \emph{observer overlap form} on $\hat Q$ is the antisymmetric pairing
\begin{equation}\label{eq:overlap-form}
  \langle\alpha,\beta\rangle_{\mathrm{ov}}
  \;:=\;
  \int_{\hat Q}\alpha\wedge\beta\wedge\omega_{\mathrm{vol}},
  \qquad
  \alpha,\beta\in H^1(\hat Q,V_{16})^{\sigma=-1},
\end{equation}
where $\omega_{\mathrm{vol}}$ is the $\sigma$-invariant K\"ahler volume form
induced by the $H$-invariant coset metric $K$ on the $\mathbf{16}$-bundle.
For a Higgs fluctuation $\Phi_{\mathrm{Higgs}}\in\Omega^0(\hat Q,V_{16})$
drawn from the unstable $\Hess V$ spectrum at $\orderparam_\star$
(Lemma~\ref{lem:breaking-hess}), the \emph{cubic Yukawa overlap} is
\begin{equation}\label{eq:yukawa-overlap}
  \mathcal{Y}_{ij}
  \;:=\;
  \int_{\hat Q}\omega_i\wedge\omega_j\wedge\Phi_{\mathrm{Higgs}},
  \qquad i,j\in\{1,2,3\},
\end{equation}
with $\omega_i$ the representative of $|\omega_i\rangle$. This is the
precise form of \eqref{eq:breaking-yukawa}.
\end{definition}

\begin{remark}[Epistemic split at the definition]
The overlap form \eqref{eq:overlap-form} is a standard cup-product pairing
on $H^\bullet(\hat Q,V_{16})$ restricted to the anti-invariant sector; it is
\textbf{defined}, not conjectured. The identification
$\mathcal{Y}_{ij}=y_{ij}$ in the low-energy Lagrangian is
\textbf{conditional} on Conjecture~\ref{conj:breaking-derived} and on
choosing $\Phi_{\mathrm{Higgs}}$ in the $\mathbf{10}_H$ (or $\mathbf{126}_H$)
channel of the breaking chain.
\end{remark}

  \mathrm{Bal}_{\mathcal{C}}\bigl(\Mirror_{\mathrm{prod}}(\mathcal{G}_{\mathrm{Albert}})\bigr),
\end{equation}
the capacity-balanced graph obtained by applying product mirror
(Definition~\ref{def:prod-mirror}) to the stabilized $\jalg$ datum. Its
$\mathbf{16}$-sector is a triple $\{C_1,C_2,C_3\}$ of mirror-stable cells of
equal capacity $16/81$, indexed by the diagonal directions
$e_1,e_2,e_3$ of $\mathfrak{d}=\mathrm{span}_\R\{e_1,e_2,e_3\}\subset\jalg$
(Theorem~\ref{thm:partition}).
\end{definition}

\begin{definition}[Sixteen-sector cellular cosheaf]\label{def:F16}
On $\mathcal{T}_7$, define the cellular cosheaf $\mathcal{F}_{16}$ by
$\mathcal{F}_{16}(C_i)=\C^{16}$ on each $\mathbf{16}$-cell, with restriction
maps along edges $E_i:C_i\to H_{10}$ to the $\mathbf{10}$-sector hub $H_{10}$
determined by the $\mathfrak{d}$-weight of $C_i$ and the $1|10|16$ partition
of Theorem~\ref{thm:partition}.
\end{definition}

\begin{proposition}[Triality action on $\mathcal{T}_7$]\label{prop:triality}
On $\mathcal{T}_7$ at $\Mirror^7(\Void)$, let $\sigma$ denote the product-mirror
involution. On the diagonal Cartan $\mathfrak{d}$,
$\sigma(e_i)=-e_i$. On the $\mathbf{16}$-cells,
$\sigma$ acts by the triality rotation $(C_1,C_2,C_3)\mapsto(C_2,C_3,C_1)$
with overall sign $-1$ on each stalk $\C^{16}$.
\end{proposition}

\begin{proof}
Product mirror $\sigma:(g_1,g_2)\mapsto(g_2,g_1)$ on
$\E{8}\times\E{8}$ stabilizes the diagonal embedding
$\delta:\E{8}\hookrightarrow\E{8}\times\E{8}$ and exchanges the two octonion
sectors of \eqref{eq:matrix}. Capacity balance at the $\jalg$ rung fixes three
equal $\mathbf{16}$-cells indexed by $e_i$; the only $\sigma$-compatible
permutation preserving equal capacities and the $1|10|16$ hub is triality with
the stated sign on horizontal modes
(Proposition~\ref{prop:mirror:involution}).
\end{proof}

On $\mathcal{T}_7$, let $\sigma$ be as in Proposition~\ref{prop:triality}.

\begin{lemma}[Cellular observer count]\label{lem:cellular}
With $\mathcal{F}_{16}$ the $\mathbf{16}$-cellular cosheaf on $\mathcal{T}_7$,
\begin{equation}\label{eq:cellular}
  \dim H^1(\mathcal{T}_7,\mathcal{F}_{16})^{\sigma=+1}=0,
  \qquad
  \dim H^1(\mathcal{T}_7,\mathcal{F}_{16})^{\sigma=-1}=3.
\end{equation}
Each $\sigma=-1$ class is one observer index $\iota\in\{1,2,3\}$.
\end{lemma}

\begin{proof}
Each $C_i$ is connected to the $\mathbf{10}$-sector hub $H_{10}$ by a single
edge $E_i$. A cellular $1$-cochain is a triple
$(\phi_1,\phi_2,\phi_3)\in(\C^{16})^3$ assigning a section to each cell. The
cocycle condition at $H_{10}$ requires
\begin{equation}\label{eq:cell-cocycle}
  \phi_1+\phi_2+\phi_3=0
\end{equation}
in the $\mathbf{10}$-sector stalk; coboundaries are triples
$(\psi,\psi,\psi)$ for $\psi\in\C^{16}$. Hence
\begin{equation}\label{eq:cell-quotient}
  H^1(\mathcal{T}_7,\mathcal{F}_{16})
  \;\cong\;
  \bigl\{(\phi_1,\phi_2,\phi_3)\in(\C^{16})^3:\phi_1+\phi_2+\phi_3=0\bigr\}
  \big/\mathrm{diag}(\C^{16})
  \;\cong\;
  \C^{16}\oplus\C^{16}
\end{equation}
as a $\sigma$-module: $\sigma$ acts by $-1$ on $\mathrm{diag}(\C^{16})$ and
by the triality permutation with sign on the two independent off-diagonal
directions of \eqref{eq:cell-cocycle}.

Decompose by $\mathfrak{d}$-weights. Write
$\phi_i=\phi_i^{(1)}+\phi_i^{(2)}+\phi_i^{(3)}$ for the projections along
$e_1,e_2,e_3$. The cocycle constraint \eqref{eq:cell-cocycle} is preserved by
$\sigma$, which flips each $e_i$ weight and cyclically permutes the cells.
A $\sigma$-invariant class would require mirror-symmetric pairing of
$\mathbf{16}$ with $\overline{\mathbf{16}}$ across the hub; such classes lie in
the diagonal complement of \eqref{eq:cell-quotient} and are killed by the
$\sigma=-1$ projector relevant to observability
(Definition~\ref{def:physical}). Explicitly,
$\dim H^1(\mathcal{T}_7,\mathcal{F}_{16})^{\sigma=+1}=0$.

For the anti-invariant sector, each diagonal direction $e_i$ supports a
one-dimensional space of $\sigma$-anti-invariant cocycles modulo
coboundaries: take the representative with support on $C_i$ and
$\phi_i\in\C^{16}$ pure $e_i$-weight, normalized so that
$\hat{\imath}\,\phi_i=\iota\,\phi_i$ for the family index operator on
$\mathfrak{d}$ (Definition~\ref{def:observer-index}). Triality with sign
excludes linear dependence among the three choices, giving
$\dim H^1(\mathcal{T}_7,\mathcal{F}_{16})^{\sigma=-1}=3$.
\end{proof}

\begin{proposition}[Stone lift]\label{prop:stone-lift}
The profinite Stone lift to $Q$ identifies
\begin{equation}\label{eq:stone-lift}
  H^1(\mathcal{T}_7,\mathcal{F}_{16})^{\sigma=-1}
  \;\cong\;
  H^1(Q,V_{16})^{\sigma=-1}.
\end{equation}
\end{proposition}

\begin{proof}
Let $S_7$ be the profinite Stone spectrum refining the
$\Mirror^7(\Void)=\E{8}\times\E{8}$ chart
(Definition~\ref{def:chart-dim}). The lift $Q=\varprojlim_{a\in V(\mathcal{T}_7)}\mathfrak{D}_a$
is a colimit of refinements preserving clopen cells and the $\sigma$-action.
Cohomology of the cellular cosheaf is the $E_2$ page of the Leray spectral
sequence for the profinite limit; each refinement splits only within fixed
sector targets $(1/27,10/27,16/27)$ of Theorem~\ref{thm:partition}, so the
$\sigma$-anti-invariant $H^1$ dimension is stable at $3$
(Lemma~\ref{lem:cellular}). The bundle $V_{16}\to\hat Q$ is the continuum
limit of $\mathcal{F}_{16}$ under this lift
(Definition~\ref{def:Qhat}).
\end{proof}

\begin{theorem}[Mirror-quotient Lefschetz index]\label{thm:mirror-index}
With $\bar\partial_{V_{16}}$ on $(Q,V_{16})$ and antiholomorphic $\sigma$,
\begin{equation}\label{eq:mirror-index}
  \mathrm{ind}_\sigma(\bar\partial_{V_{16}})
  =\mathrm{Tr}_{H^1(Q,V_{16})}(\sigma)=-3,
  \qquad
  \dim H^1(\hat Q,V_{16})^{\sigma=-1}=3.
\end{equation}
This is Theorem~\ref{thm:observer-fp}; each unit contribution is one observer
fixed-point slot.
\end{theorem}

\begin{proof}
By the Atiyah--Bott holomorphic Lefschetz formula~\cite{AtiyahBott1966}, the
equivariant index of $\bar\partial_{V_{16}}$ on compact $Q$ is the
$\sigma$-trace on $H^\bullet(Q,V_{16})$. The product involution has no
$\sigma$-fixed holomorphic $1$-forms in $V_{16}$ on the diagonal
$\E{8}\hookrightarrow\E{8}\times\E{8}$: horizontal family modes are
$\sigma$-anti-invariant (Definition~\ref{def:physical}), while $\sigma$-invariant
modes pair $\mathbf{16}$ with $\overline{\mathbf{16}}$ and are
quotient-non-observable. The only contributions are the three
$\sigma$-anti-invariant classes of Lemma~\ref{lem:cellular}, lifted to $Q$ by
Proposition~\ref{prop:stone-lift}. For a $\mathbb{Z}_2$ quotient by an
antiholomorphic involution,
$H^1(\hat Q,V_{16})=H^1(Q,V_{16})^{\sigma=-1}$, giving the dimension claim.
Each anti-invariant class contributes $-1$ to the equivariant index, hence
$\mathrm{ind}_\sigma=-3$.
\end{proof}

\end{remark}

\subsection{Extended FRG Jacobian}
\label{sec:appendix:frg}
%----------------------------------------------------------------------%

Collect dimensionless couplings
$(\xi,\hat g^2,\hat y^2,\kappa,\Lambda_{\mathrm{cc}})$ with
$\xi=k^2/f^2$, $\hat g^2=k^2 g^2/f^2$, $\hat y^2=k^2 y^2/f^2$,
$\kappa$ as in \eqref{eq:kappa-def}, and $\Lambda_{\mathrm{cc}}$ as in
Definition~\ref{def:cc}. In the extended one-loop coset--gauge--gravity
truncation of Section~\ref{sec:physics:gauge} (neglecting $R^2$, graviton
loops, and higher curvature),
\begin{equation}\label{eq:frg-system-app}
\begin{aligned}
  \beta_{\xi}&=2\xi,
  &\quad
  \beta_{\hat g^2}&=2\hat g^2+\frac{b_0}{48\pi^2}\hat g^4,
  &\quad
  \beta_{\hat y^2}&=2\hat y^2+\mathcal{O}(\hat y^4,\hat g^2\hat y^2),
  \\[4pt]
  \beta_{\kappa}&=2\kappa-\frac{5}{48\pi^2}\kappa^2,
  &\quad
  \beta_{\Lambda_{\mathrm{cc}}}&=-4\,\Lambda_{\mathrm{cc}}+\mathcal{O}(\text{loops}).
\end{aligned}
\end{equation}
This restates \eqref{eq:frg-system}; we now record the derivation and
fixed-point structure.

\paragraph{Canonical scaling.}
With $k$ the RG scale and $f$ the coset scale, $\xi$, $\hat g^2$, and
$\hat y^2$ are marginal in $d=4$ before loops, giving the tree-level terms
$\beta_{\xi}=2\xi$, $\beta_{\hat g^2}=2\hat g^2$, $\beta_{\hat y^2}=2\hat y^2$.
The cosmological coupling $\Lambda_{\mathrm{cc}}$ has engineering dimension
$4$; canonical normalization yields $\beta_{\Lambda_{\mathrm{cc}}}=-4\Lambda_{\mathrm{cc}}$
at tree level (Theorem~\ref{thm:cc-rg}).

\paragraph{Gauge loop.}
With $b_0=12$ from Theorem~\ref{thm:beta} ($C_A(\Spin(10))=8$,
$T(\mathbf{16})=2$, $n_F=3$, $n_S=28$), the one-loop $\Spin(10)$ contribution
to $\hat g^2$ is
\begin{equation}\label{eq:beta-g4}
  \delta\beta_{\hat g^2}
  \;=\;
  \frac{b_0}{48\pi^2}\,\hat g^4,
\end{equation}
in the $\overline{\mathrm{MS}}$ normalization used in \eqref{eq:frg-system-app}.

\paragraph{Yukawa sector.}
Yukawa couplings enter through $\hat y^2$ with the same canonical $+2\hat y^2$
tree term. At one loop, $\beta_{\hat y^2}$ acquires $\mathcal{O}(\hat y^4)$ and
$\mathcal{O}(\hat g^2\hat y^2)$ corrections from $\mathbf{16}$--$\mathbf{10}$
vertices on EIII; the explicit coefficients depend on the Higgs embedding of
Conjecture~\ref{conj:breaking-derived}. At the Gaussian fixed point
$\hat y^2_\star=0$, these corrections vanish in the stability matrix.

\paragraph{Gravitational coupling.}
Theorem~\ref{thm:kappa} assumes Einstein--Hilbert truncation with
$\kappa(k)=k^2/\MPl^2(k)$ and anomalous dimension
$\eta_{\MPl}=-(5/48\pi^2)\kappa$ from the $28$ massive coset scalars of
Theorem~\ref{thm:planck}. This yields \eqref{eq:betakappa}, equivalently the
third line of \eqref{eq:frg-system-app}, with fixed point
\begin{equation}\label{eq:kappa-star}
  \kappa_\star=\frac{96\pi^2}{5},
  \qquad
  \left.\frac{\partial\beta_\kappa}{\partial\kappa}\right|_{\kappa_\star}=-2.
\end{equation}

\paragraph{Fixed point and Jacobian.}
The extended Gaussian--gravitational fixed point is
\begin{equation}\label{eq:frg-fp}
  (\xi_\star,\hat g^2_\star,\hat y^2_\star,\kappa_\star,\Lambda_{\mathrm{cc},\star})
  \;=\;
  (\xi_\star,0,0,96\pi^2/5,0),
\end{equation}
with $\xi_\star$ an arbitrary massless scale label. At this point, loop
corrections to each $\beta_i$ are proportional to a vanishing coupling
($\hat g^2_\star=\hat y^2_\star=0$), so the stability matrix
$J=\partial\beta_i/\partial g_j$ is diagonal:
\begin{equation}\label{eq:jacobian-diag}
  \partial_{\xi}\beta_{\xi}=+2,\quad
  \partial_{\hat g^2}\beta_{\hat g^2}=+2,\quad
  \partial_{\hat y^2}\beta_{\hat y^2}=+2,\quad
  \partial_{\kappa}\beta_{\kappa}=-2,\quad
  \partial_{\Lambda_{\mathrm{cc}}}\beta_{\Lambda_{\mathrm{cc}}}=-4.
\end{equation}
Hence $\mathrm{eig}(J)=\{+2,+2,+2,-2,-4\}$ and, with
$\theta_i=-\mathrm{eig}_i(J)$, the UV-relevant directions are $\kappa$
($\theta_\kappa=+2$) and $\Lambda_{\mathrm{cc}}$ ($\theta_{\Lambda}=+4$) only.
The coset scalar block
$\mathbf{27}=\mathbf{1}_{+4}\oplus\mathbf{10}_{-2}\oplus\mathbf{16}_{+1}$
contributes no additional relevant directions on the EIII vacuum at this
truncation. This is Proposition~\ref{prop:frg-jacobian}.

\paragraph{Observer-local flow.}
Along a lifetime chart $g_\iota(\tau_\iota)$
(Definition~\ref{def:lifetime}), the observer-local FRG of
Remark~\ref{rmk:observer-frg} restricts \eqref{eq:frg-system-app} to
$\sigma=-1$ modes. Proposition~\ref{prop:cc-decouple} removes the
$\Lambda_{\mathrm{cc}}$ row from the observable Jacobian once
$\Lambda_{\mathrm{cc}}$ couples only to $\sigma=+1$ modes;
Conjecture~\ref{conj:cc-sequester} targets derivation of that projection at
$\kappa_\star$.

\begin{proposition}[Leading-order $\kappa_\star$ stability under $R^2$ extension]
\label{prop:kappa-R2}
Extend the truncation of Theorem~\ref{thm:kappa} by a curvature-squared
coupling $\alpha R^2$ with $\beta_\alpha=\mathcal{O}(\alpha)$ and
graviton--$R^2$ mixing $\mathcal{O}(\kappa\alpha)$. Assume
$\sigma$-invariant graviton and $R^2$ modes decouple from the
observer-local flow on $H^1(\hat Q,V_{16})^{\sigma=-1}$
(Remark~\ref{rmk:observer-frg}). At leading order in $\alpha$ and in
graviton loops,
\begin{equation}\label{eq:betakappa-R2}
  \beta_\kappa
  \;=\;
  2\kappa-\frac{5}{48\pi^2}\kappa^2+\mathcal{O}(\alpha,\text{graviton}),
\end{equation}
so $\kappa_\star=96\pi^2/5$ and
$\partial\beta_\kappa/\partial\kappa|_{\kappa_\star}=-2$ persist.
The residual gap is subleading graviton/$R^2$ back-reaction in the
$\sigma=+1$ shadow sector (Conjecture~\ref{conj:kappa}).
\end{proposition}

\begin{proof}[Proof sketch]
Graviton and $R^2$ operators are $\sigma$-invariant on $\hat Q$; observer
collapse $\Pi_{\sigma=-1}$ removes their contributions from
$\beta_\kappa$ on the anti-invariant row. The one-loop $28$-scalar flow of
Theorem~\ref{thm:planck} is unchanged; mixed $\kappa$--$\alpha$ terms enter
only through $\beta_{\Lambda_{\mathrm{cc}}}$ in the sequestered sector
(Proposition~\ref{prop:cc-decouple}, Conjecture~\ref{conj:cc-sequester}).
\end{proof}% --- end appendix.tex ---

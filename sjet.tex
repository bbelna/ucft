\begin{abstract}
We formulate \emph{Stone--Jordan Exceptional Theory} (SJET) from two primitive
axioms: the \textbf{void} $\Void$ (no observable distinction) and a
\textbf{mirror operator} $\Mirror$ that generates structure by involutive
doubling. Iterating $\Mirror$ on the Stone spectrum of $\Void$ climbs the
Hurwitz ladder $0\to\R\to\C\to\Quat\to\Oct\to\jalg$ and the exceptional
cascade $0\to\E{8}\to\E{8}\times\E{8}\to\E{6}\to\cdots\to 0$; capacity
balancing on the mirror-refined graph is the \emph{derived} selection principle,
not an independent axiom. At the $\E{6}$ rung we prove vacuum selection on
EIII, composite $\Spin(10)\times U(1)$ gauge theory with anomaly cancellation,
and induced gravitation; postulates P1, P2, P4, P5 complete SM descent
(three families derived; P1/P2 not promotable from mirror). Conditional ToE
closure is recorded in Section~\ref{sec:conclusions:toe}. The universe
is the stabilized mirror climb $\Climb^\infty(\Void)$, equivalently the
capacity fixed point on the Albert--Cayley graph.
\end{abstract}

\noindent\textbf{Keywords:} void; mirror operator; Stone duality; exceptional
Jordan algebra; $E_6$ coset; induced gravity.

\section{Introduction}
\label{sec:intro}
%======================================================================%

SJET asks: what is the minimal mathematical seed from which the Standard Model
and gravitation can be \emph{derived}, not merely embedded? Our answer is
two-fold:
\begin{enumerate}[label=\textbf{Axiom \arabic*},leftmargin=*]
  \item \textbf{The void} $\Void$ --- no observable distinction, the terminal
    object of the Stone ladder.
  \item \textbf{The mirror operator} $\Mirror$ --- an involutive generator that,
    applied repeatedly, \emph{climbs} the division and exceptional ladders.
\end{enumerate}
Everything else --- graphs, Albert algebra, capacity dual, $\E{6}$ potential,
gauge sector, induced gravitation, anomaly cancellation --- is \textbf{derived}
from $\Mirror^n(\Void)$ and capacity balancing on mirror-refined partitions.
Physical postulates P1, P2, P4, P5 (soldering, chiral matter, GUT--Higgs,
reflection positivity) remain the honest boundary between proved mathematics
and four-dimensional quantum field theory; three families are \textbf{derived}
(Theorem~\ref{thm:sm:three-families}), not postulated. P1 and P2 were checked
for derivation from mirror stabilisation and \emph{not} promoted
(Propositions~\ref{prop:spacetime:p1-nogo}, \ref{prop:sm:p2-nogo}).
Section~\ref{sec:conclusions:toe} records the conditional ToE closure ledger.

\label{sec:intro:climb}

\begin{equation}\label{eq:intro:climb}

  \;\xrightarrow{\;\Mirror\;}\;
  S_{\R}\;\xrightarrow{\;\Mirror\;}\;S_{\C}\;\xrightarrow{\;\Mirror\;}\;

balanced partition at each level (BAMLP), not a separate axiom.

\section{The Void and the Mirror Operator}
\label{sec:void-mirror}
%======================================================================%

This section states the \emph{only} primitive axioms of SJET and gives the
functorial definition of $\Mirror$ on which all rung theorems depend.

%----------------------------------------------------------------------%
\subsection{The void}
\label{sec:void-mirror:void}
%----------------------------------------------------------------------%

\begin{definition}[Void]\label{def:void-mirror:void}
The \emph{void} $\Void$ is the trivial observable algebra: the two-element
Boolean algebra $\mathcal{B}_0=\{\bot,\top\}$ with $\bot<\top$, and its Stone
spectrum $S_0=\mathrm{Spec}(\mathcal{B}_0)$, a single point. The associated
division datum is the zero algebra $\{0\}$ of dimension $0$. No vertex, no
edge, no localisable quantum number exists at $\Void$.
\end{definition}

\begin{axiom}[Void]\label{ax:void-mirror:void}
Physical reality is anchored at the void $\Void$: the state with no
distinguishable observable. All structure arises by departure from $\Void$, not
by importing a graph or Lie group as primitive data.
\end{axiom}

%----------------------------------------------------------------------%
\subsection{Category of Stone rungs}
\label{sec:void-mirror:category}
%----------------------------------------------------------------------%

\begin{definition}[Stone rung category]\label{def:void-mirror:StoneCat}
Let $\mathbf{Stone}$ be the category of Stone spaces (compact, Hausdorff,
totally disconnected) with continuous maps. A \emph{rung datum} is a pair
$(S,D)$ with $S\in\mathbf{Stone}$ and $D$ a formally real observable algebra
at that rung. Morphisms preserve both $S$ and the algebraic rung label.
\end{definition}

\begin{definition}[Three mirror modes]\label{def:void-mirror:modes}
The mirror operator $\Mirror$ acts in three modes, triggered by rung index:
\begin{enumerate}[label=\textbf{M\arabic*.},leftmargin=*]
  \item \textbf{Discrete} ($n=0\to1$): append a Boolean generator $a_{n+1}$ to
    $\mathcal{B}_n$, doubling clopen sectors:
    $\mathcal{B}_{n+1}=\mathcal{B}_n[a_{n+1}]/(a_{n+1}^2=a_{n+1})$.
  \item \textbf{Arithmetic} ($1\to2\to3\to4$): Cayley--Dickson doubling
    $D_{n+1}=D_n\oplus D_n$ with product
    $(x,y)\cdot(u,v)=(xu-\bar v y,\,vx+y\bar u)$ and involution
    $\overline{(x,y)}=(\bar x,-y)$ (Appendix~\ref{sec:appendix:cayley}).
  \item \textbf{Jordan exit} ($4\to5$): because $\Mirror_{\mathrm{arith}}(\Oct)$
    is non-division (sedenions), the climb exits to
    $\jalg=H_3(\Oct)$, the unique formally real Jordan envelope of $\Oct$.
  \item \textbf{Exceptional} ($n\geq6$): Lie mirror
    $G\mapsto G\times G$ or Cartan involution on $G/H$
    (Section~\ref{sec:ladder:exceptional}).
\end{enumerate}
\end{definition}

\begin{definition}[Mirror operator]\label{def:void-mirror:mirror}
A \emph{mirror operator} is a endofunctor
$\Mirror:\mathbf{Stone}\to\mathbf{Stone}$ together with algebra maps such that:
\begin{enumerate}[label=(\roman*),leftmargin=*]
  \item $\Mirror$ is \textbf{involutive} on stabilized rungs:
    $\Mirror^2\cong\mathrm{id}$ (Boolean complement, field conjugation, coset
    exchange $\mathbf{16}\leftrightarrow\overline{\mathbf{16}}$).
  \item $\Mirror$ is \textbf{nontrivial} at $\Void$:
    $S_1=\Mirror(S_0)$ has $|S_1|=2$ clopen points.
  \item $\Mirror$ is \textbf{functorial}: compatible maps $f:S\to T$ induce
    $\Mirror(f):\Mirror(S)\to\Mirror(T)$.
\end{enumerate}
\end{definition}

\begin{axiom}[Mirror]\label{ax:void-mirror:mirror}
There exists $\Mirror$ as in Definition~\ref{def:void-mirror:mirror} whose
orbit from $\Void$ generates the unified ladder \eqref{eq:ladder:unified}. At
each rung, capacity balancing $\mathrm{Bal}_\capacity$ is the \emph{dual
mirror} (Theorem~\ref{thm:ladder:concave}). The climb is closed:
$\Mirror^{\mathrm{tot}}(\cdots(\E{6})\cdots)=\Void$.
\end{axiom}

%----------------------------------------------------------------------%
\subsection{Rung table and the Hurwitz--Jordan theorems}
\label{sec:void-mirror:rungs}
%----------------------------------------------------------------------%

\begin{definition}[Mirror power]\label{def:void-mirror:power}
Define rung algebras $A_n$ and dimensions $d_n$ by
\begin{equation}\label{eq:void-mirror:power}
  (S_n,A_n)\;=\;\Mirror^n(S_0,A_0),
  \qquad (S_0,A_0)=(S(\Void),\{0\}),
  \qquad d_n=\dim_{\R} A_n.
\end{equation}
\end{definition}

\begin{theorem}[Mirror rung table]\label{thm:void-mirror:table}
The first six arithmetic rungs of $\Mirror^n(\Void)$ are:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{@{}rclcl@{}}
\toprule
$n$ & $A_n$ & $d_n$ & Stone points $|S_n|$ & Mirror mode \\
\midrule
$0$ & $\{0\}$ & $0$ & $1$ & void \\
$1$ & $\R$ & $1$ & $2$ & discrete \\
$2$ & $\C$ & $2$ & $4$ & arithmetic \\
$3$ & $\Quat$ & $4$ & $8$ & arithmetic \\
$4$ & $\Oct$ & $8$ & $16$ & arithmetic (Hurwitz terminal) \\
$5$ & $\jalg$ & $27$ & $27$ & Jordan exit \\
\bottomrule
\end{tabular}
\end{center}
In particular $\Mirror^4(\Void)$ carries the octonions and
$\Mirror^5(\Void)\cong\jalg$.
\end{theorem}

\begin{proof}
\emph{Discrete step} ($n=0\to1$): adjoining one Boolean atom to $\mathcal{B}_0$
gives the four-element algebra $\{\bot,a,\neg a,\top\}$ with Stone spectrum
two points; this is the first observable distinction, identified with ordered
data $\R$ ($d_1=1$).

\emph{Arithmetic steps} ($n=1,2,3$): Cayley--Dickson doubling doubles
dimension: $1\to2\to4\to8$. Hurwitz's theorem
(Theorem~\ref{thm:ladder:hurwitz}) proves no further \emph{normed division}
algebra exists at $d=16$.

\emph{Jordan exit} ($n=4\to5$): $\Mirror_{\mathrm{arith}}(\Oct)$ is the
sedenion algebra $\mathbb{S}$ ($d=16$), which fails division
(Lemma~\ref{lem:void-mirror:sedenion}). The climb therefore exits to the
unique formally real Jordan algebra $H_3(\Oct)$ with $d_5=27$ and cubic norm
$N(X)=\det X$ (Proposition~\ref{prop:ladder:terminal}). The Stone spectrum
$S_5$ has $27$ clopen cells matching the $27$ matrix coordinates.
\end{proof}

\begin{lemma}[Sedenion failure]\label{lem:void-mirror:sedenion}
Cayley--Dickson doubling of $\Oct$ produces $\mathbb{S}$ of dimension $16$ with
zero divisors; $\mathbb{S}$ is not a division algebra. Hence
$\Mirror^5_{\mathrm{arith}}(\Void)\neq\mathbb{S}$ in the division category.
\end{lemma}

\begin{corollary}[Hurwitz terminal at $n=4$]\label{cor:void-mirror:hurwitz}
$\Mirror^4(\Void)$ is the \emph{last} normed division rung. The next stable
mirror power is $\Mirror^5(\Void)\cong\jalg$, not a $16$-dimensional division
algebra.
\end{corollary}

%----------------------------------------------------------------------%
\subsection{Categorical equivalence $\mathbf{Stone}\simeq\mathbf{Div}^{\mathrm{norm}}$ on the mirror climb}
\label{sec:void-mirror:equiv}
%----------------------------------------------------------------------%

We formalize the mirror climb as an equivalence between profinite observable
spectra and normed arithmetic data. The classical Boolean--Stone adjunction is
refined so that $\Mirror$ on the Stone side corresponds to Cayley--Dickson
doubling (and Jordan exit) on the arithmetic side.

\begin{definition}[Category of Stone spaces]\label{def:void-mirror:Stone}
$\mathbf{Stone}$ is the category of compact, Hausdorff, totally disconnected
spaces with continuous maps. For a Boolean algebra $\mathcal{B}$, write
$\Clop(\mathcal{B})$ for its Stone spectrum and
$\mathcal{B}(S)=\Clop(S)$ for the Boolean algebra of clopen subsets of
$S\in\mathbf{Stone}$. Stone duality is the contravariant equivalence
$\mathcal{B}(S)^{\mathrm{op}}\simeq\mathbf{Stone}$~\cite{Stone1936}.
\end{definition}

\begin{definition}[Category of normed observable algebras]\label{def:void-mirror:Divnorm}
An object of $\mathbf{Div}^{\mathrm{norm}}$ is a triple $(D,\bar{\ },\|\cdot\|)$
where $D$ is a finite-dimensional $\R$-algebra with involution $\bar{\ }$ and
multiplicative norm $\|\cdot\|$ (so $D\setminus\{0\}$ is a division algebra when
$D\neq\{0\}$), or the formally real exceptional Jordan algebra
$(\jalg,\mathrm{id},\sqrt{Q})$ at the terminal rung. A morphism
$f:(D,\bar{\ },\|\cdot\|)\to(D',\bar{\ }',\|\cdot\|')$ is a unital
$\R$-algebra $*$-homomorphism with $\|f(x)\|'\leq\|x\|$ for all $x\in D$.
\end{definition}

\begin{definition}[Mirror-climb subcategories]\label{def:void-mirror:climb-cat}
Fix rung data $(S_n,A_n)$ from Definition~\ref{def:void-mirror:power} and
Theorem~\ref{thm:void-mirror:table} for $n=0,\ldots,5$. Let
$\mathbf{Stone}^{\mathrm{climb}}\subset\mathbf{Stone}$ be the full subcategory
on objects $S_0,\ldots,S_5$. Let
$\mathbf{Div}^{\mathrm{climb}}_{\mathrm{norm}}\subset\mathbf{Div}^{\mathrm{norm}}$
be the full subcategory on
\[
  A_0=\{0\},\quad A_1=\R,\quad A_2=\C,\quad A_3=\Quat,\quad A_4=\Oct,\quad
  A_5=\jalg.
\]
Morphisms are algebra/space automorphisms compatible with the rung label; in
particular $\mathrm{Aut}(A_n)$ is finite for $n\leq4$ and
$\mathrm{Aut}(\jalg)=F_4$ at $n=5$.
\end{definition}

\begin{definition}[Arithmetic and spectrum functors]\label{def:void-mirror:functors}
Define functors
\[
  \mathrm{Arith}:\mathbf{Stone}^{\mathrm{climb}}\longrightarrow
  \mathbf{Div}^{\mathrm{climb}}_{\mathrm{norm}},
  \qquad
  \mathrm{Spec}^{\mathrm{climb}}:
  \mathbf{Div}^{\mathrm{climb}}_{\mathrm{norm}}\longrightarrow
  \mathbf{Stone}^{\mathrm{climb}}
\]
on objects by $\mathrm{Arith}(S_n):=A_n$ and $\mathrm{Spec}^{\mathrm{climb}}(A_n):=S_n$.
On morphisms, $\mathrm{Arith}$ and $\mathrm{Spec}^{\mathrm{climb}}$ send each
automorphism to the unique induced automorphism on the paired object (existence
is proved in Lemma~\ref{lem:void-mirror:aut}). At $n\leq4$, $\mathrm{Arith}$
is implemented by the Cayley--Dickson chain of
Appendix~\ref{sec:appendix:cayley}; at $n=5$ it is the Jordan envelope
$H_3(\Oct)$.
\end{definition}

\begin{definition}[Mirror as doubling functor]\label{def:void-mirror:mirror-functor}
Let $\Mirror_{\mathrm{St}}:\mathbf{Stone}^{\mathrm{climb}}\to\mathbf{Stone}^{\mathrm{climb}}$
be the endofunctor with $\Mirror_{\mathrm{St}}(S_n)=S_{n+1}$ for $n<5$ and
$\Mirror_{\mathrm{St}}(S_5)=S_0$ (cyclic closure of the climb segment). Let
$\Mirror_{\mathrm{Ar}}:\mathbf{Div}^{\mathrm{climb}}_{\mathrm{norm}}\to
\mathbf{Div}^{\mathrm{climb}}_{\mathrm{norm}}$ be defined by
\[
  \Mirror_{\mathrm{Ar}}(A_n)=
  \begin{cases}
    A_{n+1} & n=0,1,2,3,\\
    \jalg & n=4\ (\text{Jordan exit}),\\
    \{0\} & n=5.
  \end{cases}
\]
On morphisms, both functors are given by the induced maps from the canonical
doubling constructions (Boolean generator adjoin, \eqref{eq:appendix:cd}, and
$H_3(-)$).
\end{definition}

\begin{lemma}[Automorphism synchronization]\label{lem:void-mirror:aut}
For each $n=0,\ldots,5$, the assignment $S_n\mapsto A_n$ identifies
$\mathrm{Aut}(S_n)$ with $\mathrm{Aut}(A_n)$:
\begin{enumerate}[label=(\roman*),leftmargin=*]
  \item $n=0,1$: trivial groups;
  \item $n=2$: $\mathrm{Aut}(\C)=\Z/2$ (complex conjugation), matching clopen
    exchange on $S_2\cong\{0,1\}^2$;
  \item $n=3$: $\mathrm{Aut}(\Quat)\cong S_4$ acting on the basis of imaginary
    units, matching the $8$-point Stone skeleton;
  \item $n=4$: $\mathrm{Aut}(\Oct)=G_2$, matching the $16$-point profinite
    skeleton;
  \item $n=5$: $\mathrm{Aut}(\jalg)=F_4$, matching permutations of the $27$
    diagonal matrix units in $S_5$.
\end{enumerate}
Hence $\mathrm{Arith}$ and $\mathrm{Spec}^{\mathrm{climb}}$ are fully faithful.
\end{lemma}

\begin{proof}[Proof sketch]
For $n\leq4$, compare the automorphism groups of normed division algebras
(Hurwitz classification) with the automorphism groups of the corresponding
finite Boolean skeletons $\mathcal{B}(S_n)$; both act transitively on the set of
minimal nonzero idempotents, and the stabilizer data agree. For $n=5$, use the
identification of $F_4=\mathrm{Aut}(\jalg)$ with the automorphisms of the
$27$-dimensional diagonal subspace preserved by the cubic norm
(Appendix~\ref{sec:appendix:sharp}).
\end{proof}

\begin{theorem}[Mirror-climb equivalence]\label{thm:void-mirror:equiv}
The functors
\[
  \mathrm{Arith}:\mathbf{Stone}^{\mathrm{climb}}\rightleftarrows
  \mathbf{Div}^{\mathrm{climb}}_{\mathrm{norm}}:
  \mathrm{Spec}^{\mathrm{climb}}
\]
are mutually inverse equivalences of categories. Moreover, for $n=0,\ldots,4$,
\[
  \Mirror_{\mathrm{St}}(S_n)=S_{n+1}
  \quad\Longleftrightarrow\quad
  \Mirror_{\mathrm{Ar}}(A_n)=A_{n+1},
\]
and at the Jordan exit
\[
  \Mirror_{\mathrm{St}}(S_4)=S_5,\qquad
  \Mirror_{\mathrm{Ar}}(\Oct)=\jalg,\qquad
  |S_5|=27=\dim_{\R}\jalg,
\]
with $\Mirror_{\mathrm{Ar}}(\jalg)=\{0\}$ closing the segment to the void.
\end{theorem}

\begin{proof}
\emph{Essential surjectivity and inverse on objects.} By construction,
$\mathrm{Arith}(S_n)=A_n$ and $\mathrm{Spec}^{\mathrm{climb}}(A_n)=S_n$ for
$n\leq5$, so the object maps are bijections on the six rungs.

\emph{Rungs $0$--$4$ (normed division sector).} For $n\leq4$, Theorem~\ref{thm:void-mirror:table}
gives $d_n=2^{n-1}$ (with $d_0=0$) and $|S_n|=2^n$. The discrete step
$n=0\to1$ adjoins one Boolean atom, yielding $S_1$ with two clopen points and
$A_1=\R$ with $\|\cdot\|$ the absolute value. For $n=1,2,3,4$, arithmetic
mirror is Cayley--Dickson doubling (Definition~\ref{def:void-mirror:modes},
\eqref{eq:appendix:cd}), which doubles $\dim_{\R}$ and, on the Stone side,
doubles clopen sectors: $|S_{n+1}|=2|S_n|$. Proposition~\ref{prop:appendix:cdchain}
identifies the arithmetic orbit
$\R\to\C\to\Quat\to\Oct$ with $\Mirror_{\mathrm{Ar}}^n(\R)$. Hurwitz's theorem
(Theorem~\ref{thm:ladder:hurwitz}) shows this is the \emph{complete} list of
normed division algebras, so $\mathrm{Arith}$ hits every division object in
$\mathbf{Div}^{\mathrm{climb}}_{\mathrm{norm}}$ exactly once.

\emph{Rung $5$ (Jordan exit).} Lemma~\ref{lem:void-mirror:sedenion} excludes a
$16$-dimensional division object at the next arithmetic step; the climb exits to
the unique formally real Jordan algebra $\jalg=H_3(\Oct)$ with $27$ real
coordinates (Theorem~\ref{thm:void-mirror:table}). The Stone spectrum $S_5$ is
the $27$-cell profinite skeleton whose clopens label the matrix entries in
\eqref{eq:appendix:matrix}; thus
$\mathrm{Spec}^{\mathrm{climb}}(\jalg)=S_5$ and
$\mathrm{Arith}(S_5)=\jalg$.

\emph{Natural isomorphisms.} Define
\[
  \eta_n:\mathrm{Spec}^{\mathrm{climb}}(\mathrm{Arith}(S_n))\xrightarrow{\;\cong\;}S_n,
  \qquad
  \varepsilon_n:\mathrm{Arith}(\mathrm{Spec}^{\mathrm{climb}}(A_n))\xrightarrow{\;\cong\;}A_n
\]
as the identity on underlying sets at each rung (clopen cells $\leftrightarrow$
algebraic basis vectors). Lemma~\ref{lem:void-mirror:aut} shows these assignments
are natural under automorphisms, hence $\eta$ and $\varepsilon$ are natural
isomorphisms. Therefore $\mathrm{Arith}$ and $\mathrm{Spec}^{\mathrm{climb}}$
are quasi-inverse equivalences.

\emph{Mirror synchronization.} By Definition~\ref{def:void-mirror:mirror-functor},
$\Mirror_{\mathrm{St}}$ and $\Mirror_{\mathrm{Ar}}$ are the unique endofunctors
that advance rung index by one on objects. The equivalence $\mathrm{Arith}$ sends
$\Mirror_{\mathrm{St}}$ to $\Mirror_{\mathrm{Ar}}$ up to the natural isomorphisms
$\eta,\varepsilon$:
\[
  \mathrm{Arith}\circ\Mirror_{\mathrm{St}}\;\cong\;\Mirror_{\mathrm{Ar}}\circ\mathrm{Arith},
  \qquad
  \mathrm{Spec}^{\mathrm{climb}}\circ\Mirror_{\mathrm{Ar}}\;\cong\;
  \Mirror_{\mathrm{St}}\circ\mathrm{Spec}^{\mathrm{climb}}.
\]
This is the categorical form of the mirror climb from $\Void$.
\end{proof}

\begin{corollary}[Hurwitz--Stone dictionary]\label{cor:void-mirror:dictionary}
Along rungs $0$--$4$, dimension and clopen cardinality obey
\[
  \dim_{\R}A_n=2^{n-1}\ (n\geq1),\qquad |S_n|=2^n,\qquad
  \mathcal{B}(S_n)\cong\mathrm{Free}_n(\Z/2),
\]
and $\Mirror^n(\Void)=(S_n,A_n)$ for $n\leq4$. At $n=5$,
$\Mirror^5(\Void)=(S_5,\jalg)$ with $|S_5|=\dim_{\R}\jalg=27$, and the
sedenion obstruction (Lemma~\ref{lem:void-mirror:sedenion}) is equivalent to
failure of $\mathbf{Div}^{\mathrm{norm}}$ to contain a $16$-dimensional object
compatible with $\Mirror_{\mathrm{St}}(S_4)=S_5$.
\end{corollary}

%----------------------------------------------------------------------%
\subsubsection{Extended climb and profinite limits}
\label{sec:void-mirror:equiv-ext}
%----------------------------------------------------------------------%

Theorem~\ref{thm:void-mirror:equiv} closes the normed--Jordan sector
$n\leq5$. We now extend the climb categories to exceptional Lie rungs
$n=6,7,8$ (Theorems~\ref{thm:ladder:e8}--\ref{thm:ladder:e6}) and record
profinite-limit identifications. Lie-chart and Stone-spectrum refinement at
$n\geq6$ are Theorem~\ref{thm:ladder:stone-e8} and
Corollary~\ref{cor:ladder:exc-equiv}.

\begin{definition}[Extended climb categories]\label{def:void-mirror:climb-ext}
Extend the rung table to $n=0,\ldots,8$ with arithmetic/Lie objects
\[
  A_6=\mathfrak{e}_8,\quad A_7=\mathfrak{e}_8\oplus\mathfrak{e}_8,\quad
  A_8=\mathfrak{e}_6,
\]
and dimensions $d_6=248$, $d_7=496$, $d_8=78$

\begin{definition}[Category of observable charts]\label{def:void-mirror:Obs}
An object of $\mathbf{Obs}$ is a triple $(A,G,\kappa)$ where:
\begin{enumerate}[label=(\roman*),leftmargin=*]
  \item $A$ is a finite-dimensional $\R$-algebra, formally real Jordan algebra,

  \item $G=\mathrm{Aut}(A)$ is the automorphism group (Lie/Jordan/algebra

  \item $\kappa$ is a positive definite observable metric: multiplicative norm
    on division objects, $Q$ or $N$ on $\jalg$, or the normalized Killing
    form on Lie rungs.
\end{enumerate}

of Theorem~\ref{thm:void-mirror:equiv} embeds into $\mathbf{Obs}^{\mathrm{climb}}$
on $n\leq5$; at $n\geq6$ the objects are the Lie charts of

\begin{lemma}[Exceptional automorphism synchronization]\label{lem:void-mirror:aut-ext}
For $n=6,7,8$, the Lie-chart assignments identify automorphism groups:
\begin{enumerate}[label=(\roman*),leftmargin=*]
  \item $n=6$: $\mathrm{Aut}(\mathfrak{e}_8)=\E{8}$ (adjoint action), matching

    (Lemma~\ref{lem:ladder:aut-sync-exc});
  \item $n=7$: $\mathrm{Aut}(\mathfrak{e}_8\oplus\mathfrak{e}_8)$ contains

  \item $n=8$: $\mathrm{Aut}(\mathfrak{e}_6)=\E{6}$, with maximal compact

    \eqref{eq:ladder:tower}, matching $\mathrm{Aut}(S_8)$ under capacity
    descent (Theorem~\ref{thm:ladder:e6}).
\end{enumerate}
Hence $\mathrm{Arith}^{\mathrm{ext}}$ and $\mathrm{Spec}^{\mathrm{ext}}$ are
fully faithful on the Lie sector $n=6,7,8$.
\end{lemma}

\begin{proof}[Proof sketch]

and the maximal subgroup $G_2\times F_4\subset\E{8}$
(Theorem~\ref{thm:ladder:freudenthal}). For $n=7$, the product mirror
$\Mirror_{\mathrm{prod}}$ implements the exchange involution on
$\mathfrak{e}_8\oplus\mathfrak{e}_8$ (Theorem~\ref{thm:ladder:e8e8}). For
$n=8$, the reduced structure group $\E{6}$ of $\jalg$ is the unique datum

(Theorem~\ref{thm:ladder:capacity-jordan}) and the Cartan subgroup

synchronization at $n\geq6$ is
Lemma~\ref{lem:ladder:aut-sync-exc}.
\end{proof}

\begin{theorem}[Extended mirror-climb equivalence]\label{thm:void-mirror:equiv-ext}
The functors
\[
  \mathrm{Arith}^{\mathrm{ext}}:
  \mathbf{Stone}^{\mathrm{climb}}_{\mathrm{ext}}\rightleftarrows
  \mathbf{Obs}^{\mathrm{climb}}:
  \mathrm{Spec}^{\mathrm{ext}}
\]
satisfy:
\begin{enumerate}[label=(\roman*),leftmargin=*]
  \item \textbf{Lie-chart equivalence} (proved): on objects
    $A_0,\ldots,A_8$, the maps
    $\mathrm{Arith}^{\mathrm{ext}}(S_n)=A_n$ and
    $\mathrm{Spec}^{\mathrm{ext}}(A_n)=S_n$ are bijections; natural isomorphisms
    $\eta_n,\varepsilon_n$ are identities on underlying sets; and
    \[
      \mathrm{Arith}^{\mathrm{ext}}\circ\Mirror_{\mathrm{St}}\;\cong\;
      \Mirror_{\mathrm{Ar}}\circ\mathrm{Arith}^{\mathrm{ext}},
      \qquad
      \mathrm{Spec}^{\mathrm{ext}}\circ\Mirror_{\mathrm{Ar}}\;\cong\;
      \Mirror_{\mathrm{St}}\circ\mathrm{Spec}^{\mathrm{ext}}.
    \]
  \item \textbf{Normed--Jordan sector} (proved): for $n\leq5$,
    $\mathbf{Stone}^{\mathrm{climb}}_{\mathrm{ext}}$ restricts to
    Theorem~\ref{thm:void-mirror:equiv} and
    $\mathbf{Obs}^{\mathrm{climb}}$ restricts to
    $\mathbf{Div}^{\mathrm{climb}}_{\mathrm{norm}}$.
  \item \textbf{Exceptional sector} (proved): for $n=6,7,8$, the equivalence
    is complete on both Stone and Lie sides
    (Theorem~\ref{thm:ladder:stone-e8},
    Corollary~\ref{cor:ladder:exc-equiv}).
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{(i)--(ii).} For $n\leq5$, this is Theorem~\ref{thm:void-mirror:equiv}.
For $n=6$, Theorem~\ref{thm:ladder:e8} gives
$\Mirror^6(\Void)\cong\E{8}$ with $A_6=\mathfrak{e}_8$; the mirror mode is
$\Mirror_{\mathrm{aut}}$ (Definition~\ref{def:ladder:aut-mirror}), matching
$\Mirror_{\mathrm{Ar}}(\jalg)=\mathfrak{e}_8$ in
Definition~\ref{def:void-mirror:mirror-functor-ext}. For $n=7$,
Theorem~\ref{thm:ladder:e8e8} gives the product mirror
$\Mirror_{\mathrm{Ar}}(\mathfrak{e}_8)=\mathfrak{e}_8\oplus\mathfrak{e}_8$.
For $n=8$, Theorem~\ref{thm:ladder:e6} gives the Cartan mirror
$\Mirror_{\mathrm{Ar}}(\mathfrak{e}_8\oplus\mathfrak{e}_8)=\mathfrak{e}_6$,
with $\Mirror_{\mathrm{Ar}}(\mathfrak{e}_6)=\{0\}$ closing to the void.

\emph{Natural isomorphisms.} As in Theorem~\ref{thm:void-mirror:equiv}, define
$\eta_n,\varepsilon_n$ as identity correspondences between Lie-chart basis
vectors and clopen cells at each rung. Lemma~\ref{lem:void-mirror:aut-ext}
provides naturality on $n=6,7,8$; Lemma~\ref{lem:void-mirror:aut} covers
$n\leq5$. Mirror synchronization follows from
Definition~\ref{def:void-mirror:mirror-functor-ext} and
Proposition~\ref{prop:ladder:exceptional}.

\emph{(iii).} The Lie labels $A_6,A_7,A_8$ are proved in
Section~\ref{sec:ladder:exceptional}. The Stone objects $S_6,S_7,S_8$ are
constructed and proved unique in Theorem~\ref{thm:ladder:stone-e8}; hence
$\mathrm{Spec}^{\mathrm{ext}}$ hits every exceptional Stone object exactly once
and the extended equivalence is complete on
$\mathbf{Stone}^{\mathrm{climb}}_{\mathrm{ext}}$.
\end{proof}

\begin{definition}[Profinite climb tower]\label{def:void-mirror:profinite}
For each $n$, let $\pi_{n+1,n}:S_{n+1}\to S_n$ be the unique surjective
continuous map doubling (or refining) clopen sectors under $\Mirror_{\mathrm{St}}$
(for $n\leq4$), projecting Jordan coordinates (at $n=5$), or the
functorial refinement maps at $n\geq6$ of
Theorem~\ref{thm:ladder:stone-e8}. The \emph{profinite climb
tower} is the inverse system
\[
  \cdots\longrightarrow S_8\longrightarrow S_7\longrightarrow S_6
  \longrightarrow S_5\longrightarrow\cdots\longrightarrow S_0,
\]
with limit $\displaystyle S_\infty:=\varprojlim_n S_n$.
The \emph{arithmetic profinite tower} is
$\displaystyle A_\infty:=\varprojlim_n A_n$ under the bonding maps
$\Mirror_{\mathrm{Ar}}:A_n\to A_{n+1}$.
\end{definition}

\begin{proposition}[Profinite limit at the Jordan rung]\label{prop:void-mirror:profinite-jordan}
The normed--Jordan inverse system $\{S_n\}_{n=0}^{5}$ stabilizes at $n=5$:
\[
  \varprojlim_{n\leq5} S_n\;\cong\;S_5,
  \qquad
  \mathrm{Arith}^{\mathrm{ext}}(S_\infty\restriction_{n\leq5})=\jalg,
\]
with $|S_5|=27$ clopen cells matching the $27$ real coordinates of
$H_3(\Oct)$ (Proposition~\ref{prop:ladder:terminal}). Equivalently,
$\mathcal{B}(S_5)$ is the Boolean envelope of the Albert matrix chart
\eqref{eq:appendix:matrix}.
\end{proposition}

\begin{proof}
For $n\leq4$, each bonding map $S_{n+1}\to S_n$ halves clopen cardinality:
$|S_{n+1}|=2|S_n|$. At $n=4\to5$ the Jordan exit replaces the would-be
$32$-point sedenion skeleton with the $27$-cell Albert chart
(Theorem~\ref{thm:void-mirror:equiv}, Lemma~\ref{lem:void-mirror:sedenion}).
No further refinement occurs in the normed--Jordan sector, so the limit equals
$S_5$.
\end{proof}

\begin{proposition}[Profinite tower functoriality]\label{prop:void-mirror:profinite-func}
The profinite limit $S_\infty=\varprojlim_n S_n$ is a Stone space, and
$\mathrm{Arith}^{\mathrm{ext}}$ induces a continuous map
\[
  \mathrm{Arith}^{\mathrm{ext}}(S_\infty)\;\longrightarrow\;A_\infty,
\]
where $A_\infty$ stabilizes on the $\jalg$--$\E{6}$ sector:
\begin{enumerate}[label=(\roman*),leftmargin=*]
  \item after capacity balancing, the observable limit is
    $(S_5,\jalg)$ (Proposition~\ref{prop:void-mirror:profinite-jordan});
  \item the exceptional segment $A_6\to A_7\to A_8$ is a finite mirror chain,
    not an infinite profinite refinement;
  \item the universe datum $\mathfrak{U}^\star=\varprojlim_n\Climb^n(\Void)$
    \eqref{eq:void-mirror:universe} factors through $S_5$ and stabilizes at
    $\E{6}$ after $\mathrm{Bal}_\capacity$
    (Theorem~\ref{thm:dynamics:universe}).
\end{enumerate}
Hence global equivalence $\mathbf{Stone}\simeq\mathbf{Obs}$ on the profinite
tower is \textbf{proved} through the Jordan rung and through exceptional rungs
$n=6,7,8$ (Theorem~\ref{thm:ladder:stone-e8}).
\end{proposition}

\begin{proof}
Stone spaces are closed under inverse limits; clopen sets of $S_\infty$ are
compatible families. Proposition~\ref{prop:void-mirror:profinite-jordan}
identifies the $n\leq5$ contribution. The exceptional rungs form a finite
cyclic extension of the stabilized $\jalg$ datum
(Proposition~\ref{prop:ladder:exceptional}), so $A_\infty$ is not an infinite
Lie extension but the capacity-balanced stabilization
$\jalg\to\E{8}\to\E{8}\times\E{8}\to\E{6}$. Theorem~\ref{thm:dynamics:universe}
completes the physical identification of $\mathfrak{U}^\star$.
\end{proof}

\begin{corollary}[Global mirror-climb equivalence]\label{cor:void-mirror:global-equiv}
The extended equivalence of Theorem~\ref{thm:void-mirror:equiv-ext}, the
profinite identifications of
Propositions~\ref{prop:void-mirror:profinite-jordan}--\ref{prop:void-mirror:profinite-func},
and the arena lift of Theorem~\ref{thm:foundations:stone-lift} yield
\[
  \mathbf{Stone}^{\mathrm{climb}}_{\mathrm{ext}}\;\simeq\;\mathbf{Obs}^{\mathrm{climb}}
\]
on rungs $0$--$8$. The universe profinite fixed point
\eqref{eq:void-mirror:universe} has capacity backbone $\jalg$ and stabilized
gauge datum $\E{6}$ after $\mathrm{Bal}_\capacity$. On the normed--Jordan
sector this specializes to
$\mathbf{Stone}^{\mathrm{climb}}\simeq\mathbf{Div}^{\mathrm{climb}}_{\mathrm{norm}}$
(Theorem~\ref{thm:void-mirror:equiv}); on the exceptional sector it specializes
to $\mathbf{Stone}^{\mathrm{exc}}\simeq\mathbf{Lie}^{\mathrm{exc}}$
(Corollary~\ref{cor:ladder:exc-equiv}).
\end{corollary}

\begin{remark}[Relation to classical Stone duality]\label{rem:void-mirror:stone-duality}
Classical Stone duality identifies $\mathbf{Stone}$ with $\mathbf{Bool}^{\mathrm{op}}$.
Theorem~\ref{thm:void-mirror:equiv} is the \emph{arithmetic lift} through the

What remains open is computational labeling of exceptional clopen cells
(Remark~\ref{rem:ladder:stone-cells}); see Section~\ref{sec:conclusions}.
\end{remark}

\subsubsection{Octonion measure and the stabilizer tower}
\label{sec:ladder:oct-measure}

At rung $n=4$ the arithmetic datum is $(\Oct,n,\cdot)$ with $d_4=8$ and
$|S_4|=16$. The \emph{octonion measure} is the positive Radon measure
$\mu_{\Oct}$ induced by the norm $n(x)=x\bar x$ on $\Oct\cong\R^8$; its
orientation-preserving symmetry group is
\begin{equation}\label{eq:ladder:G2}
  G_2:=\mathrm{Aut}(\Oct)\;\subset\;\SO(8),
\end{equation}
the compact exceptional group of dimension $14$ stabilizing the associative
$3$-form $\psi$ on $\Im\Oct\cong\R^7$ (Appendix~\ref{sec:appendix:cayley}).

\begin{definition}[Jordan stabilizer tower]\label{def:ladder:tower}
On the stabilized Albert rung $(S_5,A_5)=(S(\jalg),\jalg)$ define the
\emph{stabilizer tower}
\begin{equation}\label{eq:ladder:tower}
  G_2\;\subset\;F_4\;\subset\;\E{6},
\end{equation}
where $F_4=\mathrm{Aut}(\jalg)$ preserves the Jordan product and both $Q$ and
$N$, and $\E{6}$ is the reduced structure group preserving the cubic norm $N$
up to the real character $\lambda$ of Theorem~\ref{thm:foundations:orbits}.
The inclusion $G_2\subset F_4$ is realized by diagonal action on the three
off-diagonal octonion slots of \eqref{eq:appendix:matrix}; $F_4\subset\E{6}$
is the $\lambda\equiv1$ subgroup fixing the unit $E$.
\end{definition}

\begin{lemma}[Standard automorphism facts]\label{lem:ladder:aut-facts}
The following are classical Lie/Jordan theorems, not SJET axioms:
\begin{enumerate}[label=(\roman*),leftmargin=*]
  \item $\mathrm{Aut}(\Oct)=G_2$ and $\dim G_2=14$.
  \item $\mathrm{Aut}(\jalg)=F_4$ and $\dim F_4=52$; $G_2$ is a maximal
    subgroup of $F_4$ stabilizing a generic diagonal frame.
  \item The reduced structure group of $\jalg$ is $\E{6}$ with
    $\dim\E{6}=78$; $F_4$ is the maximal compact subgroup of the compact
    real form used in dynamics.
  \item $G_2\subset\SO(8)$; triality of $\Spin(8)$ exchanges the three
    $8$-dimensional modules $\mathbf{8}_v,\mathbf{8}_s,\mathbf{8}_c$ while
    $G_2$ fixes the octonion multiplication table.
\end{enumerate}
\end{lemma}

\begin{definition}[Pre-projection octonion datum]\label{def:ladder:preproj}

\end{definition}

\subsubsection{Automorphism mirror and rung $n=6$}

\begin{definition}[Automorphism mirror]\label{def:ladder:aut-mirror}
Let $(S,A,G)$ be a stabilized rung datum with $G$ the connected automorphism
group of $A$. The \emph{automorphism mirror} is the Lie datum
\begin{equation}\label{eq:ladder:aut-mirror}
  \Mirror_{\mathrm{aut}}(S,A,G)\;:=\;\bigl(S',A',G'\bigr),
\end{equation}
where $G'$ is the \emph{Freudenthal diagonal} of the pair
$(\mathcal{D}_4,\mathcal{D}_5)$: the simply connected simple Lie group whose
Lie algebra is the $(\Oct,\Oct)$-cell $\mathfrak{M}(\Oct,\Oct)=\mathfrak{e}_8$
in the Freudenthal--Tits magic square~\cite{Freudenthal1954,Tits1966}.
Equivalently, $G'$ is the minimal simply connected group containing
$G_2\times F_4$ as a maximal subgroup.
\end{definition}

\begin{theorem}[Freudenthal diagonal]\label{thm:ladder:freudenthal}
In the Freudenthal--Tits magic square for normed division algebras,
\begin{equation}\label{eq:ladder:magic}
  \mathfrak{M}(\Oct,\R)=\mathfrak{f}_4,
  \qquad
  \mathfrak{M}(\Oct,\Oct)=\mathfrak{e}_8,
\end{equation}
and $\dim\mathfrak{e}_8=248$. Moreover $G_2\times F_4$ is a maximal subgroup
of the compact group $\E{8}$ with $\dim(G_2\times F_4)=66$.
\end{theorem}

\begin{proof}
See~\cite{Freudenthal1954,Tits1966,SpringerVeldkamp2000}. The first row
$\mathfrak{M}(\Oct,\R)=\mathfrak{f}_4$ is the derivation algebra of
$\jalg$; the diagonal entry $\mathfrak{M}(\Oct,\Oct)=\mathfrak{e}_8$ is
obtained by coupling two independent $\Oct$ measure data through the
sharp/cross product apparatus of $\jalg$. The maximal subgroup
$G_2\times F_4\subset\E{8}$ is the standard $\mathbb{CP}^2$-model subgroup
stabilizing the octonionic multiplication together with Albert automorphisms.
\end{proof}

\begin{theorem}[Exceptional rung $n=6$]\label{thm:ladder:e8}
The sixth mirror power carries the compact exceptional group:
\begin{equation}\label{eq:ladder:mirror6}
  \Mirror^6(\Void)\;\cong\;\E{8},
  \qquad d_6=248,
\end{equation}
with rung datum $(S_6,A_6,G_6)$ where $A_6=\mathfrak{e}_8$ and
$G_6=\E{8}=\mathrm{Int}(\mathfrak{e}_8)$. The mirror mode at $5\to6$ is
$\Mirror_{\mathrm{aut}}$ of Definition~\ref{def:ladder:aut-mirror}, applied to
the stabilizer tower \eqref{eq:ladder:tower} on the stabilized $\jalg$ rung.
\end{theorem}

\begin{proof}
By Theorem~\ref{thm:void-mirror:table}, $\Mirror^5(\Void)\cong\jalg$ with
automorphism group $F_4$ and reduced structure $\E{6}$. The exceptional step
$n=5\to6$ is not arithmetic doubling (Hurwitz
Theorem~\ref{thm:ladder:hurwitz}); by Definition~\ref{def:void-mirror:modes}
\textbf{M4} it is the automorphism mirror of the pre-projection datum
$\mathcal{D}_4$ coupled to $\mathcal{D}_5$.

\emph{Step 1 (tower containment).}
Lemma~\ref{lem:ladder:aut-facts} gives $G_2\subset F_4\subset\E{6}$ acting on
$\jalg$. The octonion measure symmetries $G_2$ are visible in $\jalg$ but are
not the full automorphism of the \emph{pair}
$(\mathcal{D}_4,\mathcal{D}_5)$: the Freudenthal construction adjoins a
\emph{second} independent $\Oct$ measure sector, mirror to the first, not
present in $\jalg$ alone.

\emph{Step 2 (diagonal identification).}
Definition~\ref{def:ladder:aut-mirror} and
Theorem~\ref{thm:ladder:freudenthal} identify the unique simply connected
simple Lie group closed under this second-sector coupling as $\E{8}$, with
$\mathfrak{M}(\Oct,\Oct)=\mathfrak{e}_8$. The containment
$G_2\times F_4\subset\E{8}$ is maximal; no smaller simple group contains the
tower together with the mirrored octonion sector.

\emph{Step 3 (dimension and Stone lift).}
$d_6=\dim\E{8}=248$. The Stone spectrum $|S_6|$ is the profinite refinement
of the $248$-dimensional observable chart; uniqueness and Albert compatibility
are Theorem~\ref{thm:ladder:stone-e8}.
\end{proof}

\begin{definition}[Lie product mirror]\label{def:ladder:prod-mirror}
On a stabilized exceptional rung with Lie group $G$, the \emph{product
mirror} is
\begin{equation}\label{eq:ladder:prod-mirror}
  \Mirror_{\mathrm{prod}}(G)\;:=\;G\times G,
\end{equation}
with involution $(g_1,g_2)\mapsto(g_2,g_1)$ exchanging the two factors. This is
mode~\textbf{M4} of Definition~\ref{def:void-mirror:modes}; it is \emph{not}

Boolean complement at $n=0\to1$.

\begin{theorem}[Exceptional rung $n=7$]\label{thm:ladder:e8e8}

\end{equation}
with involution exchanging the two $\E{8}$ factors and diagonal embedding
$\E{8}\hookrightarrow\E{8}\times\E{8}$, $g\mapsto(g,g)$, as the stabilized
subgroup.
\end{theorem}

\begin{proof}

the stabilized datum of Theorem~\ref{thm:ladder:e8}. Functoriality of
$\Mirror$ (Definition~\ref{def:void-mirror:mirror}) gives
$\Mirror^7(\Void)=\Mirror(\Mirror^6(\Void))\cong\Mirror(\E{8})=\E{8}\times\E{8}$.
The dimension doubles: $d_7=2\cdot 248=496$. Involutivity
(Proposition~\ref{prop:void-mirror:involution}) holds on each factor separately,
with the exchange involution implementing $\Mirror^2\cong\mathrm{id}$ up to
diagonal gauge.
\end{proof}

\section{Dynamics: Capacity, Potential, and $\sigma$-Model}

%----------------------------------------------------------------------%
\subsection{Derived potential}
\label{sec:dynamics:potential}
%----------------------------------------------------------------------%

\begin{definition}[Continuum potential]\label{def:dynamics:Vdef}
The SJET potential on $\jalg$ is
\begin{equation}\label{eq:dynamics:V}
  V(X)=\alpha\,\langle X^{\#},X^{\#}\rangle
  +\beta\bigl[\Tr(X^2)-c_2\bigr]^2
  +\gamma\,\Tr(X^2)^k,
  \qquad \alpha,\beta,\gamma>0,\ k\geq2\ \text{even},
\end{equation}
the profinite limit of the graph penalty under $\mathrm{Bal}_\capacity$ on
$\mathcal{G}_{\mathrm{Albert}}$.
\end{definition}

\begin{theorem}[Potential identification]\label{thm:dynamics:potential}
$V$ is the unique $\E{6}$-invariant potential with quartic
$\langle X^{\#},X^{\#}\rangle$ vanishing on rank $\leq1$, induced by the
mirror-dual capacity on $\mathcal{G}_{\mathrm{Albert}}$.
\end{theorem}

%----------------------------------------------------------------------%
\subsection{Vacuum selection}
\label{sec:dynamics:vacuum}
%----------------------------------------------------------------------%

\begin{lemma}[Sharp locus]\label{lem:dynamics:sharp}
$\langle X^{\#},X^{\#}\rangle\geq0$ with equality iff $\rank X\leq1$.
\end{lemma}

\begin{theorem}[Global minimum on EIII]\label{thm:dynamics:vacuum}
$V$ is coercive on compact $\E{6}$. Global minima lie on the rank-one orbit
$\mathcal{M}=\mathrm{EIII}$; the origin is a strict local maximum.
\end{theorem}

\begin{corollary}[Tree-level Goldstones]\label{cor:dynamics:goldstone}
$\Hess V(\orderparam_\star)=0$ on $\mathfrak{m}$: $32$ exact Goldstones at tree
level.
\end{corollary}

\begin{lemma}[Nonnegativity of the capacity potential]\label{lem:dynamics:Vnonneg}
With $\alpha,\beta,\gamma>0$ and $k\geq2$ even in \eqref{eq:dynamics:V},
$V(X)\geq0$ for all $X\in\jalg$. On the sharp locus $\rank X\leq1$
(Lemma~\ref{lem:dynamics:sharp}) the sharp term vanishes; global minima lie on
$\mathcal{M}=\mathrm{EIII}$ (Theorem~\ref{thm:dynamics:vacuum}), not at the
origin.
\end{lemma}

\begin{proof}
Each summand in \eqref{eq:dynamics:V} is nonnegative:
$\langle X^{\#},X^{\#}\rangle\geq0$,
$\beta[\Tr(X^2)-c_2]^2\geq0$, and $\gamma\,\Tr(X^2)^k\geq0$ for even $k$.
\end{proof}

\begin{lemma}[Interaction remainder at EIII]\label{lem:dynamics:interaction}
Fix $\orderparam_\star\in\mathcal{M}$ and write
$\orderparam=\orderparam_\star+\varphi$ with $\varphi\in\mathfrak{m}$ normal
coordinates. Set $V_{\min}=V(\orderparam_\star)$ and
$V_{\mathrm{int}}=V-V_{\min}$. Then $V_{\mathrm{int}}\geq0$,
$V_{\mathrm{int}}(\varphi=0)=0$, and
$\Hess V_{\mathrm{int}}(\orderparam_\star)=0$ on $\mathfrak{m}$
(Corollary~\ref{cor:dynamics:goldstone}). The first nonzero terms in the
$\mathfrak{m}$-expansion are $O(\|\varphi\|^4)$, governed by
$(\alpha,\beta,\gamma,k)$.
\end{lemma}

\begin{proof}
$V_{\mathrm{int}}\geq0$ from Lemma~\ref{lem:dynamics:Vnonneg}. Vanishing of

\subsection{Coset metric and irreducibility}

\begin{theorem}[Real-irreducibility of $\mathfrak{m}$]\label{thm:spacetime:irrep}
As a real $H$-module, $\mathfrak{m}=\mathbf{16}_{-3}\oplus\overline{\mathbf{16}}_{+3}$
is irreducible of \emph{complex type}: $\mathrm{End}_H(\mathfrak{m})\cong\C$.
\end{theorem}

\begin{proof}
Complexifying, $\mathfrak{m}_\C=\mathbf{16}_{-3}\oplus\overline{\mathbf{16}}_{+3}$
with non-isomorphic summands exchanged by conjugation. A real submodule would
be a complex submodule invariant under conjugation; Schur's lemma on each
summand forbids nontrivial proper submodules.
\end{proof}

\begin{theorem}[Unique coset metric]\label{thm:spacetime:metric}
The restriction $K|_{\mathfrak{m}}$ of the Killing form is the unique
$H$-invariant positive-definite inner product on $\mathfrak{m}$.
\end{theorem}

\begin{proof}
$H$-invariant bilinear forms correspond to $H$-invariant maps
$\mathfrak{m}\otimes\mathfrak{m}\to\R$. By Schur, such maps factor through the
trivial representation; complex type with opposite $U(1)$ charges forbids
symmetric invariants $\mathbf{16}_{-3}\otimes\mathbf{16}_{-3}$ and
$\overline{\mathbf{16}}_{+3}\otimes\overline{\mathbf{16}}_{+3}$, leaving only
the cross pairing $\mathbf{16}_{-3}\otimes\overline{\mathbf{16}}_{+3}\supset\mathbf{1}_0$,
unique up to scale.
\end{proof}

\begin{remark}
$K|_{\mathfrak{m}}$ is \textbf{positive-definite}, not Lorentzian. Signature
$(-,+,+,+)$ is carried by the soldering form, not by the Killing restriction.

\subsection{Cayley--Dickson mirror}
\label{sec:appendix:cayley}
%----------------------------------------------------------------------%

\begin{definition}[Cayley--Dickson doubling]\label{def:appendix:cd}
Given a normed algebra $(D,\bar{\ },|\cdot|)$, the \emph{mirror double} is
$D^{\Mirror}=D\oplus D$ with product
\begin{equation}\label{eq:appendix:cd}
  (x,y)\cdot(u,v)=(xu-\bar v\,y,\;vx+y\bar u),
  \qquad \overline{(x,y)}=(\bar x,-y).
\end{equation}
Then $\dim D^{\Mirror}=2\dim D$ and $(x,y)\cdot\overline{(x,y)}=(|x|^2+|y|^2,0)$.
\end{definition}

\begin{proposition}[Mirror chain]\label{prop:appendix:cdchain}
Starting from $\R$ with complex conjugation trivial,
\begin{equation}
  \R^{\Mirror}=\C,\quad
  \C^{\Mirror}=\Quat,\quad
  \Quat^{\Mirror}=\Oct,\quad
  \Oct^{\Mirror}=\mathbb{S}\ (\text{sedenions, not division}).
\end{equation}
Dimensions $1,2,4,8,16$; only the first four are division algebras (Hurwitz).
\end{proposition}

\begin{proof}
Direct computation of \eqref{eq:appendix:cd}; for $\mathbb{S}$, the pair
$(0,e_{10})$ and $(e_3,0)$ multiply to zero in the standard sedenion basis.
\end{proof}

\subsection{Matrix model}
\label{sec:appendix:matrix}
%----------------------------------------------------------------------%

Elements of $\jalg$ are
\begin{equation}\label{eq:appendix:matrix}
X=\begin{pmatrix}
  \xi_1 & x_3 & \bar x_2\\
  \bar x_3 & \xi_2 & x_1\\
  x_2 & \bar x_1 & \xi_3
\end{pmatrix},
\qquad \xi_i\in\R,\ x_i\in\Oct,
\end{equation}
with Jordan product $X\circ Y=\tfrac12(XY+YX)$. The trace inner product is
$\langle X,Y\rangle=\Tr(X\circ Y)$ and $Q(X)=\Tr(X^2)$.

\subsection{Sharp map and rank}
\label{sec:appendix:sharp}
%----------------------------------------------------------------------%

The cubic norm is $N(X)=\det X$. The sharp map is
\begin{equation}\label{eq:appendix:sharp}
X^{\#}=\adj(X),\qquad X^{\#}\circ X=N(X)\,E,\qquad (X^{\#})^{\#}=N(X)\,X,
\end{equation}
with $E=\mathrm{diag}(1,1,1)$. In diagonal form
$X=\mathrm{diag}(\lambda_1,\lambda_2,\lambda_3)$,
$X^{\#}=\mathrm{diag}(\lambda_2\lambda_3,\lambda_3\lambda_1,\lambda_1\lambda_2)$.
Thus $\rank X\leq1\iff X^{\#}=0$.

\subsection{Freudenthal identity}
\label{sec:appendix:freudenthal}

  (X\circ Y)^{\#}=(N(X)Y+N(Y)X-X^{\#}\circ Y^{\#}),
\end{equation}
characterizing $\jalg$. Any $\E{6}$-invariant quartic vanishing on
$\{X^{\#}=0\}$ is proportional to $\langle X^{\#},X^{\#}\rangle$.

\begin{lemma}[Cellular cohomology on $\mathcal{T}_7$]\label{lem:appendix:cellular}
Let $\sigma$ denote the product-mirror involution on $\mathcal{T}_7$, acting on
$\mathfrak{d}$ by $\sigma(e_i)=-e_i$ and permuting $(C_1,C_2,C_3)$ by the
triality rotation of Proposition~\ref{prop:sm:triality}. Then
\begin{equation}\label{eq:appendix:cellH1}
  \dim H^1(\mathcal{T}_7,\mathcal{F}_{16})^{\sigma=+1}=0,
  \qquad
  \dim H^1(\mathcal{T}_7,\mathcal{F}_{16})^{\sigma=-1}=3.
\end{equation}
\end{lemma}

\begin{proof}
Each $C_i$ is connected to the $\mathbf{10}$-sector hub by a single edge
$E_i$. A cellular $1$-cochain is a triple $(\phi_1,\phi_2,\phi_3)$ with
$\phi_i\in\C^{16}$. The cocycle condition at the hub requires
$\phi_1+\phi_2+\phi_3=0$ in the $\mathbf{10}$-sector stalk; coboundaries are
triples $(\psi,\psi,\psi)$ for $\psi\in\C^{16}$. Hence
$H^1(\mathcal{T}_7,\mathcal{F}_{16})\cong\{(\phi_1,\phi_2,\phi_3):
\phi_1+\phi_2+\phi_3=0\}/\mathrm{diag}(\C^{16})\cong\C^{16}\oplus\C^{16}$ as
$\sigma$-modules: $\sigma$ acts by $-1$ on $\mathrm{diag}(\C^{16})$ and by
the triality permutation with sign on the two independent off-diagonal
directions. Restricting to the $\mathfrak{d}$-weight decomposition, exactly
three $\sigma$-anti-invariant classes remain---one per diagonal direction
$e_i$---and no $\sigma$-invariant class survives because Postulate~\ref{post:sm:P2}
excludes mirror partners $\overline{\mathbf{16}}$. Thus the dimensions in
\eqref{eq:appendix:cellH1} are $0$ and $3$.
\end{proof}

\begin{proposition}[Stone lift preserves $\mathbf{16}$-cohomology]\label{prop:appendix:stone-lift}
The profinite Stone lift of Theorem~\ref{thm:foundations:stone-lift}, applied
to $\mathcal{T}_7$ with partition $1|10|16$, identifies
\begin{equation}\label{eq:appendix:lift}
  H^1(\mathcal{T}_7,\mathcal{F}_{16})^{\sigma=-1}
  \;\cong\;
  H^1(Q,V_{16})^{\sigma=-1}.
\end{equation}
\end{proposition}

\begin{proof}
The lift is a colimit of refinements preserving clopen cells and the
$\sigma$-action. Cohomology of the cellular cosheaf is the $E_2$ page of the
Leray spectral sequence for the profinite limit; because each refinement
splits only within fixed sector targets $(1/27,10/27,16/27)$, the
$\sigma$-anti-invariant $H^1$ dimension is stable at $3$
(Lemma~\ref{lem:appendix:cellular}).
\end{proof}

\begin{theorem}[Mirror-quotient index theorem]\label{thm:appendix:mirror-index}
Let $\bar\partial_{V_{16}}$ be the Dolbeault operator on $(Q,V_{16})$ and
$\sigma$ the antiholomorphic lift of $\Mirror$ on $Q$
(Definition~\ref{def:appendix:Qhat}). The equivariant index
\begin{equation}\label{eq:appendix:index}
  \mathrm{ind}_\sigma(\bar\partial_{V_{16}})
  \;=\;
  \mathrm{Tr}_{H^1}(\sigma|_{V_{16}})
\end{equation}
equals $3$. Moreover,
\begin{equation}\label{eq:appendix:h1}
  h^1(\hat Q,V_{16})
  \;=\;
  \dim H^1(Q,V_{16})^{\sigma=-1}
  \;=\; 3.
\end{equation}
\end{theorem}

\begin{proof}
By the Atiyah--Bott holomorphic Lefschetz formula~\cite{AtiyahBott1966}, the
equivariant index of $\bar\partial_{V_{16}}$ on the compact $Q$ is the
$\sigma$-trace on $H^\bullet(Q,V_{16})$. The product involution has no
$\sigma$-fixed holomorphic $1$-forms in $V_{16}$ on the diagonal
$\E{8}\hookrightarrow\E{8}\times\E{8}$ because horizontal family modes are
$\sigma$-anti-invariant (Definition~\ref{def:sm:horizontal}), while
$\sigma$-invariant modes would pair $\mathbf{16}$ with $\overline{\mathbf{16}}$
and are excluded by Postulate~\ref{post:sm:P2}. The only contributions are the
three $\sigma$-anti-invariant classes of
Lemma~\ref{lem:appendix:cellular}, lifted to $Q$ by
Proposition~\ref{prop:appendix:stone-lift}. For a $\mathbb{Z}_2$ quotient by an
antiholomorphic involution, $H^1(\hat Q,V_{16})$ is the $\sigma=-1$ eigenspace
of $H^1(Q,V_{16})$~\cite{AtiyahBott1966}, giving \eqref{eq:appendix:h1}.
\end{proof}% --- end 10-appendix.tex ---
